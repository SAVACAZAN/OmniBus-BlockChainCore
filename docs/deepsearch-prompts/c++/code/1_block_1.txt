cmake_minimum_required(VERSION 3.20)
project(omnibus-node-cpp VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Options
option(BUILD_TESTS "Build unit tests" ON)
option(USE_OPENSSL "Use OpenSSL for crypto" ON)

# Dependencies
find_package(PkgConfig REQUIRED)
pkg_check_modules(LIBSECP256K1 REQUIRED libsecp256k1)
pkg_check_modules(LIBOQS REQUIRED liboqs)
find_package(Boost REQUIRED COMPONENTS system asio)
find_package(nlohmann_json REQUIRED)
find_package(spdlog REQUIRED)
find_package(Threads REQUIRED)

if(USE_OPENSSL)
    find_package(OpenSSL REQUIRED)
    target_compile_definitions(omnibus_node PRIVATE USE_OPENSSL)
endif()

if(BUILD_TESTS)
    find_package(Catch2 REQUIRED)
    enable_testing()
endif()

# Main executable
add_executable(omnibus-node
    src/omnibus-node.cpp
    # Additional .cpp files if not header-only
)

target_include_directories(omnibus-node PRIVATE include)
target_link_libraries(omnibus-node
    PRIVATE
    ${LIBSECP256K1_LIBRARIES}
    ${LIBOQS_LIBRARIES}
    Boost::system
    Boost::asio
    nlohmann_json::nlohmann_json
    spdlog::spdlog
    Threads::Threads
)
if(USE_OPENSSL)
    target_link_libraries(omnibus-node PRIVATE OpenSSL::Crypto)
endif()
target_compile_options(omnibus-node PRIVATE -Wall -Wextra -Wpedantic)

# CLI tool
add_executable(omnibus-cli apps/omnibus-cli.cpp)
target_link_libraries(omnibus-cli PRIVATE omnibus_node_lib)

# Tests
if(BUILD_TESTS)
    add_executable(omnibus-tests tests/test_vectors.cpp)
    target_link_libraries(omnibus-tests PRIVATE Catch2::Catch2 omnibus_node_lib)
    include(CTest)
    include(Catch)
    catch_discover_tests(omnibus-tests)
endif()

# Installation
install(TARGETS omnibus-node omnibus-cli RUNTIME DESTINATION bin)
