1#!/usr/bin/env python3
2"""
3franchise_client.py — Reference Python client for OmniBus L1.
4
5The "franchise" model: the chain is public infrastructure (consensus,
6P2P, RPC, REST exchange, faucet, WebSocket events). Anyone can plug
7in with any client they like — Python, Node.js, Rust, browser, mobile,
8even a spreadsheet. The chain doesn't care who you are; it enforces
9the same on-chain rules for everyone (signatures, balances, nonces,
10fee schedule, paper-vs-real isolation).
11
12This script demonstrates the full happy path of a "franchisee":
13
14 1. Connect to a public RPC endpoint
15 2. Probe the chain (block height, peer count, status)
16 3. Claim faucet (testnet only, gives free OMNI)
17 4. Watch balance grow as the chain mines
18 5. Place a tiny order on the in-chain DEX (paper mode by default)
19 6. Cancel it after a few seconds
20 7. Print a summary
21
22Everything goes through the public RPC. There's no shared library,
23no chain-side hook, no internal state. If you can speak HTTP+JSON,
24you can build a franchisee.
25
26Run:
27
28 python3 franchise_client.py [--endpoint URL] [--mnemonic "..."]
29
30Defaults:
31
32 endpoint = https://omnibusblockchain.cc:8443/api-testnet/
33 mnemonic = generated fresh + saved to .franchise_mnemonic.txt
34
35Requires:
36
37 pip install requests
38"""

60class OmniBusClient:
61 """
62 Tiny JSON-RPC 2.0 client over the public OmniBus endpoint.
63
64 No retries, no async, no websockets. Just HTTP POST with JSON body.
65 The point of this file is to show how thin the surface area is —
66 you can re-implement it in any language in ~30 lines.
67 """
68
69 def __init__(self, endpoint: str, timeout_s: float = 10.0):
70 self.endpoint = endpoint.rstrip("/") + "/"
71 self.timeout_s = timeout_s
72 self._id = 0
73
74 def call(self, method: str, params: list[Any] | None = None) -> Any:
75 self._id += 1
76 body = {
77 "jsonrpc": "2.0",
78 "id": self._id,
79 "method": method,
80 "params": params or [],
81 }
82 r = requests.post(
83 self.endpoint,
84 json=body,
85 timeout=self.timeout_s,
86 headers={"Content-Type": "application/json"},
87 )
88 r.raise_for_status()
89 resp = r.json()
90 if "error" in resp:
91 raise RuntimeError(f"RPC error on {method}: {resp['error']}")
92 return resp.get("result")

104def step_probe_chain(c: OmniBusClient) -> dict:
105 banner("Step 1 · Probe the chain")
106 height = c.call("getblockcount")
107 print(f" current height : {height}")
108 try:
109 status = c.call("getstatus")
110 print(f" chain id : {status.get('chain') or status.get('chain_id')}")
111 print(f" difficulty : {status.get('difficulty')}")
112 except Exception as e:
113 # `getstatus` may have a different name on some deployments;
114 # don't abort, just keep going.
115 print(f" getstatus skipped : {e}")
116 return {"height": height}

119def step_load_or_create_mnemonic(path: Path) -> str:
120 banner("Step 2 · Load or create a wallet mnemonic")
121 if path.exists():
122 m = path.read_text().strip()
123 print(f" loaded mnemonic from : {path}")
124 else:
125 # Demo-only mnemonic. Real wallets MUST use a vetted BIP-39 generator
126 # with proper entropy + checksum. This 24-byte hex is enough to call
127 # the faucet on testnet, nothing more.
128 m = secrets.token_hex(24)
129 path.write_text(m)
130 print(f" generated fresh and saved to {path}")
131 print(f" mnemonic (first 16ch): {m[:16]}…")
132 return m

135def step_claim_faucet(c: OmniBusClient, address: str) -> None:
136 banner("Step 3 · Claim faucet (testnet only)")
137 print(f" address : {address}")
138 try:
139 result = c.call("claimfaucet", [{"address": address}])
140 print(f" faucet response : {json.dumps(result)[:160]}")
141 except Exception as e:
142 # Faucet may rate-limit or refuse if you've claimed already
143 # within the cooldown window. That's fine for the demo.
144 print(f" faucet refused : {e}")

147def step_watch_balance(c: OmniBusClient, address: str, seconds: int = 30) -> int:
148 banner(f"Step 4 · Watch balance for {seconds}s")
149 deadline = time.time() + seconds
150 last_seen = -1
151 while time.time() < deadline:
152 try:
153 bal = c.call("getaddressbalance", [{"address": address}])
154 cur = bal if isinstance(bal, int) else bal.get("balance", 0)
155 except Exception as e:
156 cur = -1
157 print(f" getaddressbalance error: {e}")
158 if cur != last_seen:
159 print(f" t={int(time.time() % 10000)} balance = {cur} sat")
160 last_seen = cur
161 time.sleep(2)
162 return last_seen

165def step_place_paper_order(
166 c: OmniBusClient, address: str, mnemonic: str
167) -> dict | None:
168 banner("Step 5 · Place a paper order on the DEX (demo)")
169 # Paper-mode order. Server enforces real signatures + nonce + balance
170 # checks even for paper orders, but the matched fills don't affect
171 # real OMNI balances — separate orderbook entirely.
172 #
173 # On a real client you'd derive the keypair from the mnemonic, sign
174 # a EXCHANGE_PLACE_V1 payload, and submit. To keep this demo small
175 # we don't ship the secp256k1 + bech32 dependencies inline; instead
176 # we just call the public read endpoint to show that orderbook
177 # access works, and print the URL of the place endpoint for the
178 # operator to try with curl or a real wallet client.
179 print(" reading public orderbook for OMNI/USDC (pair_id=0):")
180 try:
181 depth = c.call("exchangeGetDepth", [{"pairId": 0, "mode": "paper"}])
182 print(f" bids : {len(depth.get('bids', []))}")
183 print(f" asks : {len(depth.get('asks', []))}")
184 except Exception as e:
185 print(f" depth call error : {e}")
186 print()
187 print(" to place a real signed order, use a wallet client that")
188 print(" knows secp256k1 + EXCHANGE_PLACE_V1 signing. Example:")
189 print(" POST /exchange/0/private/AddOrder")
190 print(" body: {pair, type, ordertype, price, volume, signature, ...}")
191 return None
