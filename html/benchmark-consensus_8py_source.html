30class ConsensusBenchmark:
31 def __init__(self, txs_per_block: int = 2000, num_blocks: int = 10):
32 self.txs_per_block = txs_per_block
33 self.num_blocks = num_blocks
34 self.results: Dict[str, Any] = {}

36 def _generate_block(self, block_id: int) -> Dict[str, Any]:
37 txs = []
38 for i in range(self.txs_per_block):
39 txs.append({
40 "txid": f"tx-{block_id}-{i}",
41 "inputs": [{"prev": "00" * 32, "vout": 0}],
42 "outputs": [{"addr": f"om1{block_id:04x}{i:04x}", "value": random.randint(1000, 100000)}],
43 })
44 return {"header": {"height": block_id, "prev_hash": "00" * 32, "merkle_root": "ff" * 32}, "txs": txs}
45
46 def benchmark_block_validation(self) -> Dict[str, Any]:
47 cprint(YELLOW, "\n--- Block Validation Benchmark ---")
48 times_ms: List[float] = []
49 for b in range(self.num_blocks):
50 block = self._generate_block(b)
51 t0 = time.perf_counter()
52 # Simulate validation: verify merkle root and tx count
53 _ = len(block["txs"])
54 _ = block["header"]["merkle_root"]
55 time.sleep(0.001) # simulate 1ms base validation
56 t1 = time.perf_counter()
57 times_ms.append((t1 - t0) * 1000)
58
59 avg_ms = sum(times_ms) / len(times_ms)
60 self.results["block_validation_ms"] = {"avg": avg_ms, "min": min(times_ms), "max": max(times_ms)}
61 cprint(GREEN, f"Avg block validation: {avg_ms:.2f} ms")
62 return self.results["block_validation_ms"]
63
64 def benchmark_sub_block_latency(self) -> Dict[str, Any]:
65 cprint(YELLOW, "\n--- Sub-Block Latency Benchmark ---")
66 # Simulate 4 shards processing sub-blocks
67 latencies_ms: List[float] = []
68 for shard in range(4):
69 t0 = time.perf_counter()
70 time.sleep(0.0005) # simulate 0.5ms sub-block processing
71 t1 = time.perf_counter()
72 latencies_ms.append((t1 - t0) * 1000)
73
74 avg_ms = sum(latencies_ms) / len(latencies_ms)
75 self.results["sub_block_latency_ms"] = {"avg": avg_ms, "min": min(latencies_ms), "max": max(latencies_ms)}
76 cprint(GREEN, f"Avg sub-block latency: {avg_ms:.2f} ms")
77 return self.results["sub_block_latency_ms"]
78
79 def benchmark_tps(self) -> Dict[str, Any]:
80 cprint(YELLOW, "\n--- TPS Benchmark ---")
81 # Simulate processing a batch of transactions as fast as possible
82 batch_size = 10_000
83 t0 = time.perf_counter()
84 processed = 0
85 for _ in range(batch_size):
86 processed += 1
87 t1 = time.perf_counter()
88 elapsed = t1 - t0
89 tps = batch_size / elapsed if elapsed > 0 else 0
90 self.results["tps"] = {"throughput": round(tps, 2), "batch_size": batch_size, "elapsed_sec": round(elapsed, 4)}
91 cprint(GREEN, f"Throughput: {tps:,.0f} TPS")
92 return self.results["tps"]
