109def load_address_from_env_file() -> str | None:
110 """Read miner address from the env file. Tolerant to PowerShell's
111 UTF-16-with-BOM default — `echo > file` on Windows produces
112 UTF-16-LE with a 0xFF 0xFE prefix, which breaks naive UTF-8 reads.
113 Try utf-8 first, then utf-16, then latin-1 as a last resort."""
114 print(f"{C.DIM}[DEBUG] load_address_from_env_file: ENV_FILE = {ENV_FILE}{C.RESET}")
115 print(f"{C.DIM}[DEBUG] is_file() = {ENV_FILE.is_file()}{C.RESET}")
116 if not ENV_FILE.is_file():
117 print(f"{C.DIM}[DEBUG] env file does not exist, returning None{C.RESET}")
118 return None
119 raw_bytes = ENV_FILE.read_bytes()
120 print(f"{C.DIM}[DEBUG] read {len(raw_bytes)} bytes; first 8: {raw_bytes[:8]!r}{C.RESET}")
121 # Strip BOMs / try a few encodings in order.
122 text: str | None = None
123 used_enc: str = ""
124 for enc in ("utf-8-sig", "utf-16", "utf-8", "latin-1"):
125 try:
126 text = raw_bytes.decode(enc)
127 used_enc = enc
128 break
129 except UnicodeDecodeError as e:
130 print(f"{C.DIM}[DEBUG] decode {enc} failed: {e}{C.RESET}")
131 continue
132 if text is None:
133 print(f"{C.ERR}[DEBUG] NO encoding worked!{C.RESET}")
134 return None
135 print(f"{C.DIM}[DEBUG] decoded with {used_enc}; text repr: {text!r}{C.RESET}")
136 for line in text.splitlines():
137 line = line.strip().lstrip("﻿") # extra BOM safety
138 print(f"{C.DIM}[DEBUG] line: {line!r}{C.RESET}")
139 if line.startswith("OMNIBUS_MINER_ADDRESS="):
140 addr = line.split("=", 1)[1].strip().strip('"').strip("'")
141 if addr:
142 print(f"{C.DIM}[DEBUG] found address: {addr!r}{C.RESET}")
143 return addr
144 print(f"{C.DIM}[DEBUG] no OMNIBUS_MINER_ADDRESS line found{C.RESET}")
145 return None

148def resolve_miner_address(cli_value: str | None) -> str:
149 # Priority: CLI flag > env file > OMNIBUS_MINER_ADDRESS env var.
150 addr = cli_value
151 if not addr:
152 addr = load_address_from_env_file()
153 if not addr:
154 addr = os.environ.get("OMNIBUS_MINER_ADDRESS")
155 if not addr:
156 print(f"{C.ERR}[ERROR]{C.RESET} No miner reward address provided.")
157 print()
158 print("Provide one of:")
159 print(f" 1. CLI flag: --miner-address ob1q...")
160 print(f" 2. Env file: {ENV_FILE}")
161 print(f" Contents: OMNIBUS_MINER_ADDRESS=ob1q...")
162 print(f" 3. Env var: OMNIBUS_MINER_ADDRESS=ob1q... python {Path(__file__).name}")

185def colorize_log_line(line: str) -> str:
186 if "[REWARD]" in line:
187 return f"{C.OK}{line}{C.RESET}"
188 if "[SLOT]" in line and "Not my turn" in line:
189 return f"{C.DIM}{line}{C.RESET}"
190 if "[SLOT]" in line and "Reject" in line:
191 return f"{C.ERR}{line}{C.RESET}"
192 if "[IBD]" in line:
193 return f"{C.INFO}{line}{C.RESET}"
194 if "[FORK-RECOVERY]" in line or "panic" in line.lower() or "Segfault" in line:
195 return f"{C.ERR}{C.BOLD}{line}{C.RESET}"
196 if "[MINER] Reward address" in line:
197 return f"{C.OK}{C.BOLD}{line}{C.RESET}"
198 if "[P2P] Connected to peer" in line or "[IBD] Exited" in line:
199 return f"{C.OK}{line}{C.RESET}"
200 if "WARN" in line or "[CHECKPOINT]" in line:
201 return f"{C.WARN}{line}{C.RESET}"
202 return line

252 def tail_log():
253 try:
254 with open(log_path, "rb") as f:
255 f.seek(0, 2)
256 while not stop_event.is_set():
257 chunk = f.readline()
258 if not chunk:
259 time.sleep(0.1)
260 continue
261 try:
262 text = chunk.decode("utf-8", errors="replace")
263 except Exception:
264 text = repr(chunk) + "\n"
265 sys.stdout.write(colorize_log_line(text))
266 sys.stdout.flush()
267 except Exception as e:
268 sys.stdout.write(f"{C.ERR}[tail] {e}{C.RESET}\n")

273 def on_sigint(_sig, _frame):
274 print(f"\n{C.WARN}[CTRL-C]{C.RESET} stopping miner (PID {proc.pid})...")
275 stop_event.set()
276 try:
277 proc.terminate()
278 proc.wait(timeout=5)
279 except subprocess.TimeoutExpired:
280 print(f"{C.ERR}timeout, killing{C.RESET}")
281 proc.kill()
282 try: log_fh.close()
283 except: pass
284 sys.exit(0)
285
286 signal.signal(signal.SIGINT, on_sigint)
287 if hasattr(signal, "SIGBREAK"):
288 signal.signal(signal.SIGBREAK, on_sigint)
289
290 try:
291 rc = proc.wait()
292 except KeyboardInterrupt:
293 on_sigint(None, None)
294 finally:
295 stop_event.set()
296 try: log_fh.close()
297 except: pass
298 rc = proc.poll() if proc.poll() is not None else 0
299 print(f"\n{C.DIM}miner exited with code {rc}{C.RESET}")
