2"""
3double-spend-tester.py
4
5Test double-spend protection on a BlockChainCore network:
6 1. Create two conflicting transactions spending the same input
7 to two different recipients.
8 2. Send tx-A to node group A and tx-B to node group B
9 simultaneously (or as fast as possible).
10 3. Wait for block inclusion / confirmation.
11 4. Verify that only one transaction survives consensus;
12 the other is evicted from mempool or never confirmed.
13
14Outputs: double-spend-report.json
15"""

54def rpc(url: str, method: str, params: Any = None, auth: tuple[str, str] | None = None, timeout: float = 10.0) -> Any:
55 payload = {"jsonrpc": "2.0", "method": method, "id": random.randint(1, 100000)}
56 if params is not None:
57 payload["params"] = params
58 try:
59 resp = requests.post(url, json=payload, auth=auth, timeout=timeout)
60 data = resp.json()
61 return data.get("result")
62 except Exception as exc:
63 return {"_error": str(exc)}

101 auth = (args.user, args.password) if args.user or args.password else None

103 log_info("Preparing double-spend test …")

105 # Get an unspent output from node A wallet (or use a dummy for offline mode)
106 utxos = rpc(args.node_a, "listunspent", auth=auth)

108 log_warn("Cannot reach node A; running in structural / offline mode")
109 utxos = []

111 if utxos:

113 txid = utxo["txid"]

115 amount = utxo["amount"]

118 vout = 0

122 input_spec = [{"txid": txid, "vout": vout}]
123 output_a = {"addr_A_receive": amount - 0.001}
124 output_b = {"addr_B_receive": amount - 0.001}

126 # Create raw transactions

128 raw_b = create_raw_tx(args.node_a, input_spec, output_b, auth)

138 if hex_a and hex_b:

140 # Simultaneous send (as close as possible)
141 import threading

143 def send_to(url: str, hex_str: str, label: str) -> None:
144 send_results[label] = send_raw_tx(url, hex_str, auth)

146 t1 = threading.Thread(target=send_to, args=(args.node_a, hex_a, "tx_a"))
147 t2 = threading.Thread(target=send_to, args=(args.node_b, hex_b, "tx_b"))
148 t1.start(); t2.start()
149 t1.join(); t2.join()
150
151 log_info(f"Tx-A result: {send_results.get('tx_a')}")
152 log_info(f"Tx-B result: {send_results.get('tx_b')}")

155
156 # Wait for blocks and check inclusion
157 if not isinstance(utxos, dict):
158 log_info(f"Waiting {args.wait_blocks} blocks …")
159 initial_height = rpc(args.node_a, "getblockcount", auth=auth)
160 if not isinstance(initial_height, dict):
161 while True:
162 time.sleep(2)
163 current = rpc(args.node_a, "getblockcount", auth=auth)
164 if isinstance(current, dict):
165 break
166 if current >= initial_height + args.wait_blocks:
167 break
168
169 # Check mempool / chain for surviving tx
170 mempool_a = rpc(args.node_a, "getrawmempool", auth=auth)
171 mempool_b = rpc(args.node_b, "getrawmempool", auth=auth)
172
173 in_mempool_a = False
174 in_mempool_b = False
175 if isinstance(mempool_a, list) and send_results.get("tx_a") in mempool_a:
176 in_mempool_a = True
177 if isinstance(mempool_b, list) and send_results.get("tx_b") in mempool_b:
178 in_mempool_b = True
179
180 # If both are in mempool, that's a failure of isolation
181 both_survived = in_mempool_a and in_mempool_b
182 neither_survived = not in_mempool_a and not in_mempool_b
183
184 if both_survived:

186 elif neither_survived and send_results:
187 log_warn("Neither transaction in mempool — may have been replaced or rejected")
188 else:
189 log_pass("Only one conflicting transaction survived")
190
191 report = {

193 "node_a": args.node_a,
194 "node_b": args.node_b,
195 "utxo": {"txid": txid, "vout": vout, "amount": amount},
196 "tx_a_hex": hex_a,
197 "tx_b_hex": hex_b,
198 "send_results": send_results,
199 "in_mempool_a": in_mempool_a,
200 "in_mempool_b": in_mempool_b,
201 "both_survived": both_survived,
202 "protection_working": not both_survived,
203 }
204
205 out_path = os.path.abspath(args.output)
206 os.makedirs(os.path.dirname(out_path), exist_ok=True)
207 with open(out_path, "w", encoding="utf-8") as fh:
208 json.dump(report, fh, indent=2)
209
210 log_pass(f"Double-spend report written to {out_path}")
