53def rpc(url: str, method: str, params: Any = None, auth: tuple[str, str] | None = None, timeout: float = 10.0) -> Any:
54 payload = {"jsonrpc": "2.0", "method": method, "id": random.randint(1, 100000)}
55 if params is not None:
56 payload["params"] = params
57 try:
58 resp = requests.post(url, json=payload, auth=auth, timeout=timeout)
59 data = resp.json()
60 return data.get("result")
61 except Exception as exc:
62 return {"_error": str(exc)}

65def fetch_block_headers(url: str, count: int, auth: tuple[str, str] | None = None) -> list[dict[str, Any]]:
66 """Fetch the last *count* block headers via RPC."""
67 info = rpc(url, "getblockchaininfo", auth=auth)
68 if isinstance(info, dict) and "_error" in info:
69 return []
70 height = info.get("blocks", 0)
71 headers: list[dict[str, Any]] = []
72 for h in range(max(0, height - count + 1), height + 1):
73 hash_val = rpc(url, "getblockhash", [h], auth=auth)
74 if isinstance(hash_val, dict) and "_error" in hash_val:
75 continue
76 block = rpc(url, "getblock", [hash_val], auth=auth)
77 if isinstance(block, dict) and "_error" in block:
78 continue
79 headers.append(
80 {
81 "height": block.get("height"),
82 "hash": block.get("hash"),
83 "time": block.get("time"),
84 "bits": block.get("bits"),
85 "difficulty": block.get("difficulty"),
86 "previousblockhash": block.get("previousblockhash"),
87 }
88 )
89 return headers

95def analyze_headers(headers: list[dict[str, Any]], target_interval: float = 10.0) -> dict[str, Any]:
96 if len(headers) < 2:
97 return {"error": "Not enough blocks"}
98
99 # Sort by height
100 headers.sort(key=lambda x: x.get("height", 0))
101
102 block_times: list[float] = []
103 difficulties: list[float] = []
104 for i in range(1, len(headers)):
105 dt = headers[i]["time"] - headers[i - 1]["time"]
106 block_times.append(dt)
107 difficulties.append(headers[i].get("difficulty", 0.0))
108
109 avg_block_time = sum(block_times) / len(block_times)
110 min_bt = min(block_times)
111 max_bt = max(block_times)
112 std_bt = math.sqrt(sum((t - avg_block_time) ** 2 for t in block_times) / len(block_times))
113
114 # Difficulty oscillation: variance of difficulty changes
115 diff_changes = [difficulties[i] - difficulties[i - 1] for i in range(1, len(difficulties))]
116 avg_diff_change = sum(diff_changes) / len(diff_changes) if diff_changes else 0.0
117 diff_oscillation = math.sqrt(sum((d - avg_diff_change) ** 2 for d in diff_changes) / len(diff_changes)) if diff_changes else 0.0
118
119 # Fork detection (same height, different hash) - not available from single node
120 # We simulate a metric based on orphan probability
121 fork_guess = sum(1 for t in block_times if t < target_interval / 2) / len(block_times)
122
123 per_block = [
124 {
125 "height": headers[i]["height"],
126 "block_time_sec": round(block_times[i - 1], 2),
127 "difficulty": difficulties[i - 1],
128 }
129 for i in range(1, len(headers))
130 ]
131
132 return {
133 "blocks_analyzed": len(headers),
134 "target_interval_sec": target_interval,
135 "average_block_time_sec": round(avg_block_time, 2),
136 "min_block_time_sec": round(min_bt, 2),
137 "max_block_time_sec": round(max_bt, 2),
138 "stddev_block_time": round(std_bt, 2),
139 "difficulty_oscillation": round(diff_oscillation, 4),
140 "fork_probability_guess": round(fork_guess, 4),
141 "per_block": per_block,
142 }

145def simulate_chain(num_blocks: int = 200, target_interval: float = 10.0) -> list[dict[str, Any]]:
146 """Generate a synthetic chain for offline analysis."""
147 headers: list[dict[str, Any]] = []
148 t = int(time.time()) - num_blocks * target_interval
149 diff = 1.0
150 for h in range(num_blocks):
151 # Variable block time with noise
152 noise = random.gauss(0, target_interval * 0.3)
153 bt = max(1.0, target_interval + noise)
154 t += int(bt)
155 diff *= target_interval / bt # naive difficulty adjustment
156 diff = max(0.1, diff)
157 headers.append(
158 {
159 "height": h,
160 "hash": f"simhash{h:06x}",
161 "time": t,
162 "bits": int(diff * 65535),
163 "difficulty": round(diff, 4),
164 "previousblockhash": f"simhash{h - 1:06x}" if h > 0 else "0" * 64,
165 }
166 )
167 return headers

182
183 if args.simulate:
184 log_info("Running in simulation mode …")
185 headers = simulate_chain(args.blocks)
186 else:
187 auth = (args.user, args.password) if args.user or args.password else None

189 headers = fetch_block_headers(args.rpc_url, args.blocks, auth)
190 if not headers:
191 log_warn("No blocks fetched; falling back to simulation")
192 headers = simulate_chain(args.blocks)
193
194 analysis = analyze_headers(headers)
195
196 report = {
197 "generated_at": datetime.now(timezone.utc).isoformat(),
198 "mode": "simulation" if args.simulate else "rpc",

201

203 os.makedirs(os.path.dirname(out_path), exist_ok=True)
204 with open(out_path, "w", encoding="utf-8") as fh:
