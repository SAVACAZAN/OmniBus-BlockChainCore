79def parse_pe_header(binary_path: str) -> dict:
80 """Parse PE headers for security characteristics."""
81 result = {
82 "format": "unknown",
83 "nx_dep": "unknown",
84 "aslr_pie": "unknown",
85 "high_entropy_aslr": "unknown",
86 "cfg": "unknown",
87 "entry_point": "unknown",
88 }
89 try:
90 with open(binary_path, "rb") as f:
91 # Check MZ header
92 magic = f.read(2)
93 if magic != b"MZ":
94 result["format"] = "not_pe"
95 return result
96 result["format"] = "PE"
97
98 # PE header offset at 0x3C
99 f.seek(0x3C)
100 pe_offset = struct.unpack("<I", f.read(4))[0]
101
102 # PE signature
103 f.seek(pe_offset)
104 pe_sig = f.read(4)
105 if pe_sig != b"PE\x00\x00":
106 result["format"] = "invalid_pe"
107 return result
108
109 # COFF header (20 bytes)
110 machine = struct.unpack("<H", f.read(2))[0]
111 num_sections = struct.unpack("<H", f.read(2))[0]
112 f.read(12) # skip timestamp, symbol table, etc.
113 optional_header_size = struct.unpack("<H", f.read(2))[0]
114 characteristics = struct.unpack("<H", f.read(2))[0]
115
116 result["machine"] = f"0x{machine:04X}"
117 result["num_sections"] = num_sections
118
119 # Optional header
120 if optional_header_size > 0:
121 opt_magic = struct.unpack("<H", f.read(2))[0]
122 is_pe32_plus = opt_magic == 0x20B
123
124 if is_pe32_plus:
125 result["format"] = "PE32+ (64-bit)"
126 f.read(22) # skip to AddressOfEntryPoint
127 entry_rva = struct.unpack("<I", f.read(4))[0] if False else 0
128 # Re-read properly
129 f.seek(pe_offset + 24 + 16) # AddressOfEntryPoint
130 entry_rva = struct.unpack("<I", f.read(4))[0]
131 result["entry_point"] = f"0x{entry_rva:08X}"
132
133 # DllCharacteristics at offset 70 from optional header start
134 f.seek(pe_offset + 24 + 70)
135 dll_chars = struct.unpack("<H", f.read(2))[0]
136 else:
137 result["format"] = "PE32 (32-bit)"
138 f.seek(pe_offset + 24 + 16)
139 entry_rva = struct.unpack("<I", f.read(4))[0]
140 result["entry_point"] = f"0x{entry_rva:08X}"
141 f.seek(pe_offset + 24 + 70)
142 dll_chars = struct.unpack("<H", f.read(2))[0]
143
144 # DllCharacteristics flags
145 IMAGE_DLLCHARACTERISTICS_DYNAMIC_BASE = 0x0040 # ASLR
146 IMAGE_DLLCHARACTERISTICS_HIGH_ENTROPY_VA = 0x0020
147 IMAGE_DLLCHARACTERISTICS_NX_COMPAT = 0x0100 # DEP/NX
148 IMAGE_DLLCHARACTERISTICS_GUARD_CF = 0x4000 # CFG
149
150 result["nx_dep"] = bool(dll_chars & IMAGE_DLLCHARACTERISTICS_NX_COMPAT)
151 result["aslr_pie"] = bool(dll_chars & IMAGE_DLLCHARACTERISTICS_DYNAMIC_BASE)
152 result["high_entropy_aslr"] = bool(dll_chars & IMAGE_DLLCHARACTERISTICS_HIGH_ENTROPY_VA)
153 result["cfg"] = bool(dll_chars & IMAGE_DLLCHARACTERISTICS_GUARD_CF)
154 result["dll_characteristics_raw"] = f"0x{dll_chars:04X}"
155
156 except Exception as e:
157 result["error"] = str(e)
158
159 return result

162def analyze_sections_objdump(binary_path: str) -> list[dict]:
163 """Parse sections via objdump -h."""
164 rc, stdout, stderr = run_cmd(["objdump", "-h", binary_path])
165 if rc != 0:
166 # Try with full path in Git Bash
167 rc, stdout, stderr = run_cmd(["C:/Program Files/Git/usr/bin/objdump.exe", "-h", binary_path])
168 if rc != 0:
169 return [{"error": f"objdump failed: {stderr.strip()}"}]
170
171 sections = []
172 for line in stdout.splitlines():
173 # Match section lines: " 0 .text 0003a4f0 0000000140001000 ..."
174 m = re.match(r"\s+(\d+)\s+(\.\S+)\s+([0-9a-fA-F]+)\s+([0-9a-fA-F]+)\s+([0-9a-fA-F]+)\s+([0-9a-fA-F]+)\s+.*?([\w, ]+)$", line)
175 if m:
176 flags_str = m.group(7).strip()
177 sections.append({
178 "index": int(m.group(1)),
179 "name": m.group(2),
180 "size_hex": m.group(3),
181 "size_bytes": int(m.group(3), 16),
182 "vma": f"0x{m.group(4)}",
183 "file_offset": f"0x{m.group(6)}",
184 "flags": flags_str,
185 })
186
187 # Check for executable stack (unusual .bss or unnamed sections with EXEC)
188 for sec in sections:
189 if "CODE" in sec.get("flags", "") and sec["name"] not in (".text",):
190 sec["warning"] = "Unexpected executable section"
191
192 return sections

195def scan_strings(binary_path: str) -> dict:
196 """Scan binary strings for unsafe C patterns."""
197 rc, stdout, stderr = run_cmd(["strings", binary_path])
198 if rc != 0:
199 rc, stdout, stderr = run_cmd(["C:/Program Files/Git/usr/bin/strings.exe", binary_path])
200 if rc != 0:
201 return {"error": f"strings command failed: {stderr.strip()}"}
202
203 all_strings = stdout.splitlines()
204 findings = {}
205
206 import re as _re
207 for pattern in UNSAFE_PATTERNS:
208 # Use word-boundary matching to avoid false positives (e.g., "widgets" matching "gets")
209 pat_re = _re.compile(r'\b' + _re.escape(pattern) + r'\b', _re.IGNORECASE)
210 matches = [s for s in all_strings if pat_re.search(s)]
211 if matches:
212 findings[pattern] = {
213 "count": len(matches),
214 "severity": "HIGH" if pattern in ("strcpy", "gets", "sprintf") else "MEDIUM",
215 "samples": matches[:5],
216 }
217
218 # Also check for common crypto weaknesses
219 crypto_concerns = []
220 for s in all_strings:
221 sl = s.lower()
222 if "rand()" in sl or "srand(" in sl:
223 crypto_concerns.append(s[:80])
224 if "md5" in sl and "hmac" not in sl:
225 crypto_concerns.append(s[:80])
226
227 return {
228 "total_strings": len(all_strings),
229 "unsafe_patterns_found": findings,
230 "crypto_concerns": crypto_concerns[:10] if crypto_concerns else [],
231 "verdict": "CLEAN" if not findings else "INVESTIGATE",
232 }

235def check_canaries(binary_path: str) -> dict:
236 """Check for stack canary / stack protector usage."""
237 rc, stdout, stderr = run_cmd(["objdump", "-d", binary_path])
238 if rc != 0:
239 rc, stdout, stderr = run_cmd(["C:/Program Files/Git/usr/bin/objdump.exe", "-d", binary_path])
240
241 if rc != 0:
242 # Fallback: search binary for known canary patterns
243 try:
244 with open(binary_path, "rb") as f:
245 data = f.read()
246 has_stack_chk = b"__stack_chk_fail" in data or b"__stack_chk_guard" in data
247 has_gs = b"__security_cookie" in data or b"__GSHandlerCheck" in data
248 return {
249 "stack_canary_linux": has_stack_chk,
250 "stack_canary_windows_gs": has_gs,
251 "method": "binary_scan",
252 "note": "Zig uses SafeStack by default; explicit canaries may not be present",
253 }
254 except Exception as e:
255 return {"error": str(e)}
256
257 # Search disassembly for canary references
258 has_chk_fail = "__stack_chk_fail" in stdout
259 has_security_cookie = "__security_cookie" in stdout or "GSHandler" in stdout
260 canary_count = stdout.count("__stack_chk_fail") + stdout.count("__security_cookie")
261
262 return {
263 "stack_canary_linux": has_chk_fail,
264 "stack_canary_windows_gs": has_security_cookie,
265 "canary_references": canary_count,
266 "method": "disassembly_scan",
267 "note": "Zig relies on bounds checking + SafeStack rather than traditional canaries",
268 }

313 print(f"{BOLD}{'='*60}{RESET}\n")

315 if not os.path.isfile(args.binary):

318 log_info("Skipping binary analysis — no binary to analyze.")

346 log_pass("ASLR/DYNAMIC_BASE enabled")

374 print(f" {sec['name']:12s} {size_kb:10.1f} KB {sec.get('flags', '')}{warning}")
375 else:
376 log_warn("Could not parse sections (objdump not available?)")

411 print(f" {YELLOW}{c}{RESET}")
412
413 # 6. pwntools checksec (optional)
414 pw = try_pwntools_checksec(args.binary)

426 issues += 1
427 if pe.get("aslr_pie") is False:
428 issues += 1
429 if strings_result.get("unsafe_patterns_found"):
430 issues += len([p for p in strings_result["unsafe_patterns_found"].values() if p["severity"] == "HIGH"])
431
432 if issues == 0:
433 print(f"{GREEN}{BOLD} RESULT: Binary appears well-hardened (0 critical issues){RESET}")
434 else:

436 print(f"{BOLD}{'='*60}{RESET}\n")
437
438 report["summary"] = {"critical_issues": issues}
439
440 if args.json:
441 print(json.dumps(report, indent=2))
442
443 if args.output:
444 with open(args.output, "w") as f:
445 json.dump(report, f, indent=2)
446 log_info(f"Report saved to {args.output}")
447
