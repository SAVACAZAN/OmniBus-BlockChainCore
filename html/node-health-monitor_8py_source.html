32def rpc_call(url: str, method: str, params: list = None, timeout: int = 10) -> dict:
33 """Make a JSON-RPC 2.0 call. Uses only stdlib (no requests dependency)."""
34 payload = json.dumps({
35 "jsonrpc": "2.0",
36 "id": 1,
37 "method": method,
38 "params": params or [],
39 }).encode("utf-8")
40
41 req = urllib.request.Request(
42 url,
43 data=payload,
44 headers={"Content-Type": "application/json"},
45 method="POST",
46 )
47 with urllib.request.urlopen(req, timeout=timeout) as resp:
48 return json.loads(resp.read().decode("utf-8"))

52 def __init__(self, host: str, port: int, interval: int = 30, stuck_threshold: int = 3):
53 self.url = f"http://{host}:{port}"

63 def check_health(self) -> dict:
64 """Returns {"ok": bool, "block_height": int|None, "error": str|None}"""
65 try:
66 result = rpc_call(self.url, "getblockcount")
67 if "result" in result:
68 return {"ok": True, "block_height": result["result"], "error": None}
69 elif "error" in result:
70 return {"ok": False, "block_height": None, "error": result["error"].get("message", "RPC error")}
71 else:
72 return {"ok": False, "block_height": None, "error": "Unexpected response"}
73 except urllib.error.URLError as e:
74 return {"ok": False, "block_height": None, "error": f"Connection failed: {e.reason}"}
75 except Exception as e:
76 return {"ok": False, "block_height": None, "error": str(e)}

78 def run(self) -> None:
79 print(f"{BOLD}{CYAN}=== OmniBus Node Health Monitor ==={RESET}")
80 print(f"{DIM}Target: {self.url}{RESET}")
81 print(f"{DIM}Interval: {self.interval}s{RESET}")
82 print(f"{DIM}Stuck threshold: {self.stuck_threshold} checks{RESET}")
83 print()
84
85 while True:
86 self.total_checks += 1
87 status = self.check_health()
88
89 if status["ok"]:
90 self.consecutive_failures = 0
91 self.total_ok += 1
92 height = status["block_height"]
93
94 # Check if block height is advancing
95 if self.last_block_height is not None and height == self.last_block_height:
96 self.same_height_count += 1
97 else:
98 self.same_height_count = 0
99
100 self.last_block_height = height
101
102 if self.same_height_count >= self.stuck_threshold:
103 print(
104 f"{RED}[{timestamp()}] ALERT: Node stuck at block {height} "
105 f"for {self.same_height_count} consecutive checks!{RESET}"
106 )
107 else:
108 advancing = ""
109 if self.same_height_count > 0:
110 advancing = f" {YELLOW}(unchanged x{self.same_height_count}){RESET}"
111 print(
112 f"{GREEN}[{timestamp()}] OK{RESET} "
113 f"block={height}{advancing} "
114 f"{DIM}uptime={self.total_ok}/{self.total_checks}{RESET}"
115 )
116 else:
117 self.consecutive_failures += 1
118 self.same_height_count = 0 # reset since we can't check height
119 print(
120 f"{RED}[{timestamp()}] FAIL ({self.consecutive_failures}x){RESET} "
121 f"{status['error']}"
122 )
123
124 if self.consecutive_failures >= self.stuck_threshold:
125 print(
126 f"{RED}{BOLD}[{timestamp()}] CRITICAL: Node unreachable "
127 f"for {self.consecutive_failures} consecutive checks!{RESET}"
128 )
129
130 try:
131 time.sleep(self.interval)
132 except KeyboardInterrupt:
133 break
134
135 print(f"\n{YELLOW}Monitor stopped. Total checks: {self.total_checks}, "
136 f"OK: {self.total_ok}, Failed: {self.total_checks - self.total_ok}{RESET}")
