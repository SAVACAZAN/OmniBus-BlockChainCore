28def bech32_polymod(values: list) -> int:
29 generator = [0x3b6a57b2, 0x26508e6d, 0x1ea119fa, 0x3d4233dd, 0x2a1462b3]
30 chk = 1

47def validate_bech32(addr: str) -> Tuple[bool, str]:
48 if not (1 <= len(addr) <= 74):
49 return False, "Length out of range"
50 addr = addr.lower()
51 if "1" not in addr:
52 return False, "Missing separator '1'"
53 idx = addr.rindex("1")
54 hrp = addr[:idx]
55 data = addr[idx + 1 :]
56 if not all(c in BECH32_CHARSET for c in data):
57 return False, "Invalid characters in data part"
58 data_values = [BECH32_CHARSET.find(c) for c in data]
59 if not bech32_verify_checksum(hrp, data_values):
60 return False, "Checksum mismatch"
61 return True, "Valid bech32"
