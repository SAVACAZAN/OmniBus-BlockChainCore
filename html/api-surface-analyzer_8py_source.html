40# Patterns
41PUB_FN_RE = re.compile(r'pub\s+(?:export\s+)?fn\s+(\w+)\s*\‍(([^)]*)\‍)')
42PUB_CONST_RE = re.compile(r'pub\s+const\s+(\w+)')
43PUB_VAR_RE = re.compile(r'pub\s+var\s+(\w+)')
44PUB_STRUCT_RE = re.compile(r'pub\s+const\s+(\w+)\s*=\s*(?:extern\s+)?struct')
45PUB_ENUM_RE = re.compile(r'pub\s+const\s+(\w+)\s*=\s*enum')
46IMPORT_RE = re.compile(r'@import\s*\‍(\s*"([^"]+)"\s*\‍)')
47FIELD_ACCESS_RE = re.compile(r'\.(\w+)')

50def analyze_file(filepath: Path) -> dict:
51 """Extract all public symbols from a .zig file."""
52 try:
53 content = filepath.read_text(encoding="utf-8", errors="replace")
54 except Exception as exc:
55 return {"error": str(exc)}
56
57 module = filepath.stem
58
59 pub_fns = []
60 for match in PUB_FN_RE.finditer(content):
61 name = match.group(1)
62 params_raw = match.group(2).strip()
63 # Count params (rough)
64 param_count = len([p for p in params_raw.split(",") if p.strip()]) if params_raw else 0
65 # Find line number
66 line_num = content[:match.start()].count("\n") + 1
67 pub_fns.append({
68 "name": name,
69 "params": param_count,
70 "params_raw": params_raw[:80],
71 "line": line_num,
72 })
73
74 pub_structs = [m.group(1) for m in PUB_STRUCT_RE.finditer(content)]
75 pub_enums = [m.group(1) for m in PUB_ENUM_RE.finditer(content)]
76
77 # pub const that aren't structs/enums
78 all_pub_const = [m.group(1) for m in PUB_CONST_RE.finditer(content)]
79 pub_consts = [c for c in all_pub_const if c not in pub_structs and c not in pub_enums]
80
81 pub_vars = [m.group(1) for m in PUB_VAR_RE.finditer(content)]
82 imports = IMPORT_RE.findall(content)
83
84 return {
85 "module": module,
86 "path": str(filepath),
87 "pub_fns": pub_fns,
88 "pub_structs": pub_structs,
89 "pub_enums": pub_enums,
90 "pub_consts": pub_consts,
91 "pub_vars": pub_vars,
92 "imports": imports,
93 "total_pub_symbols": len(pub_fns) + len(pub_structs) + len(pub_enums) + len(pub_consts) + len(pub_vars),
94 }

97def find_unused_pub_fns(all_results: list) -> dict:
98 """
99 Find pub fn declarations that are never referenced in other modules.
100 A pub fn is 'used' if another file imports the module AND references
101 the function name.
102 """
103 # Build: which modules import which files
104 module_imports = {} # module -> list of imported module names
105 module_content = {} # module -> full file content
106
107 for r in all_results:
108 if "error" in r:
109 continue
110 mod = r["module"]
111 module_imports[mod] = [
112 imp.replace(".zig", "") for imp in r["imports"]
113 ]
114 try:
115 module_content[mod] = Path(r["path"]).read_text(encoding="utf-8", errors="replace")
116 except Exception:
117 module_content[mod] = ""
118
119 unused = defaultdict(list) # module -> [fn_name, ...]
120
121 for r in all_results:
122 if "error" in r:
123 continue
124 mod = r["module"]
125 for fn in r["pub_fns"]:
126 fn_name = fn["name"]
127 # Check if any other module references this function
128 is_used = False
129 for other_mod, content in module_content.items():
130 if other_mod == mod:
131 continue
132 # Check if other_mod imports this module and uses the fn
133 if mod in module_imports.get(other_mod, []):
134 if re.search(r'\.' + re.escape(fn_name) + r'\b', content):
135 is_used = True
136 break
137 # Also check direct name usage (less accurate)
138 if fn_name in content and fn_name not in ("init", "deinit", "main"):
139 is_used = True
140 break
141
142 if not is_used:
143 unused[mod].append(fn_name)
144
145 return dict(unused)

148def print_api_table(all_results: list):
149 """Print grouped API surface."""
150 print(f"\n{BG_BLUE}{WHITE}{BOLD} OmniBus Public API Surface {RESET}\n")
151
152 # Summary table
153 print(f" {BOLD}{'Module':<30s} {'PubFn':>5s} {'Struct':>6s} {'Enum':>4s} {'Const':>5s} {'Var':>3s} {'Total':>5s}{RESET}")
154 print(f" {DIM}{'─' * 75}{RESET}")
155
156 for r in sorted(all_results, key=lambda x: x.get("total_pub_symbols", 0), reverse=True):
157 if "error" in r:
158 continue
159 mod = r["module"]
160 if len(mod) > 30:
161 mod = mod[:27] + "..."
162 print(
163 f" {CYAN}{mod:<30s}{RESET} "
164 f"{len(r['pub_fns']):5d} "
165 f"{len(r['pub_structs']):6d} "
166 f"{len(r['pub_enums']):4d} "
167 f"{len(r['pub_consts']):5d} "
168 f"{len(r['pub_vars']):3d} "
169 f"{BOLD}{r['total_pub_symbols']:5d}{RESET}"
170 )
171
172 print(f" {DIM}{'─' * 75}{RESET}")

175def print_detailed(all_results: list, show_params: bool):
176 """Print detailed function list per module."""
177 print(f"\n{MAGENTA}{BOLD} Detailed Public Functions{RESET}")
178 print(f" {DIM}{'─' * 70}{RESET}")
179
180 for r in sorted(all_results, key=lambda x: x.get("module", "")):
181 if "error" in r or not r["pub_fns"]:
182 continue
183 print(f"\n {CYAN}{BOLD}{r['module']}{RESET} ({len(r['pub_fns'])} pub fn)")
184 for fn in sorted(r["pub_fns"], key=lambda f: f["line"]):
185 params_hint = f" ({fn['params']} params)" if fn["params"] > 0 else "()"
186 line_hint = f" L{fn['line']}"
187 print(f" {GREEN}fn{RESET} {fn['name']}{DIM}{params_hint}{line_hint}{RESET}")

190def print_unused(unused: dict):
191 """Print unused public functions."""
192 if not unused:
193 print(f"\n {GREEN}All public functions appear to be referenced externally.{RESET}")
194 return
195
196 total = sum(len(fns) for fns in unused.values())
197 print(f"\n {YELLOW}{BOLD}Potentially Unused Public Functions: {total}{RESET}")
198 print(f" {DIM}(pub fn not referenced via @import in other modules){RESET}")
199 print(f" {DIM}{'─' * 60}{RESET}")
200
201 for mod in sorted(unused):
202 fns = unused[mod]
203 print(f" {CYAN}{mod}{RESET}:")
204 for fn_name in sorted(fns):
205 print(f" {YELLOW}pub fn {fn_name}{RESET}")
206
207 print(f"\n {DIM}Note: Some may be used via comptime or as callbacks.{RESET}")
