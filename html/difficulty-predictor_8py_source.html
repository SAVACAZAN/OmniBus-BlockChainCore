26def generate_synthetic_series(length: int = 2000) -> List[Dict[str, float]]:
27 import random
28 series = []
29 diff = 1.0
30 for i in range(length):
31 target = 120.0
32 noise = random.gauss(0, 10)
33 block_time = target + noise + math.sin(i / 200.0) * 15
34 diff *= max(0.25, min(4.0, target / max(block_time, 1)))
35 series.append({"height": i, "block_time": block_time, "difficulty": diff})
36 return series

39def prepare_sequences(series: List[Dict[str, float]], seq_len: int = 10) -> tuple:
40 diffs = [s["difficulty"] for s in series]
41 X, y = [], []
42 for i in range(len(diffs) - seq_len):
43 X.append(diffs[i : i + seq_len])
44 y.append(diffs[i + seq_len])
45 return X, y

48def train_lstm(X: List[List[float]], y: List[float]) -> Any:
49 try:
50 import torch
51 import torch.nn as nn
52 except ImportError:
53 cprint(RED, "PyTorch not installed. Using linear regression fallback.")
54 # Fallback: simple linear regression on last value
55 return None
56
57 class LSTMModel(nn.Module):
58 def __init__(self, input_size: int = 1, hidden_size: int = 32, num_layers: int = 1):
59 super().__init__()
60 self.lstm = nn.LSTM(input_size, hidden_size, num_layers, batch_first=True)
61 self.fc = nn.Linear(hidden_size, 1)
62
63 def forward(self, x):
64 out, _ = self.lstm(x)
65 out = self.fc(out[:, -1, :])
66 return out
67
68 Xt = torch.tensor(X, dtype=torch.float32).unsqueeze(-1)
69 yt = torch.tensor(y, dtype=torch.float32).unsqueeze(-1)
70 model = LSTMModel()
71 criterion = nn.MSELoss()
72 optimizer = torch.optim.Adam(model.parameters(), lr=0.01)
73
74 for epoch in range(200):
75 model.train()
76 optimizer.zero_grad()
77 output = model(Xt)
78 loss = criterion(output, yt)
79 loss.backward()
80 optimizer.step()
81 if epoch % 50 == 0:
82 cprint(YELLOW, f"Epoch {epoch}, Loss={loss.item():.6f}")
83
84 return model

87def predict_next(model: Any, last_seq: List[float]) -> float:
88 try:
89 import torch
90 model.eval()
91 with torch.no_grad():
92 x = torch.tensor([last_seq], dtype=torch.float32).unsqueeze(-1)
93 pred = model(x).item()
94 return pred
95 except Exception:
96 # fallback: average of last 3
