49def download_vectors(url: str, cache_path: str) -> dict:
50 """Download and cache test vectors."""
51 if os.path.exists(cache_path):
52 print(f" {GREEN}Using cached vectors: {cache_path}{RESET}")
53 with open(cache_path, "r") as f:
54 return json.load(f)

56 print(f" {CYAN}Downloading from: {url}{RESET}")
57 os.makedirs(os.path.dirname(cache_path), exist_ok=True)

60 req = urllib.request.Request(url, headers={"User-Agent": "OmniBus-BlockChainCore/1.0"})

73def verify_signature_coincurve(pub_key_hex: str, msg_hex: str, sig_hex: str) -> bool:
74 """Verify ECDSA signature using coincurve."""
75 try:
76 msg_bytes = bytes.fromhex(msg_hex)
77 msg_hash = hashlib.sha256(msg_bytes).digest()
78
79 # Wycheproof provides uncompressed public keys (04 || x || y)
80 pk_bytes = bytes.fromhex(pub_key_hex)
81 pk = coincurve.PublicKey(pk_bytes)
82
83 # Wycheproof provides DER-encoded signatures
84 sig_bytes = bytes.fromhex(sig_hex)
85
86 # coincurve.verify expects raw signature, need to convert from DER
87 return pk.verify(sig_bytes, msg_hash, hasher=None)
88 except Exception:
89 return False

92def parse_der_signature(der_hex: str) -> tuple:
93 """Parse DER-encoded ECDSA signature into (r, s) integers."""
94 try:
95 der = bytes.fromhex(der_hex)
96 if len(der) < 8 or der[0] != 0x30:
97 return None, None
98
99 total_len = der[1]
100 pos = 2
101
102 # R
103 if der[pos] != 0x02:
104 return None, None
105 r_len = der[pos + 1]
106 r_bytes = der[pos + 2:pos + 2 + r_len]
107 r = int.from_bytes(r_bytes, "big")
108 pos += 2 + r_len
109
110 # S
111 if pos >= len(der) or der[pos] != 0x02:
112 return None, None
113 s_len = der[pos + 1]
114 s_bytes = der[pos + 2:pos + 2 + s_len]
115 s = int.from_bytes(s_bytes, "big")
116
117 return r, s
118 except Exception:
119 return None, None

122def is_high_s(s: int) -> bool:
123 """Check if S value is high (Bitcoin requires low-S normalization)."""
124 SECP256K1_N = 0xFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFEBAAEDCE6AF48A03BBFD25E8CD0364141
125 return s > SECP256K1_N // 2

128def run_vectors(vectors: dict) -> dict:
129 """Run all test vectors."""
130 stats = {
131 "total": 0,
132 "pass": 0,
133 "fail": 0,
134 "skip": 0,
135 "high_s_detected": 0,
136 "der_edge_cases": 0,
137 "zero_length_msg": 0,
138 "failures": [],
139 }
140
141 if not vectors or "testGroups" not in vectors:
142 print(f"{RED} Invalid vector format{RESET}")
143 return stats
144
145 for group in vectors["testGroups"]:
146 curve = group.get("key", {}).get("curve", "")
147 if curve != "secp256k1":
148 continue
149
150 pub_key_hex = group.get("key", {}).get("uncompressed", "")
151 if not pub_key_hex:
152 # Try alternative key format
153 pub_key_hex = group.get("key", {}).get("wx", "") + group.get("key", {}).get("wy", "")
154
155 for test in group.get("tests", []):
156 stats["total"] += 1
157 tc_id = test.get("tcId", "?")
158 expected = test.get("result", "") # "valid", "invalid", "acceptable"
159 msg_hex = test.get("msg", "")
160 sig_hex = test.get("sig", "")
161 comment = test.get("comment", "")
162 flags = test.get("flags", [])
163
164 # Track special cases
165 if not msg_hex:
166 stats["zero_length_msg"] += 1
167
168 # Parse DER for analysis
169 r, s = parse_der_signature(sig_hex)
170 if r is not None and s is not None:
171 if is_high_s(s):
172 stats["high_s_detected"] += 1
173
174 # Check for DER edge cases
175 if any(f in flags for f in ["BER", "InvalidEncoding", "MissingZero"]):
176 stats["der_edge_cases"] += 1
177
178 if not HAS_COINCURVE:
179 # Without coincurve, we can only do structural analysis
180 if expected == "invalid":
181 # Check if signature is structurally invalid
182 if r is None or s is None:
183 stats["pass"] += 1 # Correctly detected bad DER
184 elif r == 0 or s == 0:
185 stats["pass"] += 1 # Zero r/s should be invalid
186 elif is_high_s(s):
187 stats["pass"] += 1 # High-S should be rejected
188 else:
189 stats["skip"] += 1 # Can't verify without crypto lib
190 else:
191 stats["skip"] += 1
192 continue
193
194 # Full verification with coincurve
195 try:
196 verified = verify_signature_coincurve(pub_key_hex, msg_hex, sig_hex)
197 except Exception:
198 verified = False
199
200 if expected == "valid":
201 if verified:
202 stats["pass"] += 1
203 else:
204 stats["fail"] += 1
205 stats["failures"].append({
206 "tcId": tc_id,
207 "expected": "valid",
208 "got": "rejected",
209 "comment": comment,
210 "flags": flags,
211 })
212 elif expected == "invalid":
213 if not verified:
214 stats["pass"] += 1
215 else:
216 stats["fail"] += 1
217 stats["failures"].append({
218 "tcId": tc_id,
219 "expected": "invalid",
220 "got": "accepted",
221 "comment": comment,
222 "flags": flags,
223 })
224 elif expected == "acceptable":
225 # Both accept and reject are fine
226 stats["pass"] += 1
227
228 if stats["total"] % 50 == 0:
229 print(f" {CYAN}Processed {stats['total']} vectors ...{RESET}")
230
231 return stats
