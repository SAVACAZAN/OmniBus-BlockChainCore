84def analyze_module_coverage():
85 """Analyze which modules implement which features."""
86 core_files = list(CORE.glob("*.zig")) if CORE.exists() else []
87 modules = [f.stem for f in core_files]
88
89 coverage = {
90 "crypto": ["crypto", "secp256k1", "pq_crypto", "ripemd160", "schnorr", "bls_signatures"],
91 "consensus": ["consensus", "finality", "governance", "staking"],
92 "network": ["p2p", "network", "sync", "kademlia_dht", "bootstrap"],
93 "storage": ["storage", "state_trie", "database"],
94 "economic": ["bread_ledger", "ubi_distributor", "vault_engine", "domain_minter"],
95 "scaling": ["shard_coordinator", "metachain", "sub_block", "compact_transaction"],
96 }
97
98 result = {}
99 for category, expected in coverage.items():
100 found = [m for m in expected if m in modules]
101 result[category] = {
102 "expected": len(expected),
103 "found": len(found),
104 "coverage": len(found) / len(expected) * 100 if expected else 0,
105 "modules": found
106 }
107
108 return result
