34def file_has_keyword(filename, keyword):
35 """Check if a .zig file contains a keyword."""
36 path = os.path.join(CORE_DIR, filename)
37 if not os.path.exists(path):
38 return False
39 try:
40 with open(path, "r", encoding="utf-8", errors="ignore") as f:
41 content = f.read()
42 return keyword.lower() in content.lower()
43 except Exception:
44 return False

297def evaluate_item(item):
298 """Evaluate if OmniBus has this component."""
299 name, btc_has, our_file, notes = item
300
301 if our_file is None:
302 # Check notes for hints
303 if "NOT YET" in notes:
304 return "N", our_file
305 if "partial" in notes.lower():
306 return "P", our_file
307 if "EXTRA" in notes:
308 return "+", our_file
309 # Check frontend
310 if "React" in notes or "frontend" in notes:
311 if os.path.exists(FRONTEND_DIR):
312 return "P", "frontend/"
313 return "N", our_file
314
315 if check_file_exists(our_file):
316 if "[EXTRA]" in notes:
317 return "+", our_file
318 return "Y", our_file
319
320 # Check scripts/
321 script_path = os.path.join(SCRIPTS_DIR, our_file)
322 if os.path.exists(script_path):
323 return "P", f"scripts/{our_file}"
324
325 return "N", our_file

328def generate_category_report(cat):
329 """Generate markdown report for one category."""
330 lines = []
331 cat_id = cat["id"]
332 cat_name = cat["name"]
333
334 lines.append(f"# {cat_id}. {cat_name}")
335 lines.append("")
336 lines.append(f"> OmniBus vs Bitcoin — Category {cat_id}/10")
337 lines.append(f"> Generated: {datetime.datetime.now().strftime('%Y-%m-%d %H:%M')}")
338 lines.append("")
339 lines.append("| # | Component | BTC | OMNI | File | Notes |")
340 lines.append("|:-:|-----------|:---:|:----:|------|-------|")
341
342 y_count = 0
343 p_count = 0
344 n_count = 0
345 extra_count = 0
346 total = len(cat["items"])
347
348 for idx, item in enumerate(cat["items"], 1):
349 name, btc_has, our_file_orig, notes = item
350 status, resolved_file = evaluate_item(item)
351
352 btc_col = "Y" if btc_has else "N"
353 file_col = resolved_file or "-"
354
355 global_idx = (cat_id - 1) * 20 + idx
356
357 if status == "Y":
358 y_count += 1
359 elif status == "P":
360 p_count += 1
361 elif status == "+":
362 extra_count += 1
363 y_count += 1 # extras count as having it
364 else:
365 n_count += 1
366
367 lines.append(f"| {global_idx} | {name} | {btc_col} | {status} | {file_col} | {notes} |")
368
369 lines.append("")
370 lines.append("---")
371 lines.append("")
372
373 btc_items = sum(1 for item in cat["items"] if item[1])
374 pct = int((y_count / btc_items * 100)) if btc_items > 0 else 0
375
376 lines.append(f"**BTC has: {btc_items} items**")
377 lines.append(f"**OmniBus: {y_count} implemented, {p_count} partial, {n_count} missing, {extra_count} extras**")
378 lines.append(f"**Score: {pct}%** ({y_count}/{btc_items} BTC features" +
379 (f" + {extra_count} unique extras)" if extra_count else ")"))
380 lines.append("")
381
382 if n_count > 0:
383 lines.append("### Missing (TODO):")
384 for item in cat["items"]:
385 status, _ = evaluate_item(item)
386 if status == "N":
387 lines.append(f"- [ ] {item[0]} — {item[3]}")
388 lines.append("")
389
390 if extra_count > 0:
391 lines.append("### Extras (OmniBus-only):")
392 for item in cat["items"]:
393 status, _ = evaluate_item(item)
394 if status == "+":
395 lines.append(f"- {item[0]} — {item[3]}")
396 lines.append("")
397
398 return lines, {
399 "name": cat_name,
400 "btc_items": btc_items,
401 "implemented": y_count,
402 "partial": p_count,
403 "missing": n_count,
404 "extras": extra_count,
405 "pct": pct,
406 }
