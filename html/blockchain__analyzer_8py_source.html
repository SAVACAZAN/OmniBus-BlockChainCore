249def check_syntax(path: Path) -> Tuple[Optional[bool], str]:
250 """Check Zig syntax using zig ast-check."""
251 try:
252 result = subprocess.run(
253 ["zig", "ast-check", str(path)],
254 capture_output=True, text=True, timeout=15
255 )
256 return result.returncode == 0, result.stderr[:300] if result.stderr else ""
257 except FileNotFoundError:
258 return None, "zig not found in PATH"
259 except subprocess.TimeoutExpired:
260 return None, "ast-check timeout"
261 except Exception as e:
262 return None, str(e)[:100]

265def extract_imports(code: str) -> List[str]:
266 """Extract @import statements."""
267 imports = []
268 for m in re.finditer(r'@import\s*\‍(\s*"([^"]+)"\s*\‍)', code):
269 imp = m.group(1)
270 if imp != "std" and imp != "builtin":
271 imports.append(imp.replace(".zig", ""))
272 return imports

275def count_tests(code: str) -> int:
276 """Count test blocks."""
277 return len(re.findall(r'test\s+"[^"]*"', code))

280def analyze_module(module_path: Path, verbose: bool = False) -> ModuleResult:
281 """Analyze a single Zig module with blockchain-specific checks."""
282 name = module_path.stem
283
284 try:
285 code = module_path.read_text(encoding='utf-8', errors='replace')
286 except Exception as e:
287 return ModuleResult(
288 name=name, path=module_path, status=ModuleStatus.ERROR,
289 findings=[Finding("READ", Severity.FAIL, name, 0, f"Cannot read: {e}")]
290 )
291
292 lines = code.count('\n')
293 non_empty = sum(1 for l in code.splitlines() if l.strip() and not l.strip().startswith('//'))
294
295 # Count functions
296 all_fns = re.findall(r'(?:pub\s+)?(?:export\s+)?fn\s+\w+', code)
297 pub_fns = re.findall(r'pub\s+(?:export\s+)?fn\s+\w+', code)
298 exports = re.findall(r'(?:pub\s+)?export\s+fn\s+\w+', code)
299
300 # Extract imports and tests
301 imports = extract_imports(code)
302 tests = count_tests(code)
303
304 # Check key patterns
305 has_init = bool(re.search(r'(?:pub\s+)?fn\s+init(?:With\w+)?\s*\‍(', code))
306 has_deinit = bool(re.search(r'(?:pub\s+)?fn\s+deinit\s*\‍(', code))
307 has_allocator = bool(re.search(r'allocator|Allocator', code))
308
309 # Stub detection
310 stub_hits = [p.pattern for p in STUB_PATTERNS if p.search(code)]
311
312 # Real indicator counting
313 real_hits = sum(1 for p in BLOCKCHAIN_REAL_INDICATORS if p.search(code))
314
315 # Determine status
316 is_crypto_module = CATEGORY_MAP.get(name) in (ModuleCategory.CRYPTO, ModuleCategory.CONSENSUS)
317 if non_empty < 10:
318 status = ModuleStatus.EMPTY
319 elif stub_hits and real_hits < 4:
320 status = ModuleStatus.STUB
321 elif stub_hits and real_hits >= 4:
322 status = ModuleStatus.PARTIAL
323 elif real_hits >= 5 and len(pub_fns) > 0:
324 status = ModuleStatus.REAL
325 elif is_crypto_module and len(pub_fns) >= 3 and tests > 0:
326 # Crypto/stateless modules don't need init/deinit to be REAL
327 status = ModuleStatus.REAL
328 elif real_hits >= 4 and tests > 3:
329 status = ModuleStatus.REAL
330 elif real_hits >= 4:
331 status = ModuleStatus.PARTIAL
332 else:
333 status = ModuleStatus.PARTIAL
334
335 # Category
336 category = CATEGORY_MAP.get(name, ModuleCategory.UTIL)
337
338 # Security findings
339 findings = []
340 for pattern, sev, rule_id, msg in SECURITY_PATTERNS:
341 for match in pattern.finditer(code):
342 line_num = code[:match.start()].count('\n') + 1
343 findings.append(Finding(
344 rule_id=rule_id, severity=Severity(sev),
345 module=name, line=line_num, message=msg,
346 evidence=match.group(0)[:60]
347 ))
348
349 # Syntax check
350 if module_path.suffix == '.zig':
351 syntax_ok, syntax_err = check_syntax(module_path)
352 if syntax_ok is False:
353 findings.append(Finding(
354 rule_id="SYNTAX", severity=Severity.FAIL,
355 module=name, line=0,
356 message=f"Syntax error: {syntax_err[:100]}",
357 ))
358 elif syntax_ok is True:
359 findings.append(Finding(
360 rule_id="SYNTAX", severity=Severity.PASS,
361 module=name, line=0, message="Syntax OK"
362 ))
363
364 # Check init/deinit pairs for modules with allocators
365 if has_allocator and has_init and not has_deinit:
366 findings.append(Finding(
367 rule_id="SEC-ALLOC-05", severity=Severity.WARN,
368 module=name, line=0,
369 message="Has init() + allocator but no deinit() - potential memory leak"
370 ))
371
372 # Check test coverage
373 if tests == 0 and status in (ModuleStatus.REAL, ModuleStatus.PARTIAL):
374 findings.append(Finding(
375 rule_id="TEST-01", severity=Severity.WARN,
376 module=name, line=0,
377 message="No inline tests found in module"
378 ))
379
380 # Calculate score (0-100)
381 score = 0
382 if status == ModuleStatus.REAL: score += 40
383 elif status == ModuleStatus.PARTIAL: score += 20
384 elif status == ModuleStatus.STUB: score += 5
385
386 if len(all_fns) > 3: score += 5
387 if len(all_fns) > 10: score += 5
388 if len(pub_fns) > 0: score += 5
389 if len(exports) > 0: score += 5
390 if non_empty > 50: score += 5
391 if non_empty > 200: score += 5
392 if tests > 0: score += 10
393 if tests > 3: score += 5
394 if has_init: score += 5
395 if has_deinit: score += 5
396 if has_allocator and has_deinit: score += 5
397
398 # Penalty for failures
399 fail_count = sum(1 for f in findings if f.severity == Severity.FAIL)
400 score -= fail_count * 10
401
402 return ModuleResult(
403 name=name, path=module_path, status=status, category=category,
404 lines=lines, non_empty_lines=non_empty,
405 functions=len(all_fns), pub_functions=len(pub_fns),
406 exports=len(exports), tests=tests,
407 imports=imports, findings=findings,
408 has_init=has_init, has_deinit=has_deinit,
409 has_allocator=has_allocator,
410 score=max(0, min(100, score))
411 )

414def scan_all_modules() -> List[ModuleResult]:
415 """Scan core/, test/, agent/ directories."""
416 results = []
417
418 dirs = [(CORE, "core"), (TEST, "test"), (AGENT, "agent")]
419 for d, label in dirs:
420 if not d.exists():
421 continue
422 zig_files = sorted(d.glob("*.zig"))
423 print(f" Scanning {len(zig_files)} Zig files in {label}/...")
424 for fpath in zig_files:
425 result = analyze_module(fpath)
426 results.append(result)
427
428 return results

472 print(f" {'-' * 76}")

504

507 print(f"\n Average Quality Score: {BOLD}{avg:.1f}%{W}")
508

513

517 print(f" [{f.rule_id}] {f.module}:{f.line} - {f.message}")

525
526 # Modules without tests
527 no_tests = [r for r in results if r.tests == 0 and r.status == ModuleStatus.REAL]
528 if no_tests:
529 print(f"\n {Y}REAL modules without tests:{W}")
530 for r in no_tests:
531 print(f" - {r.name}")
532
533 print(f"\n{SEP}\n")
