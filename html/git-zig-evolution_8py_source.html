56def run_git(args: list[str], cwd: str) -> str:
57 """Run a git command and return stdout. Raises on failure."""
58 result = subprocess.run(
59 ["git"] + args,
60 cwd=cwd,
61 capture_output=True,

71def get_commits(cwd: str, max_commits: int = 200) -> list[dict[str, Any]]:
72 """Return list of commits with hash, date, and message."""
73 fmt = "%H|%aI|%s"
74 out = run_git(
75 ["log", f"--format={fmt}", f"-n{max_commits}", "--", "core/*.zig"],
76 cwd=cwd,
77 )
78 commits: list[dict[str, Any]] = []
79 for line in out.strip().splitlines():
80 if "|" not in line:
81 continue
82 parts = line.split("|", 2)
83 commits.append(
84 {
85 "hash": parts[0],
86 "date": parts[1],
87 "message": parts[2],
88 }
89 )
90 return commits

116def count_loc(source: str) -> int:
117 """Count non-empty, non-comment lines."""
118 lines = 0
119 in_multiline_comment = False
120 for raw in source.splitlines():
121 line = raw.strip()
122 if not line:
123 continue
124 if line.startswith("//"):
125 continue
126 if "/*" in line:
127 if "*/" not in line:
128 in_multiline_comment = True
129 continue
130 if in_multiline_comment:
131 if "*/" in line:
132 in_multiline_comment = False
133 continue
134 lines += 1
135 return lines

138def complexity_score(source: str) -> int:
139 """Rough complexity: number of functions + branching keywords."""
140 funcs = len(re.findall(r"\bfn\s+\w+", source))
141 branches = len(
142 re.findall(
143 r"\b(if|else|switch|while|for|return|break|continue|catch|try)\b",
144 source,
145 )
146 )
147 return funcs + branches

150def has_allocations(source: str) -> bool:
151 """Detect allocator usage patterns in Zig source."""
152 patterns = [
153 r"\ballocator\b",
154 r"\bAllocator\b",
155 r"\bstd\.heap\b",
156 r"\bpage_allocator\b",
157 r"\bGeneralPurposeAllocator\b",
158 r"\bFixedBufferAllocator\b",
159 r"\.alloc\b",
160 r"\.create\b",
161 r"\.realloc\b",
162 r"\.free\b",
163 ]
164 return any(re.search(p, source) for p in patterns)

167def fix_indicator(commit_msg: str) -> bool:
168 """Heuristic: does the commit message indicate a fix?"""
169 msg = commit_msg.lower()
170 keywords = ["fix", "bug", "patch", "repair", "correct", "resolve", "hotfix"]
171 return any(kw in msg for kw in keywords)

177def analyze_evolution(cwd: str, max_commits: int) -> dict[str, Any]:
178 log_info(f"Fetching up to {max_commits} commits touching core/*.zig …")
179 commits = get_commits(cwd, max_commits)
180 if not commits:
181 log_warn("No commits found for core/*.zig")
182 return {"generated_at": datetime.now(timezone.utc).isoformat(), "modules": {}}
183
184 log_info(f"Analyzing {len(commits)} commits …")
185
186 # Per-commit per-module metrics
187 module_history: dict[str, list[dict[str, Any]]] = defaultdict(list)
188 module_fix_count: dict[str, int] = defaultdict(int)
189
190 for idx, commit in enumerate(commits):
191 files = list_zig_files(cwd, commit["hash"])
192 for fpath in files:
193 source = get_file_at_commit(cwd, commit["hash"], fpath)
194 if not source:
195 continue
196 module_name = Path(fpath).stem
197 module_history[module_name].append(
198 {
199 "commit": commit["hash"],
200 "date": commit["date"],
201 "loc": count_loc(source),
202 "complexity": complexity_score(source),
203 "allocations": has_allocations(source),
204 }
205 )
206 if fix_indicator(commit["message"]):
207 module_fix_count[module_name] += 1
208
209 if (idx + 1) % 20 == 0 or idx == len(commits) - 1:
210 log_info(f" processed {idx + 1}/{len(commits)} commits")
211
212 # Aggregate per-module
213 modules: dict[str, Any] = {}
214 for mod, history in module_history.items():
215 history.sort(key=lambda x: x["date"])
216 loc_start = history[0]["loc"] if history else 0
217 loc_end = history[-1]["loc"] if history else 0
218 loc_growth = loc_end - loc_start
219 avg_complexity = sum(h["complexity"] for h in history) / len(history)
220 max_complexity = max(h["complexity"] for h in history)
221
222 # Detect allocation divergence: first snapshot no alloc, later snapshot has alloc
223 alloc_diverged = False
224 first_no_alloc = any(not h["allocations"] for h in history)
225 later_has_alloc = any(h["allocations"] for h in history[1:])
226 if first_no_alloc and later_has_alloc:
227 alloc_diverged = True
228
229 modules[mod] = {
230 "file": f"core/{mod}.zig",
231 "commits_analyzed": len(history),
232 "fix_count": module_fix_count[mod],
233 "loc_start": loc_start,
234 "loc_end": loc_end,
235 "loc_growth": loc_growth,
236 "growth_percent": round((loc_growth / max(loc_start, 1)) * 100, 2),
237 "average_complexity": round(avg_complexity, 2),
238 "max_complexity": max_complexity,
239 "allocation_diverged": alloc_diverged,
240 "history": history,
241 }
242
243 report = {
244 "generated_at": datetime.now(timezone.utc).isoformat(),
245 "commits_scanned": len(commits),
246 "modules": modules,
247 }
248 return report
