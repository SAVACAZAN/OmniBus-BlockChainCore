175def infer_function_description(func_name: str, params: List[Tuple[str, str]],
176 returns: str, module_name: str, body_lines: List[str]) -> str:
177 """Generate a human-readable description from function name, params, and context."""
178
179 # Common patterns
180 name = func_name
181 param_names = [p[0] for p in params if p[0] != 'self']
182
183 # init/deinit
184 if name == "init":

186 if name == "deinit":

188
189 # get* pattern
190 if name.startswith("get"):
191 subject = camel_to_words(name[3:])

374def camel_to_words(name: str) -> str:
375 """Convert camelCase or PascalCase to readable words."""
376 if not name:
377 return ""
378 # Insert space before uppercase letters
379 result = re.sub(r'([A-Z])', r' \1', name)
380 # Handle consecutive uppercase (e.g., "DHT" -> "DHT")
381 result = re.sub(r'([A-Z]+)([A-Z][a-z])', r'\1 \2', result)
382 return result.strip().lower()

416def extract_doc_comments(content: str, start_line: int) -> str:
417 """Extract consecutive /// comments before a line."""
418 lines = content.split('\n')
419 comments = []
420 for i in range(start_line - 2, -1, -1):
421 line = lines[i].strip()
422 if line.startswith('///'):
423 comments.insert(0, line[3:].strip())
424 elif line and not line.startswith('//'):
425 break
426 return '\n'.join(comments)

429def extract_body_lines(content: str, start_line: int, max_lines: int = 15) -> List[str]:
430 """Extract the first N lines of a function body for context."""
431 lines = content.split('\n')
432 body = []
433 brace_depth = 0
434 started = False
435 for i in range(start_line - 1, min(len(lines), start_line + max_lines)):
436 line = lines[i]
437 if '{' in line:
438 brace_depth += line.count('{')
439 started = True
440 if started:
441 body.append(line.strip())
442 if '}' in line:
443 brace_depth -= line.count('}')
444 if brace_depth <= 0 and started:
445 break
446 return body

449def extract_struct_fields(content: str, start_line: int) -> List[Tuple[str, str]]:
450 """Extract fields from a struct definition."""
451 lines = content.split('\n')
452 fields = []
453 brace_depth = 0
454 started = False
455 for i in range(start_line - 1, min(len(lines), start_line + 50)):
456 line = lines[i].strip()
457 if '{' in line:
458 brace_depth += line.count('{')
459 started = True
460 continue
461 if '}' in line:
462 brace_depth -= line.count('}')
463 if brace_depth <= 0 and started:
464 break
465 if started and brace_depth == 1 and ':' in line and not line.startswith('//'):
466 # Match: field_name: Type,
467 m = re.match(r'(\w+)\s*:\s*([^,=]+)', line)
468 if m and not m.group(1).startswith('pub') and m.group(1) != 'fn':
469 fields.append((m.group(1), m.group(2).strip().rstrip(',')))
470 return fields

473def parse_function(line: str) -> Optional[Tuple[str, str, List[Tuple[str, str]]]]:
474 """Parse a function signature."""
475 match = re.match(r'(?:pub\s+)?(?:export\s+)?fn\s+(\w+)\s*\‍(([^)]*)\‍)\s*(!?[^\{]+)?', line)
476 if not match:
477 return None
478 name = match.group(1)
479 params_str = match.group(2).strip()
480 returns = match.group(3).strip() if match.group(3) else "void"
481 params = []
482 if params_str:
483 for param in params_str.split(','):
484 param = param.strip()
485 if ':' in param:
486 pname, ptype = param.split(':', 1)
487 params.append((pname.strip(), ptype.strip()))
488 return name, returns, params

491def parse_struct(line: str) -> Optional[str]:
492 """Parse struct name."""
493 match = re.match(r'(?:pub\s+)?const\s+(\w+)\s*=\s*(?:extern\s+)?struct', line)
494 if match:
495 return match.group(1)
496 return None

499def parse_module(filepath: Path) -> ModuleDoc:
500 """Parse a Zig module and extract documentation with smart descriptions."""
501 content = filepath.read_text(encoding='utf-8', errors='replace')
502 lines = content.split('\n')
503
504 module = ModuleDoc(name=filepath.stem)
505 module.line_count = len(lines)
506 module.test_count = content.count('test "')
507
508 # Module description: use curated if available, otherwise from top comments
509 if filepath.stem in MODULE_DESCRIPTIONS:
510 module.description = MODULE_DESCRIPTIONS[filepath.stem]
511 else:
512 module_desc = []
513 for line in lines:
514 stripped = line.strip()
515 if stripped.startswith('//!'):
516 module_desc.append(stripped[3:].strip())
517 elif stripped and not stripped.startswith('//'):
518 break
519 module.description = '\n'.join(module_desc) if module_desc else f"Module for {camel_to_words(filepath.stem)} functionality."
520
521 for i, line in enumerate(lines, 1):
522 stripped = line.strip()
523
524 # Parse pub functions
525 if re.match(r'pub\s+(?:export\s+)?fn\s+\w+', stripped):
526 func_info = parse_function(stripped)
527 if func_info:
528 name, returns, params = func_info
529 doc = extract_doc_comments(content, i)
530 body = extract_body_lines(content, i)
531
532 # Auto-generate description if none from comments
533 if not doc:
534 doc = infer_function_description(name, params, returns,
535 filepath.stem, body)
536
537 func_doc = FunctionDoc(
538 name=name,
539 signature=stripped,
540 description=doc,
541 params=params,
542 returns=returns,
543 line=i,
544 body_preview=body[:5],
545 )
546 module.functions.append(func_doc)
547
548 # Parse structs
549 if 'struct' in stripped and 'const' in stripped:
550 struct_name = parse_struct(stripped)
551 if struct_name:
552 doc = extract_doc_comments(content, i)
553 fields = extract_struct_fields(content, i)
554
555 if not doc:
556 doc = infer_struct_description(struct_name, fields, filepath.stem)
557
558 module.structs.append(StructDoc(
559 name=struct_name,
560 description=doc,
561 fields=fields,
