1#!/usr/bin/env python3
2"""
3OmniBus BlockChainCore — Node Deployment Tool
4==============================================
5Automates the full deployment of an OmniBus blockchain node:
6
7 1. Verifies Zig compiler is installed and accessible
8 2. Runs ``zig build`` to compile the node binary
9 3. Copies omnibus-node binary to target directory
10 4. Generates a default omnibus.toml configuration file
11 5. Creates a launcher script (systemd .service on Linux, .bat on Windows)
12
13Usage:
14 python deploy-node.py --target-dir /opt/omnibus --mode seed
15 python deploy-node.py --target-dir C:\\omnibus --mode miner --seed-host 10.0.0.1
16"""

68def check_zig() -> str | None:
69 """Return zig version string, or None if not found."""
70 try:
71 result = subprocess.run(["zig", "version"], capture_output=True, text=True, timeout=10)
72 if result.returncode == 0:
73 return result.stdout.strip()
74 except FileNotFoundError:

79def build_node(project_root: Path) -> bool:
80 step("Building node binary with zig build")
81 try:
82 result = subprocess.run(
83 ["zig", "build"],
84 cwd=str(project_root),
85 capture_output=True, text=True, timeout=300,
86 )
87 if result.returncode == 0:
88 ok("Build succeeded")
89 return True
90 else:
91 fail("Build failed")
92 print(f" {DIM}{result.stderr[:500]}{RESET}")
93 return False
94 except subprocess.TimeoutExpired:
95 fail("Build timed out (300s)")
96 return False

110def generate_toml(target_dir: Path, mode: str, node_id: str,
111 port: int, rpc_port: int, seed_host: str, seed_port: int) -> Path:
112 """Generate omnibus.toml config."""

146 """Generate a systemd .service file."""
147 binary_path = target_dir / binary_name
148 service = textwrap.dedent(f"""\

193 formatter_class=argparse.RawDescriptionHelpFormatter,
194 epilog=(
195 "Examples:\n"

201 parser.add_argument("--mode", choices=["seed", "miner"], default="seed",
202 help="Node mode: seed or miner (default: seed)")
203 parser.add_argument("--node-id", default=None, help="Node identifier (default: auto-generated)")
204 parser.add_argument("--port", type=int, default=9000, help="P2P port (default: 9000)")
205 parser.add_argument("--rpc-port", type=int, default=8332, help="RPC port (default: 8332)")
206 parser.add_argument("--seed-host", default="127.0.0.1", help="Seed node host (for miner mode)")
207 parser.add_argument("--seed-port", type=int, default=9000, help="Seed node port (default: 9000)")
208 parser.add_argument("--skip-build", action="store_true", help="Skip zig build step")
209 args = parser.parse_args()
210
211 node_id = args.node_id or f"{args.mode}-{os.getpid()}"
212 target_dir = Path(args.target_dir).resolve()
213 project_root = find_project_root()
214 is_windows = platform.system() == "Windows"
215
216 print(f"\n{CYAN}{BOLD} OmniBus Node Deployment{RESET}")
217 print(f" {DIM}{'─' * 50}{RESET}")
218 print(f" {CYAN}Project root:{RESET} {project_root}")
219 print(f" {CYAN}Target dir:{RESET} {target_dir}")
220 print(f" {CYAN}Mode:{RESET} {args.mode}")

223
224 # Step 1: Check Zig
225 step("Checking Zig compiler")
226 zig_ver = check_zig()
227 if zig_ver:
228 ok(f"Zig {zig_ver}")

242 # Step 3: Find and copy binary

266 ok(f"Windows launcher: {launcher}")

269 ok(f"Systemd service: {launcher}")
270 print(f" {DIM}Install: sudo cp {launcher} /etc/systemd/system/{RESET}")

272

276 print(f" {DIM}To start: {'double-click ' + launcher.name if is_windows else f'systemctl start omnibus-{args.mode}'}{RESET}\n")
