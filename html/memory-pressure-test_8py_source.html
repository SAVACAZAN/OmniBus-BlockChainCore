41def rpc_raw(host: str, port: int, payload: str, timeout: int = 10) -> dict:
42 """Send raw JSON payload to RPC."""
43 try:
44 conn = http.client.HTTPConnection(host, port, timeout=timeout)
45 conn.request("POST", "/", payload, {
46 "Content-Type": "application/json",
47 "Content-Length": str(len(payload)),
48 })
49 resp = conn.getresponse()
50 status = resp.status
51 data = resp.read().decode("utf-8", errors="replace")
52 conn.close()
53 return {"status": status, "body": data[:1000], "crashed": False}
54 except (ConnectionRefusedError, ConnectionResetError, BrokenPipeError):
55 return {"status": 0, "body": "", "crashed": True}
56 except Exception as exc:
57 return {"status": 0, "body": str(exc), "crashed": "refused" in str(exc).lower()}

60def ping_node(host: str, port: int) -> bool:
61 """Verify node is alive with a simple getblockcount."""
62 payload = json.dumps({"jsonrpc": "2.0", "id": 1, "method": "getblockcount", "params": []})
63 try:
64 conn = http.client.HTTPConnection(host, port, timeout=5)
65 conn.request("POST", "/", payload, {"Content-Type": "application/json"})
66 resp = conn.getresponse()
67 resp.read()
68 conn.close()
69 return resp.status in (200, 400, 500) # Any HTTP response = alive
70 except Exception:
71 return False

95def test_long_string(host: str, port: int) -> TestResult:
96 """Send 1MB string parameter."""
97 result = TestResult("long_string_1mb", "1MB string in JSON-RPC params")
98 long_str = "A" * (1024 * 1024) # 1MB
99 payload = json.dumps({
100 "jsonrpc": "2.0", "id": 1,
101 "method": "getblock",
102 "params": [long_str],
103 })
104
105 t0 = time.time()
106 resp = rpc_raw(host, port, payload, timeout=30)
107 result.elapsed_ms = round((time.time() - t0) * 1000)
108 result.response_status = resp["status"]
109 result.response_body = resp["body"]
110
111 if resp["crashed"]:
112 result.error_type = "crash"
113 result.node_alive_after = False
114 elif resp["status"] == 0:
115 result.error_type = "timeout"
116 else:
117 result.error_type = "error_response" # Expected: node rejects gracefully
118
119 # Verify alive
120 time.sleep(0.5)
121 result.node_alive_after = ping_node(host, port)
122 return result
