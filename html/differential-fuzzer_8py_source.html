54def rpc_call(host: str, port: int, method: str, params=None):
55 """Send a JSON-RPC 2.0 request to the OmniBus node."""
56 payload = json.dumps({
57 "jsonrpc": "2.0",
58 "id": 1,
59 "method": method,
60 "params": params or [],
61 })
62 try:
63 conn = http.client.HTTPConnection(host, port, timeout=5)
64 conn.request("POST", "/", payload, {"Content-Type": "application/json"})
65 resp = conn.getresponse()
66 data = json.loads(resp.read().decode())
67 conn.close()
68 return data
69 except Exception as exc:
70 return {"error": str(exc)}

73def python_sha256(data: bytes) -> bytes:
74 return hashlib.sha256(data).digest()

77def python_double_sha256(data: bytes) -> bytes:
78 return hashlib.sha256(hashlib.sha256(data).digest()).digest()

81def python_ripemd160(data: bytes) -> bytes:
82 h = hashlib.new("ripemd160")
83 h.update(data)
84 return h.digest()

87def python_hash160(data: bytes) -> bytes:
88 """SHA-256 then RIPEMD-160 — standard Bitcoin HASH160."""
89 return python_ripemd160(python_sha256(data))

92def generate_private_key() -> bytes:
93 """32-byte random scalar in secp256k1 valid range."""
94 # n = order of secp256k1
95 n = 0xFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFEBAAEDCE6AF48A03BBFD25E8CD0364141
96 while True:
97 key = secrets.token_bytes(32)
98 k = int.from_bytes(key, "big")
99 if 0 < k < n:

103def fuzz_sha256(iterations: int, rpc_host: str) -> dict:
104 """Fuzz SHA-256: compare Python hashlib vs node RPC (if available)."""
105 stats = {"tested": 0, "pass": 0, "fail": 0, "rpc_unavailable": 0}
106 node_available = rpc_call(rpc_host, RPC_PORT, "getblockcount").get("result") is not None
107
108 for i in range(iterations):
109 msg = secrets.token_bytes(secrets.randbelow(512) + 1)
110 py_hash = python_sha256(msg).hex()
111
112 if node_available:
113 resp = rpc_call(rpc_host, RPC_PORT, "sha256", [msg.hex()])
114 if "result" in resp and resp["result"] is not None:
115 node_hash = resp["result"]
116 if node_hash != py_hash:
117 stats["fail"] += 1
118 print(f"{RED}[DIVERGENCE] SHA-256 iteration {i}: "
119 f"python={py_hash[:16]}... node={node_hash[:16]}...{RESET}")
120 else:
121 stats["pass"] += 1
122 else:
123 stats["rpc_unavailable"] += 1
124 else:
125 # Self-consistency: double-hash must match
126 dh1 = python_double_sha256(msg).hex()
127 dh2 = hashlib.sha256(bytes.fromhex(py_hash)).hexdigest()
128 if dh1 != dh2:
129 stats["fail"] += 1
130 else:
131 stats["pass"] += 1
132
133 stats["tested"] += 1
134 if (i + 1) % 1000 == 0:
135 print(f"{CYAN} SHA-256: {i+1}/{iterations} tested ...{RESET}")
136
137 return stats

141 """Fuzz HASH160 (SHA-256 + RIPEMD-160) self-consistency."""
142 stats = {"tested": 0, "pass": 0, "fail": 0}
143 for i in range(iterations):
144 msg = secrets.token_bytes(secrets.randbelow(256) + 1)

156def fuzz_secp256k1(iterations: int) -> dict:
157 """If coincurve available, generate random keys, sign, verify."""
158 stats = {"tested": 0, "pass": 0, "fail": 0, "skipped": 0}
159 if not HAS_COINCURVE:
160 print(f"{YELLOW}[SKIP] coincurve not installed — secp256k1 cross-check disabled{RESET}")
161 stats["skipped"] = iterations
162 return stats
163
164 for i in range(iterations):
165 sk_bytes = generate_private_key()
166 msg = secrets.token_bytes(32)
167 msg_hash = python_sha256(msg)
168
169 try:
170 sk = coincurve.PrivateKey(sk_bytes)
171 sig = sk.sign(msg_hash, hasher=None)
172 pk = sk.public_key
173
174 # Verify must pass
175 if not pk.verify(sig, msg_hash, hasher=None):
176 stats["fail"] += 1
177 print(f"{RED}[DIVERGENCE] secp256k1 verify failed iteration {i}{RESET}")
178 continue
179
180 # Wrong message must fail
181 wrong_msg = secrets.token_bytes(32)
182 if pk.verify(sig, wrong_msg, hasher=None):
183 stats["fail"] += 1
184 print(f"{RED}[DIVERGENCE] secp256k1 wrong-msg verify passed iteration {i}{RESET}")
185 continue
186
187 stats["pass"] += 1
188 except Exception as exc:
189 stats["fail"] += 1
190 print(f"{RED}[ERROR] secp256k1 iteration {i}: {exc}{RESET}")
191
192 stats["tested"] += 1
193 if (i + 1) % 1000 == 0:
194 print(f"{CYAN} secp256k1: {i+1}/{iterations} tested ...{RESET}")
195
196 return stats

218 h1 = python_double_sha256(data)
