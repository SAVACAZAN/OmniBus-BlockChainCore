51BLOCK_TIME_SEC = 10.0
52BLOCK_REWARD = 50.0 # OMNI per block
53HASHRATE_NETWORK_GHPS = 1_000_000 # 1 EH/s = 1e6 GH/s
54COST_PER_GHPS_HOUR = 0.05 # USD
55COST_ECLIPSE_NODE_RENT = 500.0 # USD per malicious peer
56COST_STAKE_PER_OMNI = 1.0 # USD for PoS simulation

62def simulate_51_attack(network_ghps: float, attacker_fraction: float, duration_hours: float) -> dict[str, Any]:
63 """Estimate 51% attack feasibility."""
64 attacker_ghps = network_ghps * attacker_fraction
65 cost = attacker_ghps * COST_PER_GHPS_HOUR * duration_hours
66 expected_blocks = (duration_hours * 3600.0 / BLOCK_TIME_SEC) * attacker_fraction
67 revenue = expected_blocks * BLOCK_REWARD * COST_STAKE_PER_OMNI
68 return {
69 "attack": "51_percent",
70 "attacker_fraction": attacker_fraction,
71 "duration_hours": duration_hours,
72 "attacker_ghps": attacker_ghps,
73 "estimated_cost_usd": round(cost, 2),
74 "expected_blocks": round(expected_blocks, 2),
75 "revenue_usd": round(revenue, 2),
76 "profitable": revenue > cost,
77 }

80def simulate_selfish_mining(network_ghps: float, attacker_fraction: float, rounds: int = 1000) -> dict[str, Any]:
81 """Simple selfish mining simulation (Eyal & Sirer model approximation)."""
82 alpha = attacker_fraction
83 gamma = 0.5 # propagation factor
84 # Revenue approximation from the paper
85 if alpha < 0.5:
86 revenue_ratio = (alpha * (1 - alpha) ** 2 * (4 * alpha + gamma * (1 - 2 * alpha)) - alpha ** 3) / (
87 1 - alpha * (1 + (2 - alpha) * alpha)
88 )
89 else:
90 revenue_ratio = alpha
91
92 # Monte-Carlo style simple block race
93 attacker_wins = 0
94 honest_wins = 0
95 lead = 0
96 for _ in range(rounds):
97 if random.random() < alpha:
98 lead += 1
99 else:
100 if lead == 0:
101 honest_wins += 1
102 else:
103 lead -= 1
104 if lead == 0:
105 attacker_wins += 1
106 total = attacker_wins + honest_wins or 1
107 simulated_ratio = attacker_wins / total
108
109 cost = network_ghps * alpha * COST_PER_GHPS_HOUR * 24 # 24h baseline
110 return {
111 "attack": "selfish_mining",
112 "attacker_fraction": alpha,
113 "rounds_simulated": rounds,
114 "theoretical_revenue_ratio": round(revenue_ratio, 4),
115 "simulated_revenue_ratio": round(simulated_ratio, 4),
116 "estimated_cost_24h_usd": round(cost, 2),
117 }

120def simulate_eclipse_attack(num_peers: int = 8, malicious_peers: int = 8) -> dict[str, Any]:
121 """Estimate eclipse attack cost."""
122 # If attacker controls all outbound peer slots
123 success_probability = 1.0 if malicious_peers >= num_peers else (
124 math.comb(malicious_peers, min(num_peers, malicious_peers)) /
125 max(math.comb(1000, num_peers), 1)
126 )
127 cost = malicious_peers * COST_ECLIPSE_NODE_RENT
128 return {
129 "attack": "eclipse",
130 "target_outbound_peers": num_peers,
131 "malicious_peers": malicious_peers,
132 "success_probability": round(min(success_probability, 1.0), 6),
133 "estimated_cost_usd": round(cost, 2),
134 }

137def simulate_long_range_attack(stake_fraction: float, history_days: int) -> dict[str, Any]:
138 """PoS long-range: attacker with historical stake fraction rewinds chain."""
139 cost = stake_fraction * 1_000_000 * COST_STAKE_PER_OMNI # assume 1M OMNI total stake
140 blocks_to_rewrite = int(history_days * 24 * 3600 / BLOCK_TIME_SEC)
141 return {
142 "attack": "long_range",
143 "stake_fraction": stake_fraction,
144 "history_days": history_days,
145 "blocks_to_rewrite": blocks_to_rewrite,
146 "estimated_cost_usd": round(cost, 2),
147 "feasible": stake_fraction > 0.3,
148 }

151def simulate_nothing_at_stake(validators: int, attacker_validators: int, forks: int = 10) -> dict[str, Any]:
152 """Simulate nothing-at-stake: attacker stakes on multiple forks."""
153 if validators <= 0:
154 validators = 100
155 attacker_fraction = attacker_validators / validators
156 # Expected reward per fork
157 reward_per_fork = BLOCK_REWARD * attacker_fraction
158 total_reward = reward_per_fork * forks
159 # Cost is negligible because same stake is reused
160 return {
161 "attack": "nothing_at_stake",
162 "validators": validators,
163 "attacker_validators": attacker_validators,
164 "forks_bet_on": forks,
165 "attacker_fraction": round(attacker_fraction, 4),
166 "expected_reward_usd": round(total_reward * COST_STAKE_PER_OMNI, 2),
167 "cost_usd": 0.0,
168 "exploit_value": round(total_reward * COST_STAKE_PER_OMNI, 2),
169 }

216 report = {
