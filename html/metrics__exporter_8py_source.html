43def rpc_call(method: str, params: list = None) -> Optional[dict]:
44 """Make RPC call to blockchain node."""
45 try:
46 req = urllib.request.Request(
47 f"http://{RPC_HOST}:{RPC_PORT}",
48 data=json.dumps({
49 "jsonrpc": "2.0",
50 "method": method,
51 "params": params or [],
52 "id": 1
53 }).encode(),
54 headers={"Content-Type": "application/json"},
55 method="POST"
56 )
57 with urllib.request.urlopen(req, timeout=5) as resp:
58 data = json.loads(resp.read().decode())
59 return data.get("result")
60 except Exception as e:
61 return None

63def collect_metrics() -> BlockchainMetrics:
64 """Collect metrics from blockchain node."""
65 metrics = BlockchainMetrics()
66 metrics.timestamp = time.time()
67
68 # Get pool stats
69 pool_stats = rpc_call("getpoolstats")
70 if pool_stats:
71 metrics.block_height = pool_stats.get("blockHeight", 0)
72 metrics.registered_miners = pool_stats.get("registeredMiners", 0)
73 metrics.total_supply = pool_stats.get("totalRewards", 0.0)
74
75 # Get mempool
76 mempool = rpc_call("getmempoolsize")
77 if mempool is not None:
78 metrics.mempool_size = mempool
79
80 # Get miner connections
81 connections = rpc_call("getminerconnections")
82 if connections:
83 metrics.active_miners = connections.get("active", 0)
84
85 # Get transaction count
86 tx_count = rpc_call("gettransactioncount")
87 if tx_count is not None:
88 metrics.total_transactions = tx_count
89
90 return metrics
