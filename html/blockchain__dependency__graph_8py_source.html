113def build_graph() -> Dict[str, ModuleNode]:
114 """Build dependency graph from all Zig files."""
115 graph = {}
116
117 scan_dirs = [(CORE, "core"), (TEST, "test"), (AGENT, "agent")]
118
119 for d, label in scan_dirs:
120 if not d.exists():
121 continue
122 for fpath in sorted(d.glob("*.zig")):
123 name = fpath.stem
124 try:
125 code = fpath.read_text(encoding='utf-8', errors='replace')
126 except Exception:
127 continue
128
129 # Extract imports (only local ones, not std/builtin)
130 imports = []
131 for m in re.finditer(r'@import\s*\‍(\s*"([^"]+)"\s*\‍)', code):
132 imp = m.group(1).replace(".zig", "")
133 if imp not in ("std", "builtin"):
134 imports.append(imp)
135
136 lines = code.count('\n')
137 pub_fns = len(re.findall(r'pub\s+(?:export\s+)?fn\s+\w+', code))
138 tests = len(re.findall(r'test\s+"[^"]*"', code))
139
140 graph[name] = ModuleNode(
141 name=name, path=str(fpath),
142 imports=imports, layer=get_layer(name),
143 lines=lines, pub_functions=pub_fns, tests=tests,
144 )
145
146 # Build reverse edges (imported_by)
147 for name, node in graph.items():
148 for imp in node.imports:
149 if imp in graph:
150 graph[imp].imported_by.append(name)
151
152 return graph

159def find_cycles(graph: Dict[str, ModuleNode]) -> List[CycleInfo]:
160 """Find circular dependencies using DFS."""
161 cycles = []
162 visited = set()
163 path = []
164 path_set = set()
165
166 def dfs(node_name):
167 if node_name in path_set:
168 # Found cycle
169 idx = path.index(node_name)
170 cycle = path[idx:] + [node_name]
171 cycles.append(CycleInfo(modules=cycle, length=len(cycle) - 1))
172 return
173 if node_name in visited:
174 return
175
176 visited.add(node_name)
177 path.append(node_name)
178 path_set.add(node_name)
179
180 node = graph.get(node_name)
181 if node:
182 for imp in node.imports:
183 if imp in graph:
184 dfs(imp)
185
186 path.pop()
187 path_set.discard(node_name)
188
189 for name in graph:
190 dfs(name)
191
192 # Deduplicate cycles
193 seen = set()
194 unique = []
195 for c in cycles:
196 key = tuple(sorted(c.modules[:-1]))
197 if key not in seen:
198 seen.add(key)
199 unique.append(c)
200
201 return unique

204def find_zombies(graph: Dict[str, ModuleNode]) -> List[str]:
205 """Find isolated modules (no imports, not imported by anyone)."""
206 zombies = []
207 for name, node in graph.items():
208 if name == "main":

215def find_hubs(graph: Dict[str, ModuleNode], threshold: int = 5) -> List[Tuple[str, int, int]]:
216 """Find hub modules (high in-degree or out-degree)."""
217 hubs = []
218 for name, node in graph.items():
219 in_deg = len(node.imported_by)
220 out_deg = len(node.imports)
221 if in_deg >= threshold or out_deg >= threshold:
222 hubs.append((name, in_deg, out_deg))
223 return sorted(hubs, key=lambda x: -(x[1] + x[2]))

226def calc_coupling(graph: Dict[str, ModuleNode]) -> Dict[str, float]:
227 """Calculate coupling score per module (0=isolated, 1=maximum coupling)."""
228 n = len(graph)
229 if n <= 1:
230 return {}
231 coupling = {}
232 for name, node in graph.items():
233 total_connections = len(set(node.imports) | set(node.imported_by))
234 coupling[name] = round(total_connections / (n - 1), 3)
235 return coupling

238def find_layer_violations(graph: Dict[str, ModuleNode]) -> List[Tuple[str, str, int, int]]:
239 """Find imports that go UP in layers (violations)."""
240 violations = []
241 for name, node in graph.items():
242 for imp in node.imports:
243 if imp in graph:
244 my_layer = node.layer
245 imp_layer = graph[imp].layer
246 # Only flag when lower layer imports higher layer
247 if imp_layer > my_layer and my_layer != 99 and imp_layer != 99:
248 violations.append((name, imp, my_layer, imp_layer))
249 return violations

385def generate_dot(graph: Dict[str, ModuleNode], path: Path):
386 """Generate Graphviz DOT file."""
387 dot_colors = {
388 0: "#c084fc", # Purple - Crypto
389 1: "#67e8f9", # Cyan - Types
390 2: "#86efac", # Green - Core
391 3: "#93c5fd", # Blue - Network
392 4: "#fde047", # Yellow - Storage
393 5: "#e5e7eb", # Gray - Node
394 6: "#fdba74", # Orange - Features
395 7: "#5eead4", # Teal - Economic
396 8: "#fca5a5", # Red - Entry
397 }
398
399 lines = ['digraph BlockchainDeps {']
400 lines.append(' rankdir=TB;')
401 lines.append(' node [shape=box, style="filled,rounded", fontname="Consolas"];')
402 lines.append(' edge [color="#666666"];')
403 lines.append('')
404
405 # Create subgraphs per layer
406 for layer_num in sorted(LAYER_MAP.keys()):
407 layer_name, _ = LAYER_MAP[layer_num]
408 layer_modules = [n for n in graph.values() if n.layer == layer_num]
409 if not layer_modules:
410 continue
411
412 color = dot_colors.get(layer_num, "#f0f0f0")
413 lines.append(f' subgraph cluster_L{layer_num} {{')
414 lines.append(f' label="L{layer_num}: {layer_name}";')
415 lines.append(f' style=dashed; color="{color}";')
416 for node in layer_modules:
417 lines.append(f' "{node.name}" [fillcolor="{color}", '
418 f'label="{node.name}\\n{node.lines}L {node.pub_functions}fn"];')
419 lines.append(' }')
420 lines.append('')
421
422 # Edges
423 for name, node in graph.items():
424 for imp in node.imports:
425 if imp in graph:
426 # Red for violations, gray for normal
427 if node.layer < graph[imp].layer and node.layer != 99 and graph[imp].layer != 99:
428 lines.append(f' "{name}" -> "{imp}" [color="red", penwidth=2];')
429 else:
430 lines.append(f' "{name}" -> "{imp}";')
431
432 lines.append('}')
433
434 with open(path, 'w') as f:
435 f.write('\n'.join(lines))
436 print(f" DOT graph exported: {path}")
437 print(f" Render with: dot -Tpng {path} -o graph.png")
