2// VaultStorage.cpp — DPAPI-encrypted multi-key API vault
3//
4// Binary format (v4) — compatible with OmnibusSidebar SuperVault:
5// [MAGIC:4][VERSION:4][EXCHANGE_COUNT:4]
6// for each exchange:
7// [SLOT_COUNT:4]

10// [name_len:4][name_utf8]
11// [key_len:4][key_utf8]

78void VaultStorage::secureWipe(QString& s) {
79 if (s.isEmpty()) return;
80 // Overwrite the internal data
81 auto* data = s.data();

89QByteArray VaultStorage::dpapiEncrypt(const QByteArray& plain) {
90#ifdef Q_OS_WIN
91 DATA_BLOB in;
92 in.cbData = static_cast<DWORD>(plain.size());
93 in.pbData = reinterpret_cast<BYTE*>(const_cast<char*>(plain.data()));
94
95 DATA_BLOB out = {};
96 if (!CryptProtectData(&in, L"OmnibusSidebarVault", nullptr, nullptr, nullptr,
97 CRYPTPROTECT_UI_FORBIDDEN, &out)) {
98 return {};
99 }
100 QByteArray result(reinterpret_cast<const char*>(out.pbData), static_cast<int>(out.cbData));
101 LocalFree(out.pbData);
102 return result;
103#else
104 // Fallback: no encryption (Linux/macOS — placeholder)
105 return plain;
106#endif
107}

109QByteArray VaultStorage::dpapiDecrypt(const QByteArray& cipher) {
110#ifdef Q_OS_WIN
111 DATA_BLOB in;
112 in.cbData = static_cast<DWORD>(cipher.size());
113 in.pbData = reinterpret_cast<BYTE*>(const_cast<char*>(cipher.data()));
114
115 DATA_BLOB out = {};
116 if (!CryptUnprotectData(&in, nullptr, nullptr, nullptr, nullptr, 0, &out)) {
117 return {};
118 }
119 QByteArray result(reinterpret_cast<const char*>(out.pbData), static_cast<int>(out.cbData));
120 SecureZeroMemory(out.pbData, out.cbData);
121 LocalFree(out.pbData);
122 return result;
123#else
124 return cipher;
125#endif
126}

130QByteArray VaultStorage::serialize() const {
131 QByteArray buf;
132 QDataStream ds(&buf, QIODevice::WriteOnly);
133 ds.setByteOrder(QDataStream::LittleEndian);
134
135 ds << VAULT_MAGIC;
136 ds << VAULT_VERSION;
137 ds << static_cast<uint32_t>(VAULT_EXCHANGE_COUNT);
138
139 for (int ex = 0; ex < VAULT_EXCHANGE_COUNT; ++ex) {
140 // Count active slots
141 uint32_t count = 0;
142 for (int s = 0; s < VAULT_MAX_KEYS; ++s) {
143 if (m_stores[ex][s].inUse) ++count;
144 }
145 ds << static_cast<uint32_t>(VAULT_MAX_KEYS); // always write all 8 slots
146
147 for (int s = 0; s < VAULT_MAX_KEYS; ++s) {
148 const auto& slot = m_stores[ex][s];
149 ds << static_cast<uint8_t>(slot.inUse ? 1 : 0);
150 ds << static_cast<uint8_t>(slot.status);
151
152 QByteArray nameUtf8 = slot.name.toUtf8();
153 ds << static_cast<uint32_t>(nameUtf8.size());
154 ds.writeRawData(nameUtf8.data(), nameUtf8.size());
155
156 QByteArray keyUtf8 = slot.apiKey.toUtf8();
157 ds << static_cast<uint32_t>(keyUtf8.size());
158 ds.writeRawData(keyUtf8.data(), keyUtf8.size());
159
160 QByteArray secretUtf8 = slot.apiSecret.toUtf8();
161 ds << static_cast<uint32_t>(secretUtf8.size());
162 ds.writeRawData(secretUtf8.data(), secretUtf8.size());
163 }
164 }
165 return buf;
166}

168bool VaultStorage::deserialize(const QByteArray& data) {
169 QDataStream ds(data);
170 ds.setByteOrder(QDataStream::LittleEndian);
171
172 uint32_t magic, version, exCount;
173 ds >> magic >> version >> exCount;
174
175 if (magic != VAULT_MAGIC || version != VAULT_VERSION) {
176 // Incompatible format — start fresh
177 return false;
178 }
179
180 if (exCount > static_cast<uint32_t>(VAULT_EXCHANGE_COUNT))
181 exCount = VAULT_EXCHANGE_COUNT;
182
183 for (uint32_t ex = 0; ex < exCount; ++ex) {
184 uint32_t slotCount;
185 ds >> slotCount;
186 if (slotCount > VAULT_MAX_KEYS)
187 slotCount = VAULT_MAX_KEYS;
188
189 for (uint32_t s = 0; s < slotCount; ++s) {
190 uint8_t inUse, status;
191 ds >> inUse >> status;
192
193 uint32_t nameLen;
194 ds >> nameLen;
195 QByteArray nameUtf8(nameLen, '\0');
196 ds.readRawData(nameUtf8.data(), nameLen);
197
198 uint32_t keyLen;
199 ds >> keyLen;
200 QByteArray keyUtf8(keyLen, '\0');
201 ds.readRawData(keyUtf8.data(), keyLen);
202
203 uint32_t secretLen;
204 ds >> secretLen;
205 QByteArray secretUtf8(secretLen, '\0');
206 ds.readRawData(secretUtf8.data(), secretLen);
207
208 auto& slot = m_stores[ex][s];
209 slot.inUse = (inUse != 0);
210 slot.status = static_cast<VaultKeyStatus>(status);
211 slot.name = QString::fromUtf8(nameUtf8);
212 slot.apiKey = QString::fromUtf8(keyUtf8);
213 slot.apiSecret = QString::fromUtf8(secretUtf8);
214 }
215 }
216 return true;
217}

221bool VaultStorage::init() {
222 QString path = vaultFilePath();
223 QFile f(path);
224 if (!f.exists()) {
225 m_loaded = true; // empty vault, will be saved on first addKey
226 return true;
227 }
228 if (!f.open(QIODevice::ReadOnly)) {
229 return false;
230 }
231 QByteArray cipher = f.readAll();
232 f.close();
233
234 QByteArray plain = dpapiDecrypt(cipher);
235 if (plain.isEmpty()) {
236 // Could not decrypt — maybe different user or corrupt file
237 m_loaded = true; // start fresh
238 return false;
239 }
240
241 m_loaded = deserialize(plain);
242
243 // Secure-wipe the plaintext buffer
244#ifdef Q_OS_WIN
245 SecureZeroMemory(plain.data(), plain.size());
246#else
247 memset(plain.data(), 0, plain.size());
248#endif
249
250 return m_loaded;
251}

253bool VaultStorage::save() {
254 QByteArray plain = serialize();
255 QByteArray cipher = dpapiEncrypt(plain);
256
257 // Secure-wipe plaintext
258#ifdef Q_OS_WIN
259 SecureZeroMemory(plain.data(), plain.size());
260#else
261 memset(plain.data(), 0, plain.size());
262#endif
263
264 if (cipher.isEmpty()) return false;
265
266 QString path = vaultFilePath();
267 QFile f(path);
268 if (!f.open(QIODevice::WriteOnly | QIODevice::Truncate)) {
269 return false;
270 }
271 f.write(cipher);
272 f.close();
273 return true;
274}

276void VaultStorage::lock() {
277 for (int ex = 0; ex < VAULT_EXCHANGE_COUNT; ++ex) {
278 for (int s = 0; s < VAULT_MAX_KEYS; ++s) {
279 auto& slot = m_stores[ex][s];
280 secureWipe(slot.name);
281 secureWipe(slot.apiKey);
282 secureWipe(slot.apiSecret);
283 slot.inUse = false;
284 slot.status = KEY_STATUS_FREE;
285 }
286 }
287 m_loaded = false;
288}

292bool VaultStorage::addKey(VaultExchange ex, const QString& name,
293 const QString& apiKey, const QString& apiSecret,
294 VaultKeyStatus status)
295{
296 if (ex < 0 || ex >= VAULT_EXCHANGE_COUNT) return false;
297
298 // Find first empty slot
299 for (int s = 0; s < VAULT_MAX_KEYS; ++s) {
300 if (!m_stores[ex][s].inUse) {
301 m_stores[ex][s].inUse = true;
302 m_stores[ex][s].name = name;
303 m_stores[ex][s].apiKey = apiKey;
304 m_stores[ex][s].apiSecret = apiSecret;
305 m_stores[ex][s].status = status;
306 return save();
307 }
308 }
309 return false; // all 8 slots full
310}

314 if (slot < 0 || slot >= VAULT_MAX_KEYS) return false;
