57def sha256_file(path: str) -> str:
58 """Compute SHA-256 of a file."""
59 h = hashlib.sha256()
60 with open(path, "rb") as f:
61 while True:
62 chunk = f.read(8192)
63 if not chunk:
64 break
65 h.update(chunk)
66 return h.hexdigest()
67

69def sha256_bytes(data: bytes) -> str:
70 return hashlib.sha256(data).hexdigest()

73def load_rules() -> list[dict]:
74 """Load all vulnerability rules from rules directory."""
75 rules = []
76
77 # Load default rules
78 if DEFAULT_RULES_FILE.exists():
79 try:
80 with open(DEFAULT_RULES_FILE, "r", encoding="utf-8") as f:
81 data = json.load(f)
82 if isinstance(data, dict) and "rules" in data:
83 rules.extend(data["rules"])
84 elif isinstance(data, list):
85 rules.extend(data)
86 log_info(f"Loaded {len(rules)} rules from {DEFAULT_RULES_FILE.name}")
87 except (json.JSONDecodeError, IOError) as e:
88 log_fail(f"Error loading {DEFAULT_RULES_FILE}: {e}")
89
90 # Load any additional rule files
91 if RULES_DIR.exists():
92 for rule_file in RULES_DIR.glob("*.json"):
93 if rule_file.name in ("default-rules.json", "manifest.json"):
94 continue
95 try:
96 with open(rule_file, "r", encoding="utf-8") as f:
97 data = json.load(f)
98 extra = data.get("rules", data) if isinstance(data, dict) else data
99 if isinstance(extra, list):
100 rules.extend(extra)
101 log_info(f"Loaded {len(extra)} additional rules from {rule_file.name}")
102 except (json.JSONDecodeError, IOError):
103 pass
104
105 return rules

108def update_rules(url: str) -> bool:
109 """Check for and download updated rules from URL."""
110 if not url:
111 log_warn("No update URL configured. Use --url to specify one.")
112 log_info("Example: --url https://raw.githubusercontent.com/org/repo/main/rules/manifest.json")
113 return False
114
115 log_info(f"Checking for updates at: {url}")
116
117 try:
118 req = urllib.request.Request(url, headers={"User-Agent": "OmniBus-VulnUpdater/1.0"})
119 with urllib.request.urlopen(req, timeout=15) as resp:
120 remote_data = resp.read()
121 except urllib.error.URLError as e:
122 log_fail(f"Failed to fetch manifest: {e}")
123 return False
124 except Exception as e:
125 log_fail(f"Unexpected error: {e}")
126 return False
127
128 try:
129 remote_manifest = json.loads(remote_data.decode("utf-8"))
130 except json.JSONDecodeError as e:
131 log_fail(f"Invalid JSON in remote manifest: {e}")
132 return False
133
134 # Compare versions
135 remote_version = remote_manifest.get("version", "0.0.0")
136 local_version = "0.0.0"
137 if MANIFEST_FILE.exists():
138 try:
139 with open(MANIFEST_FILE, "r") as f:
140 local_manifest = json.load(f)
141 local_version = local_manifest.get("version", "0.0.0")
142 except (json.JSONDecodeError, IOError):
143 pass
144
145 log_info(f"Local version: {local_version}")
146 log_info(f"Remote version: {remote_version}")
147
148 if remote_version <= local_version:
149 log_pass("Rules are up to date.")
150 return True
151
152 # Download and verify rules
153 rules_url = remote_manifest.get("rules_url", "")
154 expected_sha = remote_manifest.get("sha256", "")
155
156 if rules_url:
157 try:
158 req = urllib.request.Request(rules_url, headers={"User-Agent": "OmniBus-VulnUpdater/1.0"})
159 with urllib.request.urlopen(req, timeout=15) as resp:
160 rules_data = resp.read()
161 except Exception as e:
162 log_fail(f"Failed to download rules: {e}")
163 return False
164
165 # Verify checksum
166 actual_sha = sha256_bytes(rules_data)
167 if expected_sha and actual_sha != expected_sha:
168 log_fail(f"SHA-256 mismatch! Expected: {expected_sha}, Got: {actual_sha}")
169 log_fail("Rules file may be tampered. Aborting update.")
170 return False
171
172 if expected_sha:
173 log_pass(f"SHA-256 verified: {actual_sha[:16]}...")
174
175 # Save rules
176 RULES_DIR.mkdir(parents=True, exist_ok=True)
177 rules_path = RULES_DIR / "downloaded-rules.json"
178 with open(rules_path, "wb") as f:
179 f.write(rules_data)
180 log_pass(f"Updated rules saved to {rules_path}")
181
182 # Save manifest
183 with open(MANIFEST_FILE, "w") as f:
184 json.dump(remote_manifest, f, indent=2)
185 log_pass(f"Manifest updated to version {remote_version}")
186
187 return True

190def scan_zig_files(rules: list[dict]) -> list[dict]:
191 """Scan core/*.zig files for vulnerability pattern matches."""
192 if not CORE_DIR.exists():
193 log_fail(f"Core directory not found: {CORE_DIR}")
194 return []
195
196 zig_files = list(CORE_DIR.glob("*.zig"))
197 log_info(f"Scanning {len(zig_files)} Zig source files in {CORE_DIR}")
198
199 findings = []
200 compiled_rules = []
201 for rule in rules:
202 pattern = rule.get("pattern", "")
203 if not pattern:
204 continue
205 try:
206 regex = re.compile(pattern, re.IGNORECASE | re.MULTILINE)
207 compiled_rules.append((rule, regex))
208 except re.error as e:
209 log_warn(f"Invalid regex in rule {rule.get('id', '?')}: {e}")
210
211 for zig_file in sorted(zig_files):
212 try:
213 content = zig_file.read_text(encoding="utf-8", errors="replace")
214 except IOError:
215 continue
216
217 lines = content.splitlines()
218
219 for rule, regex in compiled_rules:
220 for i, line in enumerate(lines, 1):
221 if regex.search(line):
222 findings.append({
223 "rule_id": rule.get("id", "?"),
224 "severity": rule.get("severity", "INFO"),
225 "description": rule.get("description", ""),
226 "file": str(zig_file.name),
227 "line": i,
228 "matched_text": line.strip()[:120],
229 "fix_suggestion": rule.get("fix_suggestion", ""),
230 })
231
232 return findings
