141 def run_build_step(self, step: str, timeout: int = 120) -> TestResult:
142 """Rulează un build step de test."""
143 start = time.time()
144
145 cmd = ['zig', 'build', step, '-Doqs=false']
146
147 try:
148 result = subprocess.run(
149 cmd,
150 cwd=self.project_root,
151 capture_output=True,
152 text=True,
153 timeout=timeout
154 )
155
156 duration = (time.time() - start) * 1000
157
158 # Parse output pentru a găsi numărul de teste
159 passed = 0
160 failed = 0
161 output = result.stdout + result.stderr
162
163 for line in output.split('\n'):
164 if 'passed' in line.lower():
165 try:
166 # Caută pattern "X passed, Y failed"
167 import re
168 match = re.search(r'(\d+)\s+passed', line, re.IGNORECASE)
169 if match:
170 passed = int(match.group(1))
171 match = re.search(r'(\d+)\s+failed', line, re.IGNORECASE)
172 if match:
173 failed = int(match.group(1))
174 except:
175 pass
176
177 success = result.returncode == 0 and failed == 0
178
179 return TestResult(
180 name=step,
181 passed=success,
182 duration_ms=duration,
183 output=output[-500:] if output else "",
184 error=result.stderr[-500:] if result.stderr and not success else ""
185 )
186
187 except subprocess.TimeoutExpired:
188 return TestResult(
189 name=step,
190 passed=False,
191 duration_ms=timeout * 1000,
192 error=f"Timeout după {timeout} secunde"
193 )
194 except FileNotFoundError:
195 return TestResult(
196 name=step,
197 passed=False,
198 duration_ms=0,
199 error="Zig nu e în PATH"
200 )

307def watch_mode(runner: TestRunner, group_name: Optional[str] = None):
308 """Watch mode pentru TDD."""
309 import time
310
311 print("\n [WATCH MODE] Aștept modificări... (Ctrl+C pentru oprire)\n")
312
313 last_mtime = 0
314
315 try:
316 while True:
317 # Verifică dacă s-a modificat vreun fișier .zig
318 max_mtime = 0
319 for zig_file in runner.project_root.rglob('*.zig'):
320 if any(part.startswith('.') for part in zig_file.parts):
321 continue
322 mtime = zig_file.stat().st_mtime
323 if mtime > max_mtime:
324 max_mtime = mtime
325
326 if max_mtime > last_mtime:
327 last_mtime = max_mtime
328 print(f"\n [CHANGE] Modificare detectată, rulez testele...\n")
329
330 if group_name:
331 result = runner.run_group(group_name)
332 print_results([result])
333 else:
334 results = runner.run_all()
335 print_results(results)
336
337 print(" [WATCH] Aștept următoarea modificare...\n")
338
339 time.sleep(1)
340
341 except KeyboardInterrupt:
342 print("\n [WATCH] Oprit de utilizator.\n")
