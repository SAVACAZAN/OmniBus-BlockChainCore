90 def detect_outlier(self, data: PriceData, other_prices: List[float]) -> Tuple[bool, float]:
91 """Detect if price is an outlier compared to other sources."""
92 if len(other_prices) < 2:
93 return False, 0.0
94
95 median_price = statistics.median(other_prices)
96 if median_price == 0:
97 return False, 0.0
98
99 deviation = abs(data.price - median_price) / median_price * 100
100
101 # Get threshold for this asset
102 thresholds = self.ASSET_THRESHOLDS.get(data.asset, {})
103 max_deviation = thresholds.get("volatility", 0.05) * 100
104
105 is_outlier = deviation > max_deviation
106 return is_outlier, deviation

108 def check_manipulation(self, asset: str, current_price: float) -> Tuple[bool, str]:
109 """Check for potential price manipulation."""
110 history = self.price_history.get(asset, [])
111
112 if len(history) < 10:
113 return False, "Insufficient history"
114
115 # Get recent prices
116 recent_prices = [p.price for p in history[-10:]]
117 avg_price = statistics.mean(recent_prices)
118 std_dev = statistics.stdev(recent_prices) if len(recent_prices) > 1 else 0
119
120 if std_dev == 0:
121 return False, "No price movement"
122
123 # Check if current price is > 3 standard deviations
124 z_score = abs(current_price - avg_price) / std_dev
125
126 if z_score > 3:
127 return True, f"Unusual price movement (z-score: {z_score:.2f})"
128
129 # Check for sudden large change
130 if len(recent_prices) >= 2:
131 prev_price = recent_prices[-1]
132 if prev_price > 0:
133 change = abs(current_price - prev_price) / prev_price * 100
134 if change > 20: # 20% change in one update
135 return True, f"Sudden price change: {change:.2f}%"
136
137 return False, "Normal price movement"

139 def verify_consensus(self, prices: List[PriceData]) -> VerificationResult:
140 """Verify consensus among multiple oracle sources."""
141 if not prices:
142 return VerificationResult(
143 status=DataStatus.UNAVAILABLE,
144 message="No price data available",
145 confidence=0.0,
146 sources_used=0
147 )
148
149 asset = prices[0].asset
150
151 # Filter out stale data
152 fresh_prices = [p for p in prices if not self.is_stale(p)]
153 stale_count = len(prices) - len(fresh_prices)
154
155 if len(fresh_prices) < self.MIN_SOURCES_FOR_CONSENSUS:
156 return VerificationResult(
157 status=DataStatus.STALE,
158 message=f"Insufficient fresh data sources. Stale: {stale_count}",
159 confidence=0.3,
160 sources_used=len(fresh_prices)
161 )
162
163 # Check for outliers
164 price_values = [p.price for p in fresh_prices]
165 outlier_count = 0
166 max_deviation = 0.0
167
168 for price_data in fresh_prices:
169 other_prices = [p for p in price_values if p != price_data.price]
170 is_outlier, deviation = self.detect_outlier(price_data, other_prices)
171 if is_outlier:
172 outlier_count += 1
173 max_deviation = max(max_deviation, deviation)
174
175 # Remove outliers for consensus calculation
176 valid_prices = []
177 for p in fresh_prices:
178 other_prices = [x.price for x in fresh_prices if x.price != p.price]
179 is_outlier, _ = self.detect_outlier(p, other_prices)
180 if not is_outlier:
181 valid_prices.append(p)
182
183 if len(valid_prices) < self.MIN_SOURCES_FOR_CONSENSUS:
184 return VerificationResult(
185 status=DataStatus.OUTLIER,
186 message=f"Too many outliers ({outlier_count}). Max deviation: {max_deviation:.2f}%",
187 confidence=0.4,
188 sources_used=len(valid_prices),
189 price_deviation=max_deviation,
190 recommended_action="Investigate price sources or wait for convergence"
191 )
192
193 # Calculate consensus price (median)
194 consensus_price = statistics.median([p.price for p in valid_prices])
195
196 # Check for manipulation
197 is_manipulated, manipulation_msg = self.check_manipulation(asset, consensus_price)
198
199 if is_manipulated:
200 return VerificationResult(
201 status=DataStatus.MANIPULATED,
202 message=f"Potential manipulation detected: {manipulation_msg}",
203 confidence=0.2,
204 sources_used=len(valid_prices),
205 price_deviation=max_deviation,
206 recommended_action="Pause trading, investigate immediately"
207 )
208
209 # Calculate confidence based on source agreement
210 price_std = statistics.stdev([p.price for p in valid_prices]) if len(valid_prices) > 1 else 0
211 confidence = 1.0 - min(price_std / consensus_price, 1.0) if consensus_price > 0 else 0.0
212
213 return VerificationResult(
214 status=DataStatus.VALID,
215 message=f"Valid consensus price: ${consensus_price:,.2f}",
216 confidence=confidence,
217 sources_used=len(valid_prices),
218 price_deviation=max_deviation,
219 recommended_action="Use for on-chain operations"
220 )
