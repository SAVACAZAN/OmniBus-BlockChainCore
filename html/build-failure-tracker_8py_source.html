56ZIG_ERROR_PATTERNS = [
57 re.compile(r"error:\s*(.*)"),
58 re.compile(r"error\‍[(\w+)\‍]\s*"),
59 re.compile(r"undefined\s+reference\s+to\s+(\S+)"),
60 re.compile(r"expected\s+\w+,\s+found\s+\w+"),
61 re.compile(r"no\s+member\s+named\s+'(\w+)'"),
62 re.compile(r"unable\s+to\s+\w+"),
63]

66def extract_zig_errors(text: str) -> list[str]:
67 errors: list[str] = []
68 for line in text.splitlines():
69 for pat in ZIG_ERROR_PATTERNS:
70 m = pat.search(line)
71 if m:
72 errors.append(line.strip())
73 break
74 return errors

77def extract_module_from_error(line: str) -> str | None:
78 """Heuristic: find 'core/Foo.zig' or similar in an error line."""
79 m = re.search(r"(core[\/][\w_]+\.zig)", line)
80 if m:
81 return Path(m.group(1)).stem
82 m = re.search(r"src[\/]([\w_]+\.zig)", line)
83 if m:
84 return Path(m.group(1)).stem
85 return None

91def scan_log_files(log_dir: str) -> list[dict[str, Any]]:
92 records: list[dict[str, Any]] = []
93 if not os.path.isdir(log_dir):
94 return records
95 for fpath in glob.glob(os.path.join(log_dir, "*.log")):
96 text = Path(fpath).read_text(encoding="utf-8", errors="replace")
97 errors = extract_zig_errors(text)

107 ).isoformat(),
108 }
109 )

130def get_changed_files(cwd: str, commit_hash: str) -> list[str]:
131 out = run_git(["diff-tree", "--no-commit-id", "--name-only", "-r", commit_hash], cwd=cwd)

135def scan_git_build_failures(cwd: str, max_commits: int = 200) -> list[dict[str, Any]]:
136 fmt = "%H|%aI|%s"
137 out = run_git(["log", f"--format={fmt}", f"-n{max_commits}"], cwd=cwd)
138 records: list[dict[str, Any]] = []
139 for line in out.strip().splitlines():
140 if "|" not in line:
141 continue
142 parts = line.split("|", 2)
143 h, date, msg = parts[0], parts[1], parts[2]
144 lower = msg.lower()
145 if any(k in lower for k in ["build", "compile", "zig build", "ci fix", "fix build", "broken build"]):
146 changed = get_changed_files(cwd, h)
147 zig_files = [f for f in changed if f.endswith(".zig")]
148 records.append(
149 {
150 "source": f"git:{h}",
151 "commit": h,
152 "date": date,
153 "message": msg,
154 "changed_files": changed,
155 "zig_files": zig_files,
156 }
157 )
158 return records

161# ---------------------------------------------------------------------------
162# Pattern mining
163# ---------------------------------------------------------------------------
164def find_xy_patterns(records: list[dict[str, Any]]) -> list[dict[str, Any]]:
165 """Find 'after X, Y breaks' patterns from consecutive build failure records."""
166 xy_counts: dict[tuple[str, str], int] = defaultdict(int)
167 for i in range(1, len(records)):
168 prev_mods = set()
169 curr_mods = set()
170 for f in records[i - 1].get("zig_files", []):
171 prev_mods.add(Path(f).stem)
172 for f in records[i].get("zig_files", []):
173 curr_mods.add(Path(f).stem)
174 for p in prev_mods:
175 for c in curr_mods:
176 if p != c:
177 xy_counts[(p, c)] += 1
178
179 patterns = []
180 for (x, y), count in sorted(xy_counts.items(), key=lambda kv: kv[1], reverse=True):
181 if count >= 2:
182 patterns.append({"after_updating": x, "module_y_tends_to_break": y, "occurrences": count})
183 return patterns[:20]

189def main() -> int:

199 log_info("Scanning log files …")
200 log_records = scan_log_files(os.path.join(repo, args.log_dir))
201 log_info(f"Log errors found: {len(log_records)}")

203 log_info("Scanning git history for build failures …")
204 try:
205 git_records = scan_git_build_failures(repo, args.max_commits)
206 except Exception as exc:
207 log_warn(str(exc))
208 git_records = []
209 log_info(f"Git build failure commits: {len(git_records)}")
210
211 all_records = log_records + [
212 {"source": r["source"], "module": Path(f).stem, "error": r["message"], "timestamp": r["date"]}
213 for r in git_records
214 for f in r.get("zig_files", [])
215 ]
216
217 # Aggregate
218 module_counts: dict[str, int] = defaultdict(int)
219 error_counts: dict[str, int] = defaultdict(int)
220 for rec in all_records:
221 mod = rec.get("module")
222 if mod:
223 module_counts[mod] += 1
224 err = rec.get("error", "unknown")
225 # Bucket generic error text
226 bucket = err[:120]
227 error_counts[bucket] += 1
228
229 patterns = find_xy_patterns(git_records)
230
231 report = {
232 "generated_at": datetime.now(timezone.utc).isoformat(),
233 "log_records": len(log_records),
234 "git_records": len(git_records),
235 "top_modules_by_failure": dict(sorted(module_counts.items(), key=lambda kv: kv[1], reverse=True)[:20]),
236 "top_errors": dict(sorted(error_counts.items(), key=lambda kv: kv[1], reverse=True)[:20]),
237 "xy_patterns": patterns,
238 "raw_records": all_records,
239 }
240
241 out_path = os.path.join(repo, args.output)
242 os.makedirs(os.path.dirname(out_path), exist_ok=True)
243 with open(out_path, "w", encoding="utf-8") as fh:
244 json.dump(report, fh, indent=2)
245
246 log_pass(f"Build history written to {out_path}")
247 log_info(f"Unique modules with failures: {len(module_counts)}")
