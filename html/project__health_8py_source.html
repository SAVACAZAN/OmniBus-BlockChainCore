142def count_zig_files(root_dir: Path) -> Tuple[int, int, int, int]:
143 """Numără fișierele Zig, testele, liniile de cod."""
144 zig_count = 0
145 test_count = 0
146 loc_total = 0
147 loc_tests = 0
148 constants_count = 0
149 enums_count = 0
150
151 for filepath in root_dir.rglob('*.zig'):
152 # Skip cache și output
153 if any(part.startswith('.') or part in ['zig-out', '.zig-cache']
154 for part in filepath.parts):
155 continue
156
157 zig_count += 1
158
159 try:
160 with open(filepath, 'r', encoding='utf-8') as f:
161 content = f.read()
162 lines = content.split('\n')
163 loc_total += len(lines)
164
165 # Numără test blocks
166 if 'test "' in content or 'test {' in content:
167 test_count += 1
168 loc_tests += len([l for l in lines if l.strip().startswith('test ')])
169
170 # Numără constante și enum-uri (aproximativ)
171 constants_count += content.count('pub const')
172 enums_count += content.count('= enum')
173 except Exception:
174 pass
175
176 return zig_count, test_count, loc_total, loc_tests, constants_count, enums_count

191def analyze_project(root_dir: Path) -> HealthStatus:
192 """Analizează starea completă a proiectului."""
193 status = HealthStatus()
194
195 # Check Zig
196 status.zig_installed, status.zig_version = get_zig_version()
197 if status.zig_installed:
198 status.zig_version_ok = check_version_meets_min(
199 status.zig_version,
200 status.zig_min_version
201 )
202 else:
203 status.errors.append("Zig nu este instalat sau nu e în PATH")
204
205 # Check liboqs
206 status.liboqs_installed, status.liboqs_path = check_liboqs()
207 if not status.liboqs_installed:
208 status.warnings.append("liboqs nu e detectat (opțional, doar pentru PQ crypto)")
209
210 # Count files and LOC
211 (
212 status.files_zig,
213 status.files_test,
214 status.loc_total,
215 status.loc_tests,
216 status.constants_count,
217 status.enums_count
218 ) = count_zig_files(root_dir)
219
220 # Check structure
221 structure = check_project_structure(root_dir)
222 status.build_zig_present = structure['build_zig']
223 status.omnibus_toml_present = structure['omnibus_toml']
224 status.opcodes_md_present = structure['opcodes_md']
225 status.readme_present = structure['readme']
226
227 if not status.build_zig_present:
228 status.errors.append("Lipsește build.zig - proiectul nu poate fi compilat")
229
230 return status
