52def scan_all_modules():
53 """Scan ALL .zig files in core/ dynamically. No hardcoded lists."""
54 modules = {}
55 for path in sorted(CORE.glob("*.zig")):

57 content = path.read_text(encoding="utf-8", errors="ignore")

61 comments = len([l for l in lines if l.strip().startswith("//")])
62 pub_fns = len(re.findall(r"pub fn \w+", content))
63 tests = len(re.findall(r'^test "', content, re.MULTILINE))
64 structs = len(re.findall(r"pub const \w+ = struct", content))
65 enums = len(re.findall(r"pub const \w+ = enum", content))
66 imports = re.findall(r'@import\‍("(\w+)\.zig"\‍)', content)
67 todos = len(re.findall(r"TODO|FIXME|HACK|XXX", content, re.IGNORECASE))
68 debug_prints = len(re.findall(r"std\.debug\.print", content))
69
70 modules[name] = {
71 "path": str(path),
72 "loc": loc,
73 "comments": comments,
74 "pub_functions": pub_fns,
75 "tests": tests,
76 "structs": structs,
77 "enums": enums,
78 "imports": imports,
79 "todos": todos,
80 "debug_prints": debug_prints,
81 "is_real": loc > 20 and pub_fns > 0,
82 "has_tests": tests > 0,

125def infer_layers(modules):
126 """Infer module layers from import depth."""
127 # Start with hints
128 layers = dict(LAYER_HINTS)
129
130 # BFS from known layers
131 for _ in range(8): # max iterations
132 for name, mod in modules.items():
133 if name in layers:
134 continue
135 import_layers = [layers.get(imp, -1) for imp in mod["imports"] if imp in modules]
136 import_layers = [l for l in import_layers if l >= 0]
137 if import_layers:
138 # Module is one layer above its highest import
139 layers[name] = min(max(import_layers) + 1, 7)
140
141 # Assign remaining to middle layer
142 for name in modules:
143 if name not in layers:
144 layers[name] = 3 # default middle
145
146 return layers

149# ── Dependency Analysis ──────────────────────────────────────────────────────

151def analyze_dependencies(modules):
152 """Find circular deps, layer violations, isolated modules."""
153 layers = infer_layers(modules)
154
155 # Build import graph
156 graph = {name: set(mod["imports"]) & set(modules.keys()) for name, mod in modules.items()}
157 reverse_graph = defaultdict(set)
158 for name, imports in graph.items():
159 for imp in imports:
160 reverse_graph[imp].add(name)
161
162 # Find circular dependencies (DFS)
163 cycles = []
164 visited = set()
165 rec_stack = set()
166
167 def dfs(node, path):
168 visited.add(node)
169 rec_stack.add(node)
170 for neighbor in graph.get(node, []):
171 if neighbor not in visited:
172 dfs(neighbor, path + [neighbor])
173 elif neighbor in rec_stack:
174 # Found cycle
175 cycle_start = path.index(neighbor) if neighbor in path else len(path)
176 cycle = path[cycle_start:] + [neighbor]
177 if len(cycle) >= 2:
178 cycle_str = " -> ".join(cycle)
179 if cycle_str not in [c["path"] for c in cycles]:
180 cycles.append({"path": cycle_str, "length": len(cycle)})
181 rec_stack.discard(node)
182
183 for node in modules:
184 if node not in visited:
185 dfs(node, [node])
186
187 # Find layer violations
188 violations = []
189 for name, imports in graph.items():
190 my_layer = layers.get(name, 3)
191 for imp in imports:
192 imp_layer = layers.get(imp, 3)
193 if imp_layer > my_layer + 1: # Allow importing from same or one layer up
194 violations.append({
195 "from": name, "from_layer": my_layer,
196 "to": imp, "to_layer": imp_layer,
197 })
198
199 # Find isolated modules
200 isolated = [name for name in modules
201 if not graph.get(name) and not reverse_graph.get(name)]
202
203 return {
204 "layers": layers,
205 "cycles": cycles,
206 "violations": violations,
207 "isolated": isolated,
208 "edges": sum(len(v) for v in graph.values()),
209 "nodes": len(modules),
210 }

215def security_scan(modules):
216 """Scan for security issues dynamically."""
217 issues = {"critical": [], "medium": [], "low": []}
218
219 for name, mod in modules.items():
220 content = Path(mod["path"]).read_text(encoding="utf-8", errors="ignore")
221
222 # Critical: hardcoded secrets (but filter test files and known constants)
223 for match in re.finditer(r'const\s+(\w*(?:key|secret|password|token)\w*)\s*=\s*"([^"]+)"', content, re.IGNORECASE):
224 var_name, value = match.group(1), match.group(2)
225 # Filter known non-secrets: test data, protocol constants, crypto seeds
226 known_safe = (
227 "test" in var_name.lower() or "test" in name.lower() or
228 "Bitcoin seed" in value or "sample" in value.lower() or
229 "header" in var_name.lower() or # HTTP/WS headers
230 "pubkey" in var_name.lower() or # test public keys
231 "dGhlIHNhbXBsZSBub25jZQ==" in value or # WebSocket RFC 6455 magic
232 "258EAFA5" in value or # WebSocket GUID
233 len(value) < 6 # Too short to be a real secret
234 )
235 if known_safe:
236 continue
237 issues["critical"].append(f"{name}: hardcoded {var_name}")
238
239 # Medium: debug prints (count only)
240 if mod["debug_prints"] > 0 and "test" not in name:
241 issues["medium"].append(f"{name}: {mod['debug_prints']} debug prints")
242
243 # Low: TODOs
244 if mod["todos"] > 0:
245 issues["low"].append(f"{name}: {mod['todos']} TODO/FIXME")
246
247 return issues

295def compute_metrics(modules):
296 """Compute aggregate code metrics."""
297 total_loc = sum(m["loc"] for m in modules.values())
298 total_comments = sum(m["comments"] for m in modules.values())
299 total_tests = sum(m["tests"] for m in modules.values())
300 total_fns = sum(m["pub_functions"] for m in modules.values())
301 total_structs = sum(m["structs"] for m in modules.values())
302 total_enums = sum(m["enums"] for m in modules.values())
303 real_modules = sum(1 for m in modules.values() if m["is_real"])
304 tested_modules = sum(1 for m in modules.values() if m["has_tests"])
305 modules_without_tests = [n for n, m in modules.items() if m["is_real"] and not m["has_tests"]]
306
307 test_ratio = (total_tests / max(total_fns, 1)) * 100
308
309 return {
310 "total_modules": len(modules),
311 "real_modules": real_modules,
312 "tested_modules": tested_modules,
313 "total_loc": total_loc,
314 "total_comments": total_comments,
315 "total_tests": total_tests,
316 "total_functions": total_fns,
317 "total_structs": total_structs,
318 "total_enums": total_enums,
319 "test_ratio_pct": round(test_ratio, 1),
320 "comment_ratio_pct": round(total_comments / max(total_loc, 1) * 100, 1),
321 "avg_loc_per_module": round(total_loc / max(len(modules), 1)),
322 "modules_without_tests": modules_without_tests,
323 "top_10_largest": sorted(
324 [(n, m["loc"]) for n, m in modules.items()],
325 key=lambda x: -x[1]
326 )[:10],
327 }
