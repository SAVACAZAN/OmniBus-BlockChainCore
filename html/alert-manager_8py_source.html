64def rpc_call(host: str, port: int, method: str, params=None, timeout: int = 5):
65 """JSON-RPC 2.0 call. Returns (result, error_string)."""
66 payload = json.dumps({
67 "jsonrpc": "2.0", "id": 1,
68 "method": method, "params": params or [],
69 })
70 try:
71 conn = http.client.HTTPConnection(host, port, timeout=timeout)
72 conn.request("POST", "/", body=payload,
73 headers={"Content-Type": "application/json"})
74 resp = conn.getresponse()
75 data = json.loads(resp.read().decode())
76 conn.close()
77 if "error" in data and data["error"]:
78 return None, str(data["error"])
79 return data.get("result"), None
80 except Exception as exc:
81 return None, str(exc)

91def check_rpc_reachable(host: str, port: int, timeout: int) -> list:
92 alerts = []
93 result, err = rpc_call(host, port, "getblockchaininfo", timeout=timeout)

101def check_peers(host: str, port: int, min_peers: int, timeout: int) -> list:
102 alerts = []
103 result, err = rpc_call(host, port, "getnetworkinfo", timeout=timeout)
104 if err:
105 alerts.append(("WARNING", "Peers", f"Cannot query peers — {err}"))

110 else:

115def check_disk_space(chain_file: str, min_free_gb: float) -> list:
116 alerts = []
117 path = Path(chain_file).resolve()
118 check_dir = path.parent if path.parent.exists() else Path(".")
119 try:
120 usage = shutil.disk_usage(str(check_dir))
121 free_gb = usage.free / (1024 ** 3)
122 if free_gb < min_free_gb:
123 alerts.append(("CRITICAL", "Disk", f"Only {free_gb:.2f} GB free (threshold: {min_free_gb} GB)"))
124 else:
125 alerts.append(("OK", "Disk", f"{free_gb:.1f} GB free"))
126 except Exception as exc:
127 alerts.append(("WARNING", "Disk", f"Cannot check disk space — {exc}"))
128 return alerts

131def check_chain_growing(host, port, timeout, prev_height) -> tuple:
132 """Returns (alerts, current_height)."""
133 alerts = []
134 result, err = rpc_call(host, port, "getblockchaininfo", timeout=timeout)
135 if err:
136 return [("WARNING", "Chain", "Cannot query block height")], prev_height
137
138 height = result.get("blocks", 0) if result else 0
139 if prev_height is not None and height <= prev_height:
140 alerts.append(("WARNING", "Chain", f"Block height unchanged at {height} — node may not be mining"))
141 elif prev_height is not None:
142 alerts.append(("OK", "Chain", f"Chain growing: {prev_height} -> {height} (+{height - prev_height} blocks)"))
143 else:
144 alerts.append(("INFO", "Chain", f"Initial block height: {height}"))
145 return alerts, height

157def run_checks(host, port, cfg, prev_height):
158 """Run all checks, return (all_alerts, new_height)."""
159 timeout = cfg["rpc_timeout_seconds"]
160 all_alerts = []
161
162 rpc_alerts, rpc_result = check_rpc_reachable(host, port, timeout)
163 all_alerts.extend(rpc_alerts)
164
165 # Only run further checks if RPC is reachable
166 if rpc_result is not None:
167 all_alerts.extend(check_peers(host, port, cfg["min_peers"], timeout))
168 chain_alerts, new_height = check_chain_growing(host, port, timeout, prev_height)
169 all_alerts.extend(chain_alerts)
170 else:
171 new_height = prev_height
172
173 script_dir = Path(__file__).resolve().parent
174 chain_path = script_dir.parent.parent / cfg["chain_file"]
175 all_alerts.extend(check_disk_space(str(chain_path), cfg["min_disk_free_gb"]))
176
177 return all_alerts, new_height

185def print_summary(alerts):
186 crits = sum(1 for s, _, _ in alerts if s == "CRITICAL")
187 warns = sum(1 for s, _, _ in alerts if s == "WARNING")
188 oks = sum(1 for s, _, _ in alerts if s == "OK")
189 print(f" {DIM}{'─' * 55}{RESET}")
190 parts = []
191 if crits:
192 parts.append(f"{RED}{crits} critical{RESET}")
193 if warns:
194 parts.append(f"{YELLOW}{warns} warnings{RESET}")
195 parts.append(f"{GREEN}{oks} ok{RESET}")
196 print(f" Summary: {', '.join(parts)}")
