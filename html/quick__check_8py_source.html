63 def check_syntax(self) -> CheckResult:
64 """Verifică sintaxa Zig cu zig fmt --check."""
65 start = time.time()
66
67 # Găsește toate fișierele .zig
68 zig_files = list(self.project_root.rglob('*.zig'))
69 zig_files = [f for f in zig_files
70 if not any(part.startswith('.') or part in ['zig-out', '.zig-cache']
71 for part in f.parts)]
72
73 passed_count = 0
74 failed_files = []
75
76 for filepath in zig_files:
77 ok, stdout, stderr = self.run_command(
78 ['zig', 'fmt', '--check', str(filepath)],
79 timeout=5
80 )
81 if ok and not stderr:
82 passed_count += 1
83 else:
84 failed_files.append(filepath.name)
85
86 duration = (time.time() - start) * 1000
87
88 return CheckResult(
89 name="Sintaxă Zig",
90 passed=passed_count == len(zig_files),
91 duration_ms=duration,
92 details=[f"{passed_count}/{len(zig_files)} fișiere OK"] +
93 ([f"Failed: {', '.join(failed_files[:3])}"] if failed_files else [])
94 )

177 def run_all(self, checks_to_run: List[str] = None) -> List[CheckResult]:
178 """Rulează toate verificările."""
179 all_checks = {
180 'syntax': self.check_syntax,
181 'build': self.check_build,
182 'test-crypto': self.check_tests_crypto,
183 'test-chain': self.check_tests_chain,
184 }
185
186 if checks_to_run:
187 checks = {k: v for k, v in all_checks.items() if k in checks_to_run}
188 else:
189 checks = all_checks
190
191 # Rulează în paralel pentru viteză
192 with ThreadPoolExecutor(max_workers=2) as executor:
193 futures = {executor.submit(fn()): name for name, fn in checks.items()}
194
195 for future in as_completed(futures):
196 result = future.result()
197 self.results.append(result)
198
199 # Sortează după ordinea originală
200 order = ['syntax', 'build', 'test-crypto', 'test-chain']
201 self.results.sort(key=lambda r: order.index(next(
202 (k for k, v in all_checks.items() if v.__name__ in r.name.lower().replace(' ', '_')),
203 0
204 )))
205
206 return self.results
