109void ExchangePanel::refreshTable() {
110 auto& vault = VaultStorage::instance();
111 m_table->setRowCount(0);
112
113 int count = 0;
114 for (int s = 0; s < VAULT_MAX_KEYS; ++s) {
115 auto key = vault.getKey(m_exchange, s);
116 if (!key.inUse) continue;
117
118 int row = m_table->rowCount();
119 m_table->insertRow(row);
120
121 m_table->setItem(row, 0, new QTableWidgetItem(key.name));
122
123 // Mask the API key — show first 6 + last 4 chars
124 QString masked = key.apiKey;
125 if (masked.length() > 12) {
126 masked = masked.left(6) + "..." + masked.right(4);
127 }
128 m_table->setItem(row, 1, new QTableWidgetItem(masked));
129
130 auto* statusItem = new QTableWidgetItem(VaultStorage::statusName(static_cast<VaultKeyStatus>(key.status)));
131 switch (key.status) {
132 case KEY_STATUS_PAID:
133 statusItem->setForeground(QColor("#00b3a4"));
134 break;
135 case KEY_STATUS_NOTPAID:
136 statusItem->setForeground(QColor("#ff5555"));
137 break;
138 default:
139 statusItem->setForeground(QColor("#ccaa00"));
140 break;
141 }
142 m_table->setItem(row, 2, statusItem);
143
144 // Hidden slot index
145 m_table->setItem(row, 3, new QTableWidgetItem(QString::number(s)));
146
147 ++count;
148 }
149
150 m_countLabel->setText(QString("%1 / %2 keys").arg(count).arg(VAULT_MAX_KEYS));
151 m_addBtn->setEnabled(count < VAULT_MAX_KEYS && vault.isLoaded());
152 onSelectionChanged();
153}

258void ExchangePanel::onAddKey() {
259 auto& vault = VaultStorage::instance();
260 if (!vault.isLoaded()) {
261 QMessageBox::warning(this, "Vault Locked", "Unlock the vault first.");
262 return;
263 }
264
265 QString name, apiKey, apiSecret;
266 VaultKeyStatus status = KEY_STATUS_FREE;
267 QString title = QString("Add %1 API Key").arg(VaultStorage::exchangeName(m_exchange));
268
269 if (showKeyDialog(this, title, name, apiKey, apiSecret, status)) {
270 if (vault.addKey(m_exchange, name, apiKey, apiSecret, status)) {
271 refreshTable();
272 } else {
273 QMessageBox::critical(this, "Error", "Failed to save key. Vault may be full (max 8 keys).");
274 }
275 }
276}

278void ExchangePanel::onEditKey() {
279 int slot = selectedSlot();
280 if (slot < 0) return;
281
282 auto& vault = VaultStorage::instance();
283 auto key = vault.getKey(m_exchange, slot);
284 if (!key.inUse) return;
285
286 QString name = key.name, apiKey = key.apiKey, apiSecret = key.apiSecret;
287 VaultKeyStatus status = static_cast<VaultKeyStatus>(key.status);
288 QString title = QString("Edit %1 API Key").arg(VaultStorage::exchangeName(m_exchange));
289
290 if (showKeyDialog(this, title, name, apiKey, apiSecret, status, true)) {
291 if (vault.updateKey(m_exchange, slot, name, apiKey, apiSecret, status)) {
292 refreshTable();
293 } else {
294 QMessageBox::critical(this, "Error", "Failed to update key.");
295 }
296 }
297}

299void ExchangePanel::onDeleteKey() {
300 int slot = selectedSlot();
301 if (slot < 0) return;
302
303 auto& vault = VaultStorage::instance();
304 auto key = vault.getKey(m_exchange, slot);
305
306 auto reply = QMessageBox::question(this, "Delete Key",
307 QString("Delete key '%1' from %2?\n\nThis cannot be undone.")
308 .arg(key.name).arg(VaultStorage::exchangeName(m_exchange)),
309 QMessageBox::Yes | QMessageBox::No, QMessageBox::No);
310
311 if (reply == QMessageBox::Yes) {
312 vault.deleteKey(m_exchange, slot);
313 refreshTable();
314 }

426void ExchangeKeysTab::onLockUnlock() {
427 auto& vault = VaultStorage::instance();
428
429 if (vault.isLoaded()) {
430 // Lock
431 vault.lock();
432 } else {
433 // Unlock (re-init from disk)
434 if (!vault.init()) {
435 QMessageBox::warning(this, "Vault Error",
436 "Could not unlock vault.\n\n"
437 "The vault file may not exist yet — add a key to create it.\n"
438 "File: " + VaultStorage::vaultFilePath());
439 // Still mark as loaded for first-time use (empty vault)
440 }
441 }
442
443 updateLockState();
444 refreshAll();
