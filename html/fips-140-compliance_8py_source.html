46def bytes_to_bits(data: bytes) -> list:
47 """Convert bytes to list of bits."""
48 bits = []
49 for byte in data:
50 for i in range(7, -1, -1):
51 bits.append((byte >> i) & 1)
52 return bits

55def get_random_from_node(host: str, port: int, num_bytes: int) -> bytes:
56 """Try to get random bytes from node RPC."""
57 payload = json.dumps({
58 "jsonrpc": "2.0", "id": 1,
59 "method": "getrandom",
60 "params": [num_bytes],
61 })
62 try:
63 conn = http.client.HTTPConnection(host, port, timeout=5)
64 conn.request("POST", "/", payload, {"Content-Type": "application/json"})
65 resp = conn.getresponse()
66 data = json.loads(resp.read().decode())
67 conn.close()
68 if "result" in data and data["result"]:
69 return bytes.fromhex(data["result"])
70 except Exception:
71 pass
72 return None

83def monobit_test(bits: list) -> dict:
84 """FIPS 140-2 Monobit Test.
85 Count of 1-bits in 20000 bits should be 9725..10275.
86 """
87 ones = sum(bits)
88 passed = 9725 <= ones <= 10275
89 return {
90 "test": "monobit",
91 "ones": ones,
92 "zeros": len(bits) - ones,
93 "range": "9725-10275",
94 "passed": passed,
95 }

98def poker_test(bits: list) -> dict:
99 """FIPS 140-2 Poker Test.
100 Divide 20000 bits into 5000 4-bit groups.
101 Compute chi-squared: X = (16/5000) * sum(f_i^2) - 5000
102 Must be 2.16 < X < 46.17
103 """
104 if len(bits) < 20000:
105 return {"test": "poker", "passed": False, "error": "insufficient bits"}
106
107 # Count frequencies of each 4-bit pattern
108 freq = [0] * 16
109 for i in range(0, 20000, 4):
110 nibble = (bits[i] << 3) | (bits[i+1] << 2) | (bits[i+2] << 1) | bits[i+3]
111 freq[nibble] += 1
112
113 # Chi-squared
114 sum_sq = sum(f * f for f in freq)
115 X = (16.0 / 5000.0) * sum_sq - 5000.0
116
117 passed = 2.16 < X < 46.17
118
119 return {
120 "test": "poker",
121 "chi_squared": round(X, 4),
122 "range": "2.16-46.17",
123 "passed": passed,
124 "freq_distribution": freq,
125 }

128def runs_test(bits: list) -> dict:
129 """FIPS 140-2 Runs Test.
130 Count runs (sequences of same bit) of length 1-6+.
131 Each length must fall within specified range.
132 """
133 # FIPS 140-2 ranges for 20000 bits
134 ranges = {
135 1: (2315, 2685),
136 2: (1114, 1386),
137 3: (527, 723),
138 4: (240, 384),
139 5: (103, 209),
140 6: (103, 209), # 6+ grouped
141 }
142
143 runs_0 = {i: 0 for i in range(1, 7)} # runs of 0s
144 runs_1 = {i: 0 for i in range(1, 7)} # runs of 1s
145
146 current_bit = bits[0]
147 run_length = 1
148
149 for i in range(1, len(bits)):
150 if bits[i] == current_bit:
151 run_length += 1
152 else:
153 # Record the run
154 capped = min(run_length, 6)
155 if current_bit == 0:
156 runs_0[capped] += 1
157 else:
158 runs_1[capped] += 1
159 current_bit = bits[i]
160 run_length = 1
161
162 # Record final run
163 capped = min(run_length, 6)
164 if current_bit == 0:
165 runs_0[capped] += 1
166 else:
167 runs_1[capped] += 1
168
169 # Check ranges
170 all_passed = True
171 details = []
172 for length in range(1, 7):
173 low, high = ranges[length]
174 count_0 = runs_0[length]
175 count_1 = runs_1[length]
176 pass_0 = low <= count_0 <= high
177 pass_1 = low <= count_1 <= high
178 if not pass_0 or not pass_1:
179 all_passed = False
180 details.append({
181 "length": length if length < 6 else "6+",
182 "runs_of_0": count_0,
183 "runs_of_1": count_1,
184 "range": f"{low}-{high}",
185 "passed": pass_0 and pass_1,
186 })
187
188 return {
189 "test": "runs",
190 "passed": all_passed,
191 "details": details,
192 }

195def long_runs_test(bits: list) -> dict:
196 """FIPS 140-2 Long Runs Test.
197 No run of 26 or more consecutive identical bits.
198 """
199 max_run = 1
200 current_run = 1
201
202 for i in range(1, len(bits)):
203 if bits[i] == bits[i-1]:
204 current_run += 1
205 max_run = max(max_run, current_run)
206 else:
207 current_run = 1
208
209 passed = max_run < 26
210
211 return {
212 "test": "long_runs",
213 "max_run_length": max_run,
214 "threshold": 26,
215 "passed": passed,
216 }

219def autocorrelation_test(bits: list, shift: int = 1) -> dict:
220 """Additional: autocorrelation test (not FIPS required but useful)."""
221 n = len(bits)
222 matches = sum(1 for i in range(n - shift) if bits[i] == bits[i + shift])
223 ratio = matches / (n - shift)
224 # Should be close to 0.5
225 deviation = abs(ratio - 0.5)
226 passed = deviation < 0.02 # Within 2%
227
228 return {
229 "test": "autocorrelation",
230 "shift": shift,
231 "match_ratio": round(ratio, 6),
232 "deviation_from_half": round(deviation, 6),
233 "passed": passed,
234 }
