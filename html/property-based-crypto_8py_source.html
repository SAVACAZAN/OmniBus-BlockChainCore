71# BIP-32 derivation (simplified, non-hardened only)
72# ---------------------------------------------------------------------------
73def bip32_master(seed: bytes) -> tuple:
74 """Derive master key from seed. Returns (key_bytes, chain_code)."""
75 I = hmac_sha512(b"Bitcoin seed", seed)
76 return I[:32], I[32:]

79def bip32_derive_child(parent_key: bytes, parent_chain: bytes, index: int) -> tuple:
80 """Derive non-hardened child. Simplified — uses HMAC-SHA512."""
81 # For non-hardened: data = pubkey_compressed || index
82 # Simplified: we use the private key hash as "pubkey" stand-in
83 pubkey_stand_in = sha256(parent_key)[:33] # 33 bytes like compressed pubkey
84 data = pubkey_stand_in + struct.pack(">I", index)
85 I = hmac_sha512(parent_chain, data)
86 child_key_int = (int.from_bytes(I[:32], "big") + int.from_bytes(parent_key, "big")) % SECP256K1_N
87 child_key = child_key_int.to_bytes(32, "big")
88 child_chain = I[32:]
89 return child_key, child_chain

92def bip32_derive_path(seed: bytes, path: list) -> tuple:
93 """Derive key at path [0, 1, ...] from seed."""
94 key, chain = bip32_master(seed)
95 for idx in path:
96 key, chain = bip32_derive_child(key, chain, idx)
97 return key, chain

107 def record(self, name: str, passed: bool, detail: str = ""):
108 if name not in self.tests:
109 self.tests[name] = {"pass": 0, "fail": 0, "errors": []}
110 if passed:
111 self.tests[name]["pass"] += 1
112 else:
113 self.tests[name]["fail"] += 1
114 if detail:
115 self.tests[name]["errors"].append(detail)
116
117 def summary(self) -> dict:
118 total_pass = sum(t["pass"] for t in self.tests.values())
119 total_fail = sum(t["fail"] for t in self.tests.values())
120 return {
121 "categories": {k: {"pass": v["pass"], "fail": v["fail"]}
122 for k, v in self.tests.items()},
123 "total_pass": total_pass,
124 "total_fail": total_fail,
125 "verdict": "PASS" if total_fail == 0 else "FAIL",
126 }

129def test_sign_verify_roundtrip(stats: Stats, iterations: int):
130 """sign(msg, sk) then verify(msg, sig, pk) must pass for ANY random msg+sk."""
131 if not HAS_COINCURVE:
132 print(f" {YELLOW}[SKIP] coincurve not available — using HMAC-SHA256 as sign/verify stand-in{RESET}")
133 for i in range(iterations):
134 sk = random_privkey()
135 msg = secrets.token_bytes(secrets.randbelow(256) + 1)
136 # HMAC-based "signature"
137 sig = hmac.new(sk, msg, hashlib.sha256).digest()
138 # Verify: recompute
139 verify = hmac.new(sk, msg, hashlib.sha256).digest()
140 stats.record("sign_verify_roundtrip", sig == verify,
141 f"iter {i}: HMAC mismatch")
142 return
143
144 for i in range(iterations):
145 sk_bytes = random_privkey()
146 msg = secrets.token_bytes(secrets.randbelow(256) + 1)
147 msg_hash = sha256(msg)
148 try:
149 sk = coincurve.PrivateKey(sk_bytes)
150 sig = sk.sign(msg_hash, hasher=None)
151 pk = sk.public_key
152 ok = pk.verify(sig, msg_hash, hasher=None)
153 stats.record("sign_verify_roundtrip", ok,
154 f"iter {i}: verify returned False")
155 except Exception as exc:
156 stats.record("sign_verify_roundtrip", False, f"iter {i}: {exc}")
157
158 if (i + 1) % 1000 == 0:
159 print(f" {CYAN}sign/verify: {i+1}/{iterations}{RESET}")

218def test_corrupted_signature(stats: Stats, iterations: int):
219 """verify must FAIL for corrupted signature (flip random bit)."""
220 if not HAS_COINCURVE:
221 for i in range(iterations):
222 sk = random_privkey()
223 msg = secrets.token_bytes(32)
224 sig = bytearray(hmac.new(sk, msg, hashlib.sha256).digest())
225 bit_pos = secrets.randbelow(len(sig) * 8)
226 sig[bit_pos // 8] ^= 1 << (bit_pos % 8)
227 original = hmac.new(sk, msg, hashlib.sha256).digest()
228 stats.record("corrupted_sig", bytes(sig) != original,
229 f"iter {i}: corrupted sig still matches")
230 return
231
232 for i in range(iterations):
233 sk = coincurve.PrivateKey(random_privkey())
234 msg = sha256(secrets.token_bytes(32))
235 sig = bytearray(sk.sign(msg, hasher=None))
236 bit_pos = secrets.randbelow(len(sig) * 8)
237 sig[bit_pos // 8] ^= 1 << (bit_pos % 8)
238 try:
239 should_fail = sk.public_key.verify(bytes(sig), msg, hasher=None)
240 stats.record("corrupted_sig", not should_fail,
241 f"iter {i}: corrupted sig accepted")

256def test_bip32_path_consistency(stats: Stats, iterations: int):
257 """Deriving m/0/1 directly == deriving m/0 then /1."""
258 for i in range(iterations):
259 seed = secrets.token_bytes(64)
