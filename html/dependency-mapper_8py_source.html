39def scan_imports(filepath: Path) -> list[str]:
40 """Extract @import targets from a .zig file."""
41 try:
42 content = filepath.read_text(encoding="utf-8", errors="replace")
43 except Exception:
44 return []
45
46 pattern = re.compile(r'@import\s*\‍(\s*"([^"]+)"\s*\‍)')
47 return pattern.findall(content)

50def build_graph(source_dir: Path) -> tuple[dict, set]:
51 """Build dependency graph. Returns (graph, all_modules)."""
52 zig_files = sorted(source_dir.glob("*.zig"))
53 if not zig_files:
54 zig_files = sorted(source_dir.rglob("*.zig"))
55
56 graph = defaultdict(set) # module -> set of dependencies
57 all_modules = set()
58 external_deps = set()
59
60 for f in zig_files:
61 module_name = f.stem
62 all_modules.add(module_name)
63 imports = scan_imports(f)
64
65 for imp in imports:
66 if imp.endswith(".zig"):
67 dep_name = imp.replace(".zig", "")
68 graph[module_name].add(dep_name)
69 elif imp == "std" or imp == "builtin":
70 external_deps.add(imp)
71 else:
72 # Could be a package or relative path
73 dep_name = imp.split("/")[-1].replace(".zig", "")
74 graph[module_name].add(dep_name)
75
76 return dict(graph), all_modules, external_deps

79def find_cycles(graph: dict) -> list[list[str]]:
80 """Detect circular dependencies using DFS."""
81 cycles = []
82 visited = set()
83 rec_stack = set()
84 path = []
85
86 def dfs(node):
87 visited.add(node)
88 rec_stack.add(node)
89 path.append(node)
90
91 for neighbor in graph.get(node, set()):
92 if neighbor not in visited:
93 dfs(neighbor)
94 elif neighbor in rec_stack:
95 # Found a cycle
96 cycle_start = path.index(neighbor)
97 cycle = path[cycle_start:] + [neighbor]
98 cycles.append(cycle)
99
100 path.pop()
101 rec_stack.discard(node)
102
103 for node in graph:
104 if node not in visited:
105 dfs(node)
106
107 return cycles

110def compute_stats(graph: dict, all_modules: set) -> dict:
111 """Compute graph statistics."""
112 reverse_graph = defaultdict(set)
113 for src, deps in graph.items():
114 for dep in deps:
115 reverse_graph[dep].add(src)
116
117 stats = {}
118 for m in all_modules:
119 out_degree = len(graph.get(m, set()))
120 in_degree = len(reverse_graph.get(m, set()))
121 stats[m] = {
122 "out_degree": out_degree, # depends on N modules
123 "in_degree": in_degree, # used by N modules
124 "is_leaf": out_degree == 0 and in_degree > 0,
125 "is_root": in_degree == 0 and out_degree > 0,
126 "is_isolated": out_degree == 0 and in_degree == 0,
127 }
128 return stats

131def generate_dot(graph: dict, all_modules: set, cycles: list) -> str:
132 """Generate DOT format for Graphviz."""
133 cycle_edges = set()
134 for cycle in cycles:
135 for i in range(len(cycle) - 1):
136 cycle_edges.add((cycle[i], cycle[i + 1]))
137
138 lines = ['digraph OmniBusDeps {']
139 lines.append(' rankdir=LR;')
140 lines.append(' node [shape=box, style=filled, fillcolor="#e8f4f8", fontname="Consolas"];')
141 lines.append(' edge [color="#666666"];')
142 lines.append('')
143
144 for module in sorted(all_modules):
145 if module not in graph and not any(module in deps for deps in graph.values()):
146 lines.append(f' "{module}" [fillcolor="#ffe0e0"];') # isolated = red
147
148 for src, deps in sorted(graph.items()):
149 for dep in sorted(deps):
150 if dep in all_modules:
151 color = 'color="red", penwidth=2' if (src, dep) in cycle_edges else ''
152 lines.append(f' "{src}" -> "{dep}" [{color}];')
153
154 lines.append('}')
155 return '\n'.join(lines)

158def print_adjacency(graph: dict, stats: dict, all_modules: set):
159 """Print adjacency list as table."""
160 print(f"\n{BG_BLUE}{WHITE}{BOLD} OmniBus Dependency Map {RESET}\n")
161
162 print(f" {BOLD}{'Module':<35s} {'Out':>3s} {'In':>3s} {'Dependencies'}{RESET}")
163 print(f" {DIM}{'─' * 90}{RESET}")
164
165 for module in sorted(all_modules):
166 deps = sorted(graph.get(module, set()))
167 s = stats.get(module, {})
168 out_d = s.get("out_degree", 0)
169 in_d = s.get("in_degree", 0)
170
171 # Colour based on role
172 if s.get("is_isolated"):
173 mod_col = DIM
174 elif s.get("is_root"):
175 mod_col = YELLOW
176 elif s.get("is_leaf"):
177 mod_col = GREEN
178 else:
179 mod_col = CYAN
180
181 deps_str = ", ".join(deps[:8])
182 if len(deps) > 8:
183 deps_str += f" (+{len(deps) - 8} more)"
184
185 name = module if len(module) <= 35 else module[:32] + "..."
186 print(f" {mod_col}{name:<35s}{RESET} {out_d:3d} {in_d:3d} {DIM}{deps_str}{RESET}")
187
188 print(f" {DIM}{'─' * 90}{RESET}")

204def print_summary(all_modules: set, graph: dict, stats: dict, external_deps: set):
205 """Print summary stats."""
206 total_edges = sum(len(deps) for deps in graph.values())
207 roots = [m for m, s in stats.items() if s.get("is_root")]
208 leaves = [m for m, s in stats.items() if s.get("is_leaf")]
209 isolated = [m for m, s in stats.items() if s.get("is_isolated")]
210
211 print(f"\n {MAGENTA}{BOLD}Summary{RESET}")
212 print(f" {DIM}{'─' * 40}{RESET}")
213 print(f" Total modules: {BOLD}{len(all_modules)}{RESET}")
214 print(f" Total edges: {total_edges}")
215 print(f" Root modules: {len(roots)} {DIM}(depend on others but nobody depends on them){RESET}")
216 print(f" Leaf modules: {len(leaves)} {DIM}(depended upon but depend on nothing){RESET}")
217 print(f" Isolated: {len(isolated)} {DIM}(no internal deps){RESET}")
218 print(f" External deps: {', '.join(sorted(external_deps)) or 'none'}")
219
220 if roots:
221 print(f"\n {YELLOW}Root modules:{RESET} {', '.join(sorted(roots)[:10])}")
222 if leaves:
223 print(f" {GREEN}Leaf modules:{RESET} {', '.join(sorted(leaves)[:10])}")
