170def run_zig_build_test(name: str, verbose: bool = False, timeout: int = 120) -> TestResult:
171 """Run a zig build test target."""
172 start = datetime.now()
173 cmd = ["zig", "build", name]
174
175 try:
176 result = subprocess.run(
177 cmd, cwd=ROOT,
178 capture_output=True, text=True,
179 timeout=timeout
180 )
181 duration = int((datetime.now() - start).total_seconds() * 1000)
182
183 passed = result.returncode == 0
184 output = result.stdout
185 stderr = result.stderr
186
187 # Parse test counts from output
188 test_count = 0
189 skip_count = 0
190 for line in (output + stderr).splitlines():
191 if "test" in line.lower() and "passed" in line.lower():
192 try:
193 test_count = int(re.search(r'(\d+)\s+passed', line).group(1))
194 except Exception:
195 pass
196 if "skipped" in line.lower():
197 try:
198 skip_count = int(re.search(r'(\d+)\s+skipped', line).group(1))
199 except Exception:
200 pass
201
202 # Extract failure details
203 fail_details = ""
204 if not passed:
205 lines = stderr.strip().split('\n')
206 fail_details = '\n'.join(lines[-10:])
207
208 return TestResult(
209 name=name, command=' '.join(cmd), passed=passed,
210 output=output, stderr=stderr,
211 duration_ms=duration, test_count=test_count,
212 skip_count=skip_count, fail_details=fail_details
213 )
214
215 except subprocess.TimeoutExpired:
216 duration = int((datetime.now() - start).total_seconds() * 1000)
217 return TestResult(
218 name=name, command=' '.join(cmd), passed=False,
219 output="", stderr=f"TIMEOUT after {timeout}s",
220 duration_ms=duration, fail_details=f"Timeout after {timeout}s"
221 )
222 except FileNotFoundError:
223 return TestResult(
224 name=name, command=' '.join(cmd), passed=False,
225 output="", stderr="zig not found in PATH",
226 duration_ms=0, fail_details="zig binary not found"
227 )
228 except Exception as e:
229 return TestResult(
230 name=name, command=' '.join(cmd), passed=False,
231 output="", stderr=str(e),
232 duration_ms=0, fail_details=str(e)
233 )

236def run_single_module_test(module_name: str, verbose: bool = False) -> TestResult:
237 """Run test for a single .zig file using zig test."""
238 # Try core/ first, then test/
239 module_path = CORE / f"{module_name}.zig"
240 if not module_path.exists():
241 module_path = TEST_DIR / f"{module_name}.zig"
242 if not module_path.exists():
243 return TestResult(
244 name=module_name, command="", passed=False,
245 output="", stderr=f"Module not found: {module_name}.zig",
246 fail_details=f"File not found in core/ or test/"
247 )
248
249 start = datetime.now()
250 cmd = ["zig", "test", str(module_path)]
251
252 try:
253 result = subprocess.run(
254 cmd, cwd=ROOT,
255 capture_output=True, text=True,
256 timeout=60
257 )
258 duration = int((datetime.now() - start).total_seconds() * 1000)
259
260 return TestResult(
261 name=module_name, command=' '.join(cmd),
262 passed=result.returncode == 0,
263 output=result.stdout, stderr=result.stderr,
264 duration_ms=duration,
265 fail_details=result.stderr[-500:] if result.returncode != 0 else ""
266 )
267 except subprocess.TimeoutExpired:
268 return TestResult(
269 name=module_name, command=' '.join(cmd), passed=False,
270 output="", stderr="TIMEOUT", duration_ms=60000,
271 fail_details="Timeout after 60s"
272 )
273 except Exception as e:
274 return TestResult(
275 name=module_name, command=' '.join(cmd), passed=False,
276 output="", stderr=str(e), duration_ms=0,
277 fail_details=str(e)

314def print_summary(group_results: Dict[str, TestResult], extra_results: List[TestResult]):
315 """Print full test summary."""
316 all_results = list(group_results.values()) + extra_results
317
318 passed = sum(1 for r in all_results if r.passed)
319 failed = len(all_results) - passed
320 total_ms = sum(r.duration_ms for r in all_results)
321
322 print(f"\n{'=' * 60}")
323 print(f" {BOLD}TEST SUMMARY{W}")
324 print(f"{'=' * 60}")
325
326 # Groups
327 for name, result in group_results.items():
328 status = f"{G}PASS{W}" if result.passed else f"{R}FAIL{W}"
329 print(f" [{status}] {name:<20} ({result.duration_ms}ms)")
330
331 # Extra (test/ dir)
332 if extra_results:
333 print(f"\n {C}test/ directory:{W}")
334 for r in extra_results:
335 status = f"{G}PASS{W}" if r.passed else f"{R}FAIL{W}"
336 print(f" [{status}] {r.name:<20} ({r.duration_ms}ms)")
337
338 print(f"\n{'-' * 60}")
339 total_str = f"{G}{passed} passed{W}" if failed == 0 else f"{G}{passed} passed{W}, {R}{failed} failed{W}"
340 print(f" Total: {len(all_results)} | {total_str} | {total_ms}ms")
341 print(f"{'=' * 60}\n")
342
343 return failed == 0
