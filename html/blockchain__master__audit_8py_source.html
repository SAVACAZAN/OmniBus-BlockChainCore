79def run_tool(tool: dict, json_dir: Path = None, html_dir: Path = None,
80 dot_dir: Path = None) -> dict:
81 """Run a single audit tool."""
82 script = TOOLS / tool["script"]
83 if not script.exists():
84 return {"name": tool["name"], "status": "MISSING", "output": f"{script} not found"}
85
86 cmd = [sys.executable, str(script)] + tool["args"]
87
88 # Add JSON export
89 if json_dir and tool.get("json_arg"):
90 json_file = json_dir / f"{Path(tool['script']).stem}.json"
91 cmd += [tool["json_arg"], str(json_file)]
92
93 # Add HTML export
94 if html_dir and tool.get("html_arg"):
95 html_file = html_dir / f"{Path(tool['script']).stem}.html"
96 cmd += [tool["html_arg"], str(html_file)]
97
98 # Add DOT export
99 if dot_dir and tool.get("dot_arg"):
100 dot_file = dot_dir / f"{Path(tool['script']).stem}.dot"
101 cmd += [tool["dot_arg"], str(dot_file)]
102
103 print(f"\n {BOLD}{C}[{tool['name']}]{W}")
104 print(f" Command: {' '.join(cmd)}")
105 print(f" {'-' * 60}")
106
107 try:
108 result = subprocess.run(
109 cmd, capture_output=False, text=True,
110 timeout=tool["timeout"], cwd=ROOT
111 )
112 return {
113 "name": tool["name"],
114 "status": "PASS" if result.returncode == 0 else "FAIL",
115 "returncode": result.returncode,
116 }
117 except subprocess.TimeoutExpired:
118 print(f" {R}TIMEOUT after {tool['timeout']}s{W}")
119 return {"name": tool["name"], "status": "TIMEOUT"}
120 except Exception as e:
121 print(f" {R}ERROR: {e}{W}")
122 return {"name": tool["name"], "status": "ERROR", "error": str(e)}

125def ordinal(n: int) -> str:
126 if 11 <= (n % 100) <= 13:
127 return f"{n}th"
128 return f"{n}" + {1: "st", 2: "nd", 3: "rd"}.get(n % 10, "th")
