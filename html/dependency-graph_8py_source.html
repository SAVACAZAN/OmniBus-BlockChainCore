26def extract_imports(filepath: str) -> List[str]:
27 imports = []
28 with open(filepath, "r", encoding="utf-8", errors="ignore") as f:
29 content = f.read()
30 # Match @import("core/module.zig") or @import("module.zig")
31 for m in re.finditer(r'@import\‍("([^"]+)"\‍)', content):
32 imp = m.group(1)
33 base = os.path.basename(imp)
34 if base.endswith(".zig"):
35 imports.append(base)
36 return imports

39def build_graph(core_dir: str) -> Tuple[List[str], List[Dict[str, str]]]:
40 files = sorted([f for f in os.listdir(core_dir) if f.endswith(".zig")])
41 nodes = [f for f in files]
42 edges: List[Dict[str, str]] = []
43 for f in files:
44 fpath = os.path.join(core_dir, f)
45 imports = extract_imports(fpath)
46 for imp in imports:
47 if imp in files and imp != f:
48 edges.append({"source": f, "target": imp})
49 return nodes, edges
