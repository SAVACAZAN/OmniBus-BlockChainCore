69bool ConsoleTab::eventFilter(QObject* obj, QEvent* event) {
70 if (obj == m_input && event->type() == QEvent::KeyPress) {
71 auto* ke = static_cast<QKeyEvent*>(event);
72 if (ke->key() == Qt::Key_Up) {
73 if (m_historyIndex < m_history.size() - 1) {
74 m_historyIndex++;

84 m_historyIndex = -1;

93void ConsoleTab::executeCommand() {
94 QString cmd = m_input->text().trimmed();
95 if (cmd.isEmpty()) return;
96
97 m_history.append(cmd);
98 m_historyIndex = -1;
99 m_input->clear();
100
101 appendOutput("> " + cmd, "#4a90d9");
102
103 // Parse: first word = method, rest = params
104 QStringList parts = cmd.split(' ', Qt::SkipEmptyParts);
105 QString method = parts.takeFirst();
106 QJsonArray params;
107
108 for (const auto& p : parts) {
109 bool isNum = false;
110 int intVal = p.toInt(&isNum);
111 if (isNum) {
112 params.append(intVal);
113 continue;
114 }
115 double dblVal = p.toDouble(&isNum);
116 if (isNum) {
117 params.append(dblVal);
118 continue;
119 }
120 params.append(p);
121 }
122
123 NodeService::instance().rpc()->rawRequest(method, params,
124 [this, method](const QJsonValue& result, const QString& err) {
125 if (!err.isEmpty()) {
126 appendOutput("Error: " + err, "#d94a4a");
127 } else {
128 QJsonDocument doc;
129 if (result.isObject())
130 doc = QJsonDocument(result.toObject());
131 else if (result.isArray())
132 doc = QJsonDocument(result.toArray());
133 else {
134 appendOutput(result.toVariant().toString(), "#00b3a4");
135 return;
136 }
137 appendOutput(doc.toJson(QJsonDocument::Indented), "#00b3a4");
138 }
139 }
140 );
141}
