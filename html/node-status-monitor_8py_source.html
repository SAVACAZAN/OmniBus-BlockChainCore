41def rpc_call(host: str, port: int, method: str, params=None, timeout: int = 5):
42 """Send a JSON-RPC 2.0 request and return the result (or None on error)."""
43 payload = json.dumps({
44 "jsonrpc": "2.0",
45 "id": 1,
46 "method": method,
47 "params": params or [],
48 })
49 try:
50 conn = http.client.HTTPConnection(host, port, timeout=timeout)
51 conn.request("POST", "/", body=payload, headers={
52 "Content-Type": "application/json",
53 })
54 resp = conn.getresponse()
55 data = json.loads(resp.read().decode())
56 conn.close()
57 if "result" in data:
58 return data["result"]
59 return None
60 except Exception:
61 return None
