68 def analyze_file(self, filepath: Path) -> FileMetrics:
69 """Analizează un singur fișier."""
70 try:
71 with open(filepath, 'r', encoding='utf-8') as f:
72 content = f.read()
73 lines = content.split('\n')
74 except Exception:
75 return None
76
77 total_lines = len(lines)
78 blank_lines = 0
79 comment_lines = 0
80 code_lines = 0
81 functions = 0
82 structs = 0
83 tests = 0
84
85 in_multiline_comment = False
86
87 for line in lines:
88 stripped = line.strip()
89
90 # Blank line
91 if not stripped:
92 blank_lines += 1
93 continue
94
95 # Comment
96 if stripped.startswith('//'):
97 comment_lines += 1
98 continue
99
100 if stripped.startswith('/*'):
101 in_multiline_comment = True
102
103 if in_multiline_comment:
104 comment_lines += 1
105 if '*/' in stripped:
106 in_multiline_comment = False
107 continue
108
109 # Code line
110 code_lines += 1
111
112 # Count functions
113 if re.match(r'^\s*(pub\s+)?fn\s+\w+', line):
114 functions += 1
115
116 # Count structs
117 if re.match(r'^\s*(pub\s+)?const\s+\w+\s*=\s*struct', line):
118 structs += 1
119
120 # Count tests
121 if re.match(r'^\s*test\s+["\']', line) or re.match(r'^\s*test\s*\{', line):
122 tests += 1
123
124 # Calculate complexity (simplified)
125 complexity = self._calculate_complexity(content)
126
127 return FileMetrics(
128 path=str(filepath.relative_to(self.project_root)),
129 lines_total=total_lines,
130 lines_code=code_lines,
131 lines_blank=blank_lines,
132 lines_comment=comment_lines,
133 functions=functions,
134 structs=structs,
135 tests=tests,
136 complexity_score=complexity
137 )

139 def _calculate_complexity(self, content: str) -> int:
140 """Calculează un scor de complexitate aproximativ."""
141 complexity = 0
142
143 # Control flow statements
144 complexity += len(re.findall(r'\b(if|else|while|for|switch|return)\b', content))
145
146 # Function calls (indică modularity)
147 complexity += len(re.findall(r'\w+\‍(', content)) // 5
148
149 # Error handling
150 complexity += len(re.findall(r'try\s+|catch\s+|err\s*!=\s*null', content))
151
152 return complexity

154 def analyze_project(self) -> ProjectMetrics:
155 """Analizează întregul proiect."""
156 zig_files = list(self.project_root.rglob('*.zig'))
157 zig_files = [
158 f for f in zig_files
159 if not any(part.startswith('.') or part in ['zig-out', '.zig-cache']
160 for part in f.parts)
161 ]
162
163 for filepath in zig_files:
164 metrics = self.analyze_file(filepath)
165 if metrics:
166 self.file_metrics.append(metrics)
167
168 # Calculate totals
169 total_lines = sum(f.lines_total for f in self.file_metrics)
170 total_code = sum(f.lines_code for f in self.file_metrics)
171 total_blank = sum(f.lines_blank for f in self.file_metrics)
172 total_comment = sum(f.lines_comment for f in self.file_metrics)
173 total_functions = sum(f.functions for f in self.file_metrics)
174 total_structs = sum(f.structs for f in self.file_metrics)
175 total_tests = sum(f.tests for f in self.file_metrics)
176
177 # Find largest files
178 largest = sorted(
179 [(f.path, f.lines_total) for f in self.file_metrics],
180 key=lambda x: x[1],
181 reverse=True
182 )[:10]
183
184 # Calculate ratios
185 test_ratio = (total_tests * 10) / total_code if total_code > 0 else 0
186 avg_complexity = sum(f.complexity_score for f in self.file_metrics) / len(self.file_metrics) if self.file_metrics else 0
187
188 return ProjectMetrics(
189 total_files=len(self.file_metrics),
190 total_lines=total_lines,
191 total_code_lines=total_code,
192 total_blank_lines=total_blank,
193 total_comment_lines=total_comment,
194 total_functions=total_functions,
195 total_structs=total_structs,
196 total_tests=total_tests,
197 avg_file_size=total_lines / len(self.file_metrics) if self.file_metrics else 0,
198 largest_files=largest,
199 test_to_code_ratio=test_ratio,
200 complexity_avg=avg_complexity
201 )
