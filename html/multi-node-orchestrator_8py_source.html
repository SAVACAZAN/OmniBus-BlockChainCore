67def is_pid_alive(pid: int) -> bool:
68 """Check if a process is still running."""
69 try:
70 if os.name == "nt":
71 result = subprocess.run(
72 ["tasklist", "/FI", f"PID eq {pid}"],
73 capture_output=True, text=True, timeout=5,
74 )
75 return str(pid) in result.stdout
76 else:
77 os.kill(pid, 0)
78 return True
79 except (OSError, subprocess.TimeoutExpired):
80 return False

83def kill_process(pid: int):
84 """Kill a process by PID."""
85 try:
86 if os.name == "nt":
87 subprocess.run(["taskkill", "/F", "/PID", str(pid)],
88 capture_output=True, timeout=10)
89 else:
90 os.kill(pid, signal.SIGTERM)
91 time.sleep(1)
92 try:
93 os.kill(pid, signal.SIGKILL)
94 except OSError:
95 pass
96 except Exception:
97 pass

100def start_nodes(binary: Path, count: int, base_port: int, rpc_base: int):
101 """Start 1 seed + (count-1) miner nodes."""
102 print(f"\n{BG_BLUE}{WHITE}{BOLD} OmniBus Multi-Node Orchestrator {RESET}")
103 print(f" {DIM}{'─' * 55}{RESET}")
104 print(f" {CYAN}Binary:{RESET} {binary}")
105 print(f" {CYAN}Nodes:{RESET} {count} (1 seed + {count - 1} miners)")
106 print(f" {CYAN}Base port:{RESET} {base_port}")
107 print(f" {CYAN}RPC base:{RESET} {rpc_base}")
108 print(f" {DIM}{'─' * 55}{RESET}\n")
109
110 work_dir = binary.parent.parent.parent
111 nodes = []
112
113 # Start seed node
114 seed_port = base_port
115 seed_rpc = rpc_base
116 seed_id = "seed-1"
117 print(f" {MAGENTA}[1/{count}]{RESET} Starting {BOLD}seed{RESET} node {seed_id} "
118 f"(P2P: {seed_port}, RPC: {seed_rpc})")
119
120 seed_cmd = [
121 str(binary),
122 "--mode", "seed",
123 "--node-id", seed_id,
124 "--port", str(seed_port),
125 ]
126
127 log_dir = work_dir / "logs"
128 log_dir.mkdir(exist_ok=True)
129
130 seed_log = open(log_dir / f"{seed_id}.log", "w")
131 seed_proc = subprocess.Popen(
132 seed_cmd, cwd=str(work_dir),
133 stdout=seed_log, stderr=subprocess.STDOUT,
134 )
135 nodes.append({
136 "node_id": seed_id, "mode": "seed", "pid": seed_proc.pid,
137 "port": seed_port, "rpc_port": seed_rpc,
138 })
139 print(f" {GREEN}[OK]{RESET} PID {seed_proc.pid}")
140
141 # Give seed node time to bind
142 time.sleep(1)
143
144 # Start miner nodes
145 for i in range(1, count):
146 miner_port = base_port + i
147 miner_rpc = rpc_base + i
148 miner_id = f"miner-{i}"
149 print(f" {MAGENTA}[{i + 1}/{count}]{RESET} Starting {BOLD}miner{RESET} node {miner_id} "
150 f"(P2P: {miner_port}, RPC: {miner_rpc})")
151
152 miner_cmd = [
153 str(binary),
154 "--mode", "miner",
155 "--node-id", miner_id,
156 "--port", str(miner_port),
157 "--seed-host", "127.0.0.1",
158 "--seed-port", str(seed_port),
159 ]
160
161 miner_log = open(log_dir / f"{miner_id}.log", "w")
162 miner_proc = subprocess.Popen(
163 miner_cmd, cwd=str(work_dir),
164 stdout=miner_log, stderr=subprocess.STDOUT,
165 )
166 nodes.append({
167 "node_id": miner_id, "mode": "miner", "pid": miner_proc.pid,
168 "port": miner_port, "rpc_port": miner_rpc,
169 })
170 print(f" {GREEN}[OK]{RESET} PID {miner_proc.pid}")
171 time.sleep(0.3)
172
173 save_pids(nodes)
174 print(f"\n {DIM}{'─' * 55}{RESET}")
175 print(f" {GREEN}{BOLD}All {count} nodes started.{RESET}")
176 print(f" {DIM}PID file: {PID_FILE}{RESET}")
177 print(f" {DIM}Logs dir: {log_dir}{RESET}")
178 print(f" {DIM}Stop all: python {Path(__file__).name} --stop{RESET}\n")
179
180 return nodes

183def stop_nodes():
184 """Stop all nodes tracked in PID file."""
185 nodes = load_pids()
186 if not nodes:
187 print(f" {YELLOW}No running nodes found (PID file empty or missing).{RESET}")
188 return
189
190 print(f"\n{CYAN}{BOLD} Stopping {len(nodes)} nodes...{RESET}")
191 print(f" {DIM}{'─' * 55}{RESET}")
192
193 for node in nodes:
194 pid = node["pid"]
195 alive = is_pid_alive(pid)
196 status = f"{GREEN}running{RESET}" if alive else f"{DIM}already stopped{RESET}"
197 print(f" {node['node_id']:12s} PID {pid:6d} {status}", end="")
198
199 if alive:
200 kill_process(pid)
201 print(f" -> {RED}killed{RESET}")
202 else:
203 print()
204
205 PID_FILE.unlink(missing_ok=True)
206 print(f"\n {GREEN}{BOLD}All nodes stopped.{RESET}\n")

229def monitor_loop(interval: int):
230 """Continuously monitor node status."""
231 print(f"{CYAN}Monitoring nodes (interval: {interval}s). Ctrl+C to stop.{RESET}")
232 try:
233 while True:
234 os.system("cls" if os.name == "nt" else "clear")
235 status_nodes()
236 nodes = load_pids()
237 dead = [n for n in nodes if not is_pid_alive(n["pid"])]
238 if dead:
239 for n in dead:
240 print(f" {RED}WARNING:{RESET} {n['node_id']} (PID {n['pid']}) has stopped!")
241 time.sleep(interval)
242 except KeyboardInterrupt:
243 print(f"\n{YELLOW}Monitoring stopped.{RESET}")
