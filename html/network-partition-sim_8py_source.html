2"""
3network-partition-sim.py
4
5Simulate a network partition on a BlockChainCore-like network:
6 1. Launch 6 virtual/minimal nodes.
7 2. Partition into 2 groups of 3.
8 3. Each group mines independently.
9 4. Reconnect and verify:
10 - chain reorganisation correctness
11 - no double-spend survived
12 - no lost transactions
13
14This script uses RPC endpoints (assumed local on ports 18332-18337) to
15orchestrate the simulation. It can also run in pure-simulation mode if
16no live nodes are available.
17
18Outputs: network-partition-sim.json
19"""

58def rpc(url: str, method: str, params: Any = None, auth: tuple[str, str] | None = None, timeout: float = 5.0) -> Any:
59 payload = {"jsonrpc": "2.0", "method": method, "id": random.randint(1, 100000)}
60 if params is not None:
61 payload["params"] = params
62 try:
63 resp = requests.post(url, json=payload, auth=auth, timeout=timeout)
64 data = resp.json()
65 return data.get("result")
66 except Exception as exc:
67 return {"_error": str(exc)}

73class SimNode:
74 def __init__(self, node_id: int) -> None:
75 self.node_id = node_id
76 self.chain: list[dict[str, Any]] = [{"height": 0, "hash": f"genesis-{node_id}", "txs": []}]
77 self.peers: set[int] = set()
78 self.mempool: list[dict[str, Any]] = []
79
80 def mine_block(self, txs: list[dict[str, Any]] | None = None) -> dict[str, Any]:
81 prev = self.chain[-1]
82 block = {
83 "height": prev["height"] + 1,
84 "hash": f"block-{self.node_id}-{prev['height'] + 1}-{random.randint(0, 999999)}",
85 "prev": prev["hash"],
86 "txs": txs or [],
87 }
88 self.chain.append(block)
89 return block
90
91 def tip(self) -> dict[str, Any]:
92 return self.chain[-1]

95def simulate_partition() -> dict[str, Any]:
96 log_info("Starting pure simulation with 6 nodes …")
97 nodes = [SimNode(i) for i in range(6)]
98
99 # Fully connected mesh initially
100 for n in nodes:
101 for p in range(6):
102 if p != n.node_id:
103 n.peers.add(p)
104
105 # Seed some transactions
106 shared_txs = [{"txid": f"tx-shared-{i}", "value": 100} for i in range(5)]
107 for n in nodes:
108 n.mempool = list(shared_txs)
109
110 # Partition: group A = 0,1,2 ; group B = 3,4,5
111 group_a = [0, 1, 2]
112 group_b = [3, 4, 5]
113 for i in group_a:
114 nodes[i].peers = set(group_a) - {i}
115 for i in group_b:
116 nodes[i].peers = set(group_b) - {i}
117
118 log_info("Partitioned. Mining in both groups …")
119 # Group A mines 4 blocks
120 for _ in range(4):
121 for i in group_a:
122 nodes[i].mine_block(txs=list(nodes[i].mempool))
123 # Group B mines 6 blocks (longer chain)
124 for _ in range(6):
125 for i in group_b:
126 nodes[i].mine_block(txs=list(nodes[i].mempool))
127
128 tip_a = nodes[group_a[0]].tip()["height"]
129 tip_b = nodes[group_b[0]].tip()["height"]
130 log_info(f"Group A tip height: {tip_a}, Group B tip height: {tip_b}")
131
132 # Reconnect
133 for n in nodes:
134 n.peers = set(range(6)) - {n.node_id}
135
136 # Reconciliation: longest chain wins (Group B)
137 longest_chain = max(nodes, key=lambda n: n.tip()["height"]).chain
138 for n in nodes:
139 if n.chain != longest_chain:
140 n.chain = list(longest_chain)
141
142 # Verify all nodes converged
143 converged = all(n.tip()["height"] == longest_chain[-1]["height"] for n in nodes)
144 log_pass("All nodes converged to longest chain") if converged else log_fail("Nodes did NOT converge")
145
146 # Verify no double-spend survived (no conflicting tx in any block)
147 all_txs: set[str] = set()
148 double_spends = 0
149 for block in longest_chain:
150 for tx in block["txs"]:
151 if tx["txid"] in all_txs:
152 double_spends += 1
153 all_txs.add(tx["txid"])
154
155 # Verify shared transactions are not lost
156 lost_txs = [tx["txid"] for tx in shared_txs if tx["txid"] not in all_txs]
157
158 return {
159 "mode": "simulation",
160 "group_a_tip": tip_a,
161 "group_b_tip": tip_b,
162 "final_converged_height": longest_chain[-1]["height"],
163 "nodes_converged": converged,
164 "double_spends_in_chain": double_spends,
165 "lost_transactions": lost_txs,
166 }

172def live_partition_test(base_ports: list[int], auth: tuple[str, str] | None = None) -> dict[str, Any]:
173 log_info("Attempting live partition test against RPC nodes …")
174 urls = [f"http://127.0.0.1:{p}" for p in base_ports]
175 infos = []
176 for url in urls:
177 info = rpc(url, "getblockchaininfo", auth=auth)

179 log_warn(f" {url} unreachable")

182 infos.append(info)

184 if any(i is None for i in infos):

186 return simulate_partition()
187
188 # Capture initial heights

190 log_info(f"Initial heights: {initial_heights}")
