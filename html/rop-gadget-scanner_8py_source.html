50# x86-64 instruction patterns for ROP gadget detection
51# Format: (name, byte_pattern, max_preceding_bytes, category)
52RET_BYTE = 0xC3
53RETF_BYTE = 0xCB
54SYSCALL_BYTES = bytes([0x0F, 0x05])
55INT80_BYTES = bytes([0xCD, 0x80])
56
57# x86-64 register encoding in ModR/M byte for pop instructions
58# pop rax=0x58, pop rcx=0x59, pop rdx=0x5A, pop rbx=0x5B
59# pop rsp=0x5C, pop rbp=0x5D, pop rsi=0x5E, pop rdi=0x5F
60POP_OPCODES = {
61 0x58: "pop rax", 0x59: "pop rcx", 0x5A: "pop rdx", 0x5B: "pop rbx",
62 0x5C: "pop rsp", 0x5D: "pop rbp", 0x5E: "pop rsi", 0x5F: "pop rdi",
63}
64
65# REX.W prefix + pop for r8-r15
66REX_W = 0x41 # REX.B prefix for extended registers
67POP_R8_R15 = {
68 0x58: "pop r8", 0x59: "pop r9", 0x5A: "pop r10", 0x5B: "pop r11",
69 0x5C: "pop r12", 0x5D: "pop r13", 0x5E: "pop r14", 0x5F: "pop r15",
70}

82 def find_text_section(self) -> tuple[int, int]:
83 """Find .text section offset and size in PE binary."""
84 if self.data[:2] != b"MZ":
85 return 0, len(self.data)
86
87 try:
88 pe_offset = struct.unpack_from("<I", self.data, 0x3C)[0]
89 if self.data[pe_offset:pe_offset + 4] != b"PE\x00\x00":
90 return 0, len(self.data)
91
92 num_sections = struct.unpack_from("<H", self.data, pe_offset + 6)[0]
93 opt_header_size = struct.unpack_from("<H", self.data, pe_offset + 20)[0]
94 section_table = pe_offset + 24 + opt_header_size
95
96 for i in range(num_sections):
97 sec_offset = section_table + i * 40
98 name = self.data[sec_offset:sec_offset + 8].rstrip(b"\x00").decode("ascii", errors="replace")
99 virt_size = struct.unpack_from("<I", self.data, sec_offset + 8)[0]
100 raw_offset = struct.unpack_from("<I", self.data, sec_offset + 20)[0]
101 raw_size = struct.unpack_from("<I", self.data, sec_offset + 16)[0]
102 chars = struct.unpack_from("<I", self.data, sec_offset + 36)[0]
103
104 # IMAGE_SCN_CNT_CODE = 0x20, IMAGE_SCN_MEM_EXECUTE = 0x20000000
105 if chars & 0x20000020:
106 return raw_offset, raw_size
107
108 return 0, len(self.data)
109 except (struct.error, IndexError):
110 return 0, len(self.data)

112 def scan_ret_gadgets(self, text_start: int, text_size: int, max_lookback: int = 10):
113 """Find all ret-terminated gadgets."""
114 text_end = text_start + text_size
115 data = self.data
116
117 for i in range(text_start, min(text_end, len(data))):
118 if data[i] != RET_BYTE:
119 continue
120
121 self.stats["total_ret"] += 1
122 addr = self.base_addr + i
123
124 # Check preceding bytes for gadget patterns
125 for lookback in range(1, min(max_lookback + 1, i - text_start + 1)):
126 start = i - lookback
127
128 # Single pop + ret
129 if lookback == 1 and data[start] in POP_OPCODES:
130 reg_name = POP_OPCODES[data[start]]
131 self.gadgets["pop_ret"].append({
132 "addr": f"0x{addr - lookback:x}",
133 "gadget": f"{reg_name}; ret",
134 "bytes": data[start:i + 1].hex(),
135 })
136 self.stats["pop_ret"] += 1
137
138 # REX + pop + ret (r8-r15)
139 if lookback == 2 and data[start] == REX_W and data[start + 1] in POP_R8_R15:
140 reg_name = POP_R8_R15[data[start + 1]]
141 self.gadgets["pop_ret"].append({
142 "addr": f"0x{addr - lookback:x}",
143 "gadget": f"{reg_name}; ret",
144 "bytes": data[start:i + 1].hex(),
145 })
146 self.stats["pop_ret"] += 1
147
148 # Double pop + ret
149 if lookback == 2 and data[start] in POP_OPCODES and data[start + 1] in POP_OPCODES:
150 g = f"{POP_OPCODES[data[start]]}; {POP_OPCODES[data[start+1]]}; ret"
151 self.gadgets["multi_pop_ret"].append({
152 "addr": f"0x{addr - lookback:x}",
153 "gadget": g,
154 "bytes": data[start:i + 1].hex(),
155 })
156 self.stats["multi_pop_ret"] += 1
157
158 # Stack pivot: xchg rsp, reg + ret or mov rsp, reg + ret
159 # xchg eax, esp = 0x94
160 if lookback >= 1 and data[start] == 0x94:
161 self.gadgets["stack_pivot"].append({
162 "addr": f"0x{addr - lookback:x}",
163 "gadget": "xchg eax, esp; ret",
164 "bytes": data[start:i + 1].hex(),
165 })
166 self.stats["stack_pivot"] += 1
167
168 # leave; ret (stack pivot via saved rbp)
169 if lookback == 1 and data[start] == 0xC9:
170 self.gadgets["stack_pivot"].append({
171 "addr": f"0x{addr - lookback:x}",
172 "gadget": "leave; ret",
173 "bytes": data[start:i + 1].hex(),
174 })
175 self.stats["stack_pivot"] += 1
176
177 # mov [reg], reg + ret patterns (write-what-where)
178 # mov [rdi], rsi = 0x48 0x89 0x37
179 if lookback >= 3 and data[start] == 0x48 and data[start + 1] == 0x89:
180 modrm = data[start + 2]
181 mod = (modrm >> 6) & 3
182 if mod == 0: # [reg] indirect
183 self.gadgets["write_what_where"].append({
184 "addr": f"0x{addr - lookback:x}",
185 "gadget": f"mov [reg], reg; ...; ret",
186 "bytes": data[start:i + 1].hex(),
187 })
188 self.stats["write_what_where"] += 1
189
190 # call reg + ret (arbitrary call)
191 # call rax = 0xFF 0xD0, call rcx = 0xFF 0xD1, etc.
192 if lookback >= 2 and data[start] == 0xFF and (data[start + 1] & 0xF8) == 0xD0:
193 reg_idx = data[start + 1] & 0x07
194 regs = ["rax", "rcx", "rdx", "rbx", "rsp", "rbp", "rsi", "rdi"]
195 self.gadgets["arbitrary_call"].append({
196 "addr": f"0x{addr - lookback:x}",
197 "gadget": f"call {regs[reg_idx]}; ...; ret",
198 "bytes": data[start:i + 1].hex(),
199 })
200 self.stats["arbitrary_call"] += 1
201
202 # jmp reg (not ret-terminated but useful)
203 if data[start] == 0xFF and (data[start + 1] & 0xF8) == 0xE0:
204 reg_idx = data[start + 1] & 0x07
205 regs = ["rax", "rcx", "rdx", "rbx", "rsp", "rbp", "rsi", "rdi"]
206 self.gadgets["jmp_reg"].append({
207 "addr": f"0x{addr - lookback:x}",
208 "gadget": f"jmp {regs[reg_idx]}",
209 "bytes": data[start:start + 2].hex(),
210 })
211 self.stats["jmp_reg"] += 1

213 def scan_syscall_gadgets(self, text_start: int, text_size: int):
214 """Find syscall; ret and int 0x80; ret gadgets."""
215 text_end = text_start + text_size
216 data = self.data
217
218 for i in range(text_start, min(text_end - 1, len(data) - 1)):
219 # syscall; ret
220 if data[i:i + 2] == SYSCALL_BYTES:
221 self.stats["syscall"] += 1
222 if i + 2 < len(data) and data[i + 2] == RET_BYTE:
223 self.gadgets["syscall_ret"].append({
224 "addr": f"0x{self.base_addr + i:x}",
225 "gadget": "syscall; ret",
226 "bytes": data[i:i + 3].hex(),
227 })
228 self.stats["syscall_ret"] += 1
229
230 # int 0x80; ret
231 if data[i:i + 2] == INT80_BYTES:
232 self.stats["int80"] += 1
233 if i + 2 < len(data) and data[i + 2] == RET_BYTE:
234 self.gadgets["int80_ret"].append({
235 "addr": f"0x{self.base_addr + i:x}",
236 "gadget": "int 0x80; ret",
237 "bytes": data[i:i + 3].hex(),
238 })
239 self.stats["int80_ret"] += 1

241 def assess_risk(self) -> dict:
242 """Compare gadget counts against hardened/unhardened thresholds."""
243 total = sum(self.stats.values())
244 pop_ret = self.stats.get("pop_ret", 0) + self.stats.get("multi_pop_ret", 0)
245 pivots = self.stats.get("stack_pivot", 0)
246 www = self.stats.get("write_what_where", 0)
247 calls = self.stats.get("arbitrary_call", 0)
248 syscalls = self.stats.get("syscall_ret", 0)
249
250 # Thresholds based on typical binary analysis
251 # Hardened binary (CFI + CET): <50 useful gadgets
252 # Normal binary: 100-500 gadgets
253 # Unhardened: 500+ gadgets
254 risk = "LOW"
255 notes = []
256
257 if pop_ret > 200:
258 risk = "HIGH"
259 notes.append(f"{pop_ret} pop+ret gadgets — sufficient for ROP chain construction")
260 elif pop_ret > 50:
261 risk = "MEDIUM"
262 notes.append(f"{pop_ret} pop+ret gadgets — limited ROP potential")
263 else:
264 notes.append(f"{pop_ret} pop+ret gadgets — minimal ROP surface")
265
266 if pivots > 10:
267 risk = max(risk, "MEDIUM")
268 notes.append(f"{pivots} stack pivots found — enables stack migration attacks")
269 if www > 5:
270 risk = max(risk, "MEDIUM")
271 notes.append(f"{www} write-what-where gadgets — enables arbitrary writes")
272 if syscalls > 0:
273 risk = "HIGH"
274 notes.append(f"{syscalls} syscall+ret gadgets — enables direct system call chains")
275 if calls > 20:
276 risk = max(risk, "MEDIUM")
277 notes.append(f"{calls} arbitrary call gadgets")
278
279 # Zig-specific note
280 notes.append("Zig binaries have bounds checking + safe memory model, reducing exploitability")
281
282 return {
283 "risk_level": risk,
284 "total_gadgets": total,
285 "usable_pop_ret": pop_ret,
286 "stack_pivots": pivots,
287 "write_what_where": www,
288 "arbitrary_calls": calls,
289 "syscall_ret": syscalls,
290 "notes": notes,
291 }
