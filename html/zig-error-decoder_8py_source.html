48def decode_errors(text: str) -> List[Dict[str, str]]:
49 results = []
50 for line in text.splitlines():
51 for pat in ERROR_PATTERNS:
52 m = pat["regex"].search(line)
53 if m:
54 suggestion = pat["suggestion"]
55 for i, g in enumerate(m.groups(), 1):
56 suggestion = suggestion.replace(f"{{group{i}}}", g)
57 suggestion = suggestion.replace("{match}", m.group(1))
58 results.append({"error": line.strip(), "suggestion": suggestion})
59 break
60 return results
