57def scan_unchecked_arithmetic(filepath: str, lines: list) -> list:
58 """Find arithmetic operations that might overflow without @addWithOverflow."""
59 findings = []
60 # Patterns: a + b, a * b, a - b on integer types without overflow check
61 arith_pattern = re.compile(
62 r'(\w+)\s*([+\-*])\s*(\w+)\s*;'
63 )
64 overflow_funcs = {"@addWithOverflow", "@subWithOverflow", "@mulWithOverflow",
65 "@shlWithOverflow", "math.add", "math.mul", "math.sub",
66 "std.math.add", "std.math.mul"}
67
68 # Track which functions use overflow-safe ops
69 in_function = None
70 func_has_overflow_check = set()
71
72 for i, line in enumerate(lines, 1):
73 stripped = line.strip()
74
75 # Track function boundaries
76 fn_match = re.match(r'(?:pub\s+)?fn\s+(\w+)', stripped)
77 if fn_match:
78 in_function = fn_match.group(1)
79
80 # Check if line uses overflow-safe builtin
81 for func in overflow_funcs:
82 if func in stripped:
83 if in_function:
84 func_has_overflow_check.add(in_function)
85
86 # Look for raw arithmetic on variables that look like they could overflow
87 if arith_pattern.search(stripped):
88 # Skip if it's in a comptime context or test
89 if "comptime" in stripped or "test " in stripped:
90 continue
91 # Skip comments
92 if stripped.startswith("//"):
93 continue
94 # Skip if the line has an overflow check
95 has_check = any(f in stripped for f in overflow_funcs)
96 if not has_check and not stripped.startswith("const "):
97 # Check for integer type context (look at surrounding lines)
98 context = "\n".join(lines[max(0, i-5):i+2])
99 if any(t in context for t in ["u64", "u128", "u256", "u32", "usize", "i64"]):
100 findings.append(Finding(
101 filepath, i, "MEDIUM", "unchecked_arithmetic",
102 f"Arithmetic without overflow check: {stripped[:80]}"
103 ))
104
105 return findings

108def scan_array_bounds(filepath: str, lines: list) -> list:
109 """Find array/slice indexing without apparent bounds checks."""
110 findings = []
111 # Pattern: arr[expr] where expr is not a literal small number
112 index_pattern = re.compile(r'(\w+)\‍[([^\‍]]+)\‍]')
113
114 for i, line in enumerate(lines, 1):
115 stripped = line.strip()
116 if stripped.startswith("//"):
117 continue
118
119 matches = index_pattern.finditer(stripped)
120 for m in matches:
121 arr_name = m.group(1)
122 index_expr = m.group(2)
123
124 # Skip known safe patterns
125 if index_expr in ("0", "1", "2", "3", "i", "j"):
126 continue
127 if arr_name in ("comptime", "std", "mem", "builtin"):
128 continue
129
130 # Check if bounds check exists in nearby lines
131 context_start = max(0, i - 5)
132 context = "\n".join(lines[context_start:i])
133 has_bounds_check = any(kw in context for kw in [
134 f"{index_expr} < ", f"{index_expr} <=",
135 f"len", f"if ({index_expr}",
136 f"@min(", f"@max(",
137 f"assert({index_expr}",
138 ])
139
140 if not has_bounds_check and not index_expr.isdigit():
141 findings.append(Finding(
142 filepath, i, "HIGH", "unbounded_array_access",
143 f"Array access without visible bounds check: {arr_name}[{index_expr}]"
144 ))
145
146 return findings

149def scan_integer_truncation(filepath: str, lines: list) -> list:
150 """Find potential integer truncation: casting wider to narrower type."""
151 findings = []
152 # Zig cast pattern: @intCast(u32, value_u64) or @truncate
153 intcast_pattern = re.compile(r'@intCast\s*\‍(\s*(u\d+|i\d+)')
154 truncate_pattern = re.compile(r'@truncate\s*\‍(\s*(u\d+|i\d+)')
155
156 # Also catch: var x: u32 = @intCast(some_u64_value)
157 type_widths = {
158 "u8": 8, "u16": 16, "u32": 32, "u64": 64, "u128": 128, "u256": 256,
159 "i8": 8, "i16": 16, "i32": 32, "i64": 64, "i128": 128,
160 "usize": 64,
161 }
162
163 for i, line in enumerate(lines, 1):
164 stripped = line.strip()
165 if stripped.startswith("//"):
166 continue
167
168 # @truncate is always potentially dangerous
169 trunc_match = truncate_pattern.search(stripped)
170 if trunc_match:
171 target_type = trunc_match.group(1)
172 findings.append(Finding(
173 filepath, i, "MEDIUM", "integer_truncation",
174 f"@truncate to {target_type}: data loss possible — {stripped[:80]}"
175 ))
176
177 # @intCast to smaller type
178 cast_match = intcast_pattern.search(stripped)
179 if cast_match:
180 target_type = cast_match.group(1)
181 target_width = type_widths.get(target_type, 0)
182 # Check if source is wider (look for variable type in context)
183 context = "\n".join(lines[max(0, i-10):i])
184 for src_type, src_width in type_widths.items():
185 if src_type in context and src_width > target_width:
186 findings.append(Finding(
187 filepath, i, "HIGH", "integer_truncation",
188 f"@intCast to {target_type} (likely from {src_type}): "
189 f"truncation risk — {stripped[:60]}"
190 ))
191 break
192
193 return findings

196def scan_null_dereference(filepath: str, lines: list) -> list:
197 """Find optional unwrap (.?) without null check."""
198 findings = []
199
200 for i, line in enumerate(lines, 1):
201 stripped = line.strip()
202 if stripped.startswith("//"):
203 continue
204
205 # .? unwrap without orelse
206 if ".?" in stripped and "orelse" not in stripped and "if " not in stripped:
207 # Check if there's an if-check in surrounding context
208 context = "\n".join(lines[max(0, i-3):i])
209 if "if (" not in context and "orelse" not in context:
210 findings.append(Finding(
211 filepath, i, "MEDIUM", "unchecked_optional_unwrap",
212 f"Optional unwrap .? without null guard: {stripped[:80]}"
213 ))
214
215 # unreachable after optional access
216 if "unreachable" in stripped:
217 context = "\n".join(lines[max(0, i-2):i])
218 if ".?" in context or "orelse" in context:
219 findings.append(Finding(
220 filepath, i, "LOW", "unreachable_after_optional",
221 f"unreachable used as optional fallback — consider orelse"
222 ))
223
224 return findings

227def scan_crypto_weaknesses(filepath: str, lines: list) -> list:
228 """Crypto-specific: look for hardcoded keys, weak RNG, timing-unsafe compares."""
229 findings = []
230
231 for i, line in enumerate(lines, 1):
232 stripped = line.strip()
233 if stripped.startswith("//"):
234 continue
235
236 # Hardcoded hex that looks like a key (64+ hex chars)
237 hex_match = re.search(r'"([0-9a-fA-F]{64,})"', stripped)
238 if hex_match and "test" not in filepath.lower():
239 findings.append(Finding(
240 filepath, i, "HIGH", "hardcoded_secret",
241 f"Possible hardcoded key/secret: {hex_match.group(1)[:32]}..."
242 ))
243
244 # std.rand instead of std.crypto.random
245 if "std.rand" in stripped and "crypto" not in stripped:
246 findings.append(Finding(
247 filepath, i, "HIGH", "weak_rng",
248 f"Using std.rand (not cryptographic): {stripped[:80]}"
249 ))
250
251 # Non-constant-time comparison for crypto
252 if "std.mem.eql" in stripped or "==" in stripped:
253 context = "\n".join(lines[max(0, i-5):i+1])
254 if any(kw in context.lower() for kw in ["signature", "hmac", "hash", "digest", "secret"]):
255 if "timingSafeEqual" not in context and "crypto.utils.timingSafeEql" not in context:
256 findings.append(Finding(
257 filepath, i, "HIGH", "timing_unsafe_compare",
258 f"Non-constant-time comparison in crypto context: {stripped[:80]}"
259 ))
260
261 return findings
