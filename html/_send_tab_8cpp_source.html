97void SendTab::onSendClicked() {

99 double amountOmni = m_amountSpin->value();

102 if (to.isEmpty()) {
103 QMessageBox::warning(this, "Error", "Please enter a recipient address.");
104 return;
105 }
106 if (!to.startsWith("ob1q")) {
107 QMessageBox::warning(this, "Error", "Address must start with 'ob1q'.");
108 return;
109 }
110 if (amountSat <= 0) {
111 QMessageBox::warning(this, "Error", "Amount must be greater than zero.");
112 return;
113 }
114 if (amountSat + m_estimatedFee > m_currentBalance) {
115 QMessageBox::warning(this, "Error", "Insufficient balance (including fee).");
116 return;
117 }
118

133 m_statusLabel->setStyleSheet("color: #d94a4a;");
