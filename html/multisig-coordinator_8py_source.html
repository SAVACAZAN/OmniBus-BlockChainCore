25def create_multisig_tx(inputs: List[Dict], outputs: List[Dict], m: int, n: int, pubkeys: List[str]) -> Dict[str, Any]:
26 tx = {
27 "version": 1,
28 "inputs": inputs,
29 "outputs": outputs,
30 "multisig": {"m": m, "n": n, "pubkeys": pubkeys, "signatures": []},
31 "txid": "",
32 }
33 # Compute placeholder txid
34 tx_bytes = json.dumps(tx, sort_keys=True).encode()
35 tx["txid"] = hashlib.sha256(tx_bytes).hexdigest()[:64]
36 return tx

39def add_signature(tx: Dict[str, Any], pubkey: str, signature_hex: str) -> Dict[str, Any]:
40 sigs = tx["multisig"]["signatures"]
41 # Prevent duplicate pubkey signatures
42 for s in sigs:
43 if s["pubkey"] == pubkey:
44 cprint(YELLOW, f"Replacing signature for {pubkey}")
45 s["signature"] = signature_hex
46 return tx
47 sigs.append({"pubkey": pubkey, "signature": signature_hex})
48 return tx

51def finalize_tx(tx: Dict[str, Any]) -> bool:
52 m = tx["multisig"]["m"]
53 if len(tx["multisig"]["signatures"]) >= m:
