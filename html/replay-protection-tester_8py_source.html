57 def call(self, method: str, params: list | dict | None = None) -> dict:
58 """Send JSON-RPC request and return parsed response."""
59 self._id += 1
60 payload = json.dumps({
61 "jsonrpc": "2.0",
62 "method": method,
63 "params": params or [],
64 "id": self._id,
65 })
66
67 try:
68 conn = http.client.HTTPConnection(self.host, self.port, timeout=self.timeout)
69 conn.request("POST", "/", payload.encode("utf-8"), {"Content-Type": "application/json"})
70 resp = conn.getresponse()
71 body = resp.read().decode("utf-8", errors="replace")
72 conn.close()
73 return json.loads(body)
74 except (ConnectionRefusedError, OSError) as e:
75 return {"error": {"code": -1, "message": f"Connection failed: {e}"}}
76 except json.JSONDecodeError:
77 return {"error": {"code": -2, "message": f"Invalid JSON response: {body[:100]}"}}
78 except Exception as e:
79 return {"error": {"code": -3, "message": str(e)}}

124def fake_sign_transaction(tx: dict, private_key_hex: str) -> str:
125 """Create a fake signature for testing (not cryptographically valid)."""
126 tx_data = json.dumps(tx, sort_keys=True).encode("utf-8")
127 key_bytes = bytes.fromhex(private_key_hex)
128 sig = hashlib.sha256(tx_data + key_bytes).hexdigest()
129 tx["signature"] = sig
130 return json.dumps(tx)
