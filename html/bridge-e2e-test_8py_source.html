54def rpc_json(url: str, method: str, params: Any = None, auth: tuple[str, str] | None = None,
55 headers: dict[str, str] | None = None, timeout: float = 15.0) -> Any:
56 payload = {"jsonrpc": "2.0", "method": method, "id": random.randint(1, 100000)}
57 if params is not None:
58 payload["params"] = params
59 h = dict(headers or {})
60 h.setdefault("Content-Type", "application/json")
61 try:
62 resp = requests.post(url, json=payload, auth=auth, headers=h, timeout=timeout)
63 data = resp.json()
64 return data.get("result")
65 except Exception as exc:
66 return {"_error": str(exc)}

101def run_bridge_test(zig_rpc: str, eth_rpc: str, bridge_contract: str,
102 zig_auth: tuple[str, str] | None = None) -> dict[str, Any]:
103 steps: list[dict[str, Any]] = []
104
105 # Step 1: BlockChainCore health
106 info = rpc_json(zig_rpc, "getblockchaininfo", auth=zig_auth)
107 if isinstance(info, dict) and "_error" in info:
108 steps.append({"step": 1, "name": "zig_health", "status": "fail", "detail": info["_error"]})
109 return {"success": False, "steps": steps}
110 steps.append({"step": 1, "name": "zig_health", "status": "pass", "blocks": info.get("blocks")})
111
112 # Step 2: aweb3 Liberty health
113 eth_block = rpc_json(eth_rpc, "eth_blockNumber")
114 if isinstance(eth_block, dict) and "_error" in eth_block:
115 steps.append({"step": 2, "name": "eth_health", "status": "fail", "detail": eth_block["_error"]})
116 return {"success": False, "steps": steps}
117 steps.append({"step": 2, "name": "eth_health", "status": "pass", "block_number": eth_block})
118
119 # Step 3: Mint on BlockChainCore (simulated via sendtoaddress or generatetoaddress)
120 # Use a dummy address if wallet commands are unavailable
121 dummy_addr = "omni1dummyaddressforbridgetest"
122 txid = rpc_json(zig_rpc, "sendtoaddress", [dummy_addr, 1.0], auth=zig_auth)
123 if isinstance(txid, dict) and "_error" in txid:
124 steps.append({"step": 3, "name": "zig_mint", "status": "warn", "detail": txid["_error"]})
125 txid = f"simulated-txid-{random.randint(0, 999999)}"
126 else:
127 steps.append({"step": 3, "name": "zig_mint", "status": "pass", "txid": txid})
128
129 # Step 4: Bridge submission (stub — real bridge would call a contract or relayer)
130 bridge_tx = f"bridge-{txid}"
131 steps.append({"step": 4, "name": "bridge_submit", "status": "info", "bridge_tx": bridge_tx})
132
133 # Step 5: Verify on aweb3 side (balance or event query)
134 if bridge_contract:
135 # ERC20 balanceOf(address) selector = 0x70a08231
136 padded_addr = "000000000000000000000000" + dummy_addr[-40:] if len(dummy_addr) >= 40 else "0" * 24 + dummy_addr
137 data = "0x70a08231" + padded_addr
138 result = eth_call(eth_rpc, bridge_contract, data)
139 steps.append({"step": 5, "name": "eth_verify", "status": "pass" if not isinstance(result, dict) else "warn",
140 "balance_raw": result})
141 else:
142 steps.append({"step": 5, "name": "eth_verify", "status": "skip", "detail": "No bridge contract address provided"})
143
144 success = all(s["status"] in ("pass", "info", "skip") for s in steps)
145 return {"success": success, "steps": steps}
