363QByteArray aes256Decrypt(const QByteArray& ciphertext, const QByteArray& key, const QByteArray& iv) {
364 if (key.size() != 32 || iv.size() != 16 || ciphertext.size() % 16 != 0 || ciphertext.isEmpty())
365 return {};
366
367 AES256 aes;
368 aes.keyExpansion(reinterpret_cast<const uint8_t*>(key.constData()));
369
370 QByteArray result;
371 result.reserve(ciphertext.size());

374 for (int offset = 0; offset < ciphertext.size(); offset += 16) {
375 uint8_t block[16];

415// secp256k1 field: p = 2^256 - 2^32 - 977
416// Order: n = FFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFEBAAEDCE6AF48A03BBFD25E8CD0364141
417
418// 256-bit unsigned integer (little-endian 32-bit limbs)
419struct U256 {
420 uint32_t d[8] = {};
421
422 static U256 fromBytes(const uint8_t* bytes, int len = 32) {
423 U256 r;
424 for (int i = 0; i < 8 && i * 4 < len; ++i) {
425 int off = len - 1 - i * 4;
426 r.d[i] = 0;
427 for (int j = 0; j < 4 && (off - j) >= 0; ++j)
428 r.d[i] |= static_cast<uint32_t>(bytes[off - j]) << (j * 8);
429 }
430 return r;
431 }
432
433 void toBytes(uint8_t out[32]) const {
434 for (int i = 0; i < 8; ++i) {
435 out[31 - i*4] = d[i] & 0xff;
436 out[31 - i*4 - 1] = (d[i] >> 8) & 0xff;
437 out[31 - i*4 - 2] = (d[i] >> 16) & 0xff;
438 out[31 - i*4 - 3] = (d[i] >> 24) & 0xff;
439 }
440 }
441
442 bool isZero() const {
443 for (auto v : d) if (v) return false;
444 return true;
445 }
446
447 int cmp(const U256& o) const {
448 for (int i = 7; i >= 0; --i) {
449 if (d[i] < o.d[i]) return -1;
450 if (d[i] > o.d[i]) return 1;
451 }
452 return 0;
453 }
454
455 bool operator>=(const U256& o) const { return cmp(o) >= 0; }
456
457 // add with carry, returns carry
458 static U256 add(const U256& a, const U256& b, uint32_t& carry) {
459 U256 r;
460 uint64_t c = 0;
461 for (int i = 0; i < 8; ++i) {
462 c += static_cast<uint64_t>(a.d[i]) + b.d[i];
463 r.d[i] = static_cast<uint32_t>(c);
464 c >>= 32;
465 }
466 carry = static_cast<uint32_t>(c);
467 return r;
468 }
469
470 // sub with borrow, returns borrow
471 static U256 sub(const U256& a, const U256& b, uint32_t& borrow) {
472 U256 r;
473 int64_t c = 0;
474 for (int i = 0; i < 8; ++i) {
475 c += static_cast<int64_t>(a.d[i]) - b.d[i];
476 r.d[i] = static_cast<uint32_t>(c);
477 c >>= 32;
478 }
479 borrow = (c < 0) ? 1 : 0;
480 return r;
481 }
482};

484// Modular arithmetic mod p (secp256k1 field prime)
485struct FieldP {
486 static const U256 P;
487
488 static U256 mod(const U256& a) {
489 U256 r = a;
490 while (r >= P) {
491 uint32_t borrow;
492 r = U256::sub(r, P, borrow);
493 }
494 return r;
495 }
496
497 static U256 add(const U256& a, const U256& b) {
498 uint32_t carry;
499 U256 r = U256::add(a, b, carry);
500 if (carry || r >= P) {
501 uint32_t borrow;
502 r = U256::sub(r, P, borrow);
503 }
504 return r;
505 }
506
507 static U256 sub(const U256& a, const U256& b) {
508 uint32_t borrow;
509 U256 r = U256::sub(a, b, borrow);
510 if (borrow) {
511 uint32_t carry;
512 r = U256::add(r, P, carry);
513 }
514 return r;
515 }
516
517 static U256 mul(const U256& a, const U256& b) {
518 // Schoolbook multiplication mod p
519 uint64_t t[16] = {};
520 for (int i = 0; i < 8; ++i)
521 for (int j = 0; j < 8; ++j)
522 t[i+j] += static_cast<uint64_t>(a.d[i]) * b.d[j];
523 // Propagate carries in t
524 for (int i = 0; i < 15; ++i) { t[i+1] += t[i] >> 32; t[i] &= 0xFFFFFFFF; }
525
526 // Barrett-like reduction: reduce 512-bit to 256-bit mod p
527 // secp256k1 p = 2^256 - 0x1000003d1
528 // So x mod p: split x = x_hi * 2^256 + x_lo, then x ≡ x_lo + x_hi * 0x1000003d1 (mod p)
529 // May need to repeat since x_hi * 0x1000003d1 can overflow
530 for (int pass = 0; pass < 2; ++pass) {
531 // Extract hi part (limbs 8..15) and multiply by 0x1000003d1
532 uint64_t acc[9] = {};
533 for (int i = 0; i < 8; ++i) {
534 // 0x1000003d1 = 0x100000000 + 0x3d1
535 acc[i] += t[i] + t[8+i] * 0x3d1ULL;
536 acc[i+1] += t[8+i]; // * 2^32 part
537 }
538 // Propagate carries
539 for (int i = 0; i < 8; ++i) { acc[i+1] += acc[i] >> 32; acc[i] &= 0xFFFFFFFF; }
540 for (int i = 0; i < 8; ++i) t[i] = acc[i];
541 for (int i = 8; i < 16; ++i) t[i] = 0;
542 t[8] = acc[8];
543 }
544
545 U256 r;
546 for (int i = 0; i < 8; ++i) r.d[i] = static_cast<uint32_t>(t[i]);
547 while (r >= P) { uint32_t borrow; r = U256::sub(r, P, borrow); }
548 return r;
549 }
550
551 static U256 inv(const U256& a) {
552 // Fermat's little theorem: a^(p-2) mod p
553 // Use square-and-multiply
554 U256 result;
555 result.d[0] = 1; // 1
556 U256 base = a;
557
558 // p-2 in binary - we compute from LSB
559 // p = FFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFEFFFFFC2F
560 // p-2 = FFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFEFFFFFC2D
561 uint8_t pMinus2[32];
562 P.toBytes(pMinus2);
563 pMinus2[31] -= 2; // p - 2
564
565 for (int byte = 31; byte >= 0; --byte) {
566 for (int bit = 0; bit < 8; ++bit) {
567 if (pMinus2[byte] & (1 << (7 - bit)))
568 result = mul(result, base);
569 if (byte > 0 || bit < 7)
570 base = mul(base, base);
571 }
572 // Careful: process MSB to LSB
573 // Actually we need to go MSB first for square-and-multiply
574 }
575 // The above loop is wrong for square-and-multiply. Let me redo it properly.
576 result.d[0] = 1;
577 for (int i = 1; i < 8; ++i) result.d[i] = 0;
578 base = a;
579
580 // Square-and-multiply from MSB
581 bool started = false;
582 for (int byte = 0; byte < 32; ++byte) {
583 for (int bit = 7; bit >= 0; --bit) {
584 if (started)
585 result = mul(result, result);
586 if (pMinus2[byte] & (1 << bit)) {
587 result = mul(result, a);
588 started = true;
589 }
590 }
591 }
592 return result;
593 }

649// EC point doubling (affine)
650ECPoint ecDouble(const ECPoint& P) {
651 if (P.infinity) return P;
652 // s = (3 * x^2) / (2 * y)
653 U256 x2 = FieldP::mul(P.x, P.x);
654 U256 three; three.d[0] = 3;
655 U256 num = FieldP::mul(three, x2); // 3*x^2 (a=0 for secp256k1)
656 U256 two; two.d[0] = 2;
657 U256 den = FieldP::mul(two, P.y);
658 U256 s = FieldP::mul(num, FieldP::inv(den));
659
660 ECPoint R;
661 R.infinity = false;
662 // rx = s^2 - 2*x
663 U256 s2 = FieldP::mul(s, s);
664 U256 twox = FieldP::mul(two, P.x);
665 R.x = FieldP::sub(s2, twox);
666 // ry = s*(x - rx) - y
667 R.y = FieldP::sub(FieldP::mul(s, FieldP::sub(P.x, R.x)), P.y);
668 return R;
669}

671// EC point addition (affine)
672ECPoint ecAdd(const ECPoint& P, const ECPoint& Q) {
673 if (P.infinity) return Q;
674 if (Q.infinity) return P;
675
676 if (P.x.cmp(Q.x) == 0) {
677 if (P.y.cmp(Q.y) == 0)
678 return ecDouble(P);
679 // P + (-P) = infinity
680 ECPoint inf; return inf;
681 }
682
683 // s = (Qy - Py) / (Qx - Px)
684 U256 dy = FieldP::sub(Q.y, P.y);
685 U256 dx = FieldP::sub(Q.x, P.x);
686 U256 s = FieldP::mul(dy, FieldP::inv(dx));
687
688 ECPoint R;
689 R.infinity = false;
690 U256 s2 = FieldP::mul(s, s);
691 R.x = FieldP::sub(FieldP::sub(s2, P.x), Q.x);
692 R.y = FieldP::sub(FieldP::mul(s, FieldP::sub(P.x, R.x)), P.y);
693 return R;
694}

697ECPoint ecMul(const U256& k, const ECPoint& P) {
698 ECPoint R; // infinity
699 ECPoint base = P;
700
701 uint8_t kBytes[32];
702 k.toBytes(kBytes);

704 // Double-and-add from MSB
705 bool started = false;
706 for (int byte = 0; byte < 32; ++byte) {
707 for (int bit = 7; bit >= 0; --bit) {
708 if (started)
709 R = ecDouble(R);
710 if (kBytes[byte] & (1 << bit)) {
711 R = ecAdd(R, P);
712 started = true;
713 }
714 }
715 }
716 return R;
717}

755 // Schoolbook multiplication mod n
756 uint64_t t[16] = {};
757 for (int i = 0; i < 8; ++i)
758 for (int j = 0; j < 8; ++j)
759 t[i+j] += static_cast<uint64_t>(a.d[i]) * b.d[j];
760 for (int i = 0; i < 15; ++i) { t[i+1] += t[i] >> 32; t[i] &= 0xFFFFFFFF; }
761
762 // Reduce mod n using repeated subtraction (simple but correct)
763 // Convert back to U256 first - take low 256 bits and handle overflow
764 // This is a simplified reduction
765 U256 r;
766 for (int i = 0; i < 8; ++i) r.d[i] = static_cast<uint32_t>(t[i]);
767 // For now, handle the simple case (most multiplications in ECDSA context)
768 while (r >= SECP256K1_N) {
769 uint32_t borrow;
770 r = U256::sub(r, SECP256K1_N, borrow);
771 }

776 // Fermat: a^(n-2) mod n
777 uint8_t nMinus2[32];
778 SECP256K1_N.toBytes(nMinus2);
779 // n-2
780 if (nMinus2[31] >= 2) { nMinus2[31] -= 2; }
781 else { nMinus2[31] += 254; nMinus2[30]--; } // borrow
782
783 U256 result; result.d[0] = 1;
784 bool started = false;
785 for (int byte = 0; byte < 32; ++byte) {
786 for (int bit = 7; bit >= 0; --bit) {
787 if (started)
788 result = modN_mul(result, result);
789 if (nMinus2[byte] & (1 << bit)) {
790 result = modN_mul(result, a);
791 started = true;
792 }
793 }
794 }
795 return result;

800QByteArray ecdsaSign(const QByteArray& hash32, const QByteArray& privKey32) {
801 if (hash32.size() != 32 || privKey32.size() != 32) return {};
802
803 U256 z = U256::fromBytes(reinterpret_cast<const uint8_t*>(hash32.constData()));
804 U256 d = U256::fromBytes(reinterpret_cast<const uint8_t*>(privKey32.constData()));
805
806 // Deterministic k: k = HMAC-SHA256(privkey, hash) mod n (simplified RFC 6979)
807 QByteArray kBytes = hmacSha512(privKey32, hash32).left(32);
808 U256 k = U256::fromBytes(reinterpret_cast<const uint8_t*>(kBytes.constData()));
809 // Ensure k < n
810 while (k >= SECP256K1_N || k.isZero()) {
811 kBytes = sha256(kBytes);
812 k = U256::fromBytes(reinterpret_cast<const uint8_t*>(kBytes.constData()));
813 }
814
815 // R = k*G
816 ECPoint R = ecMul(k, SECP256K1_G);
817 if (R.infinity) return {};
818
819 // r = R.x mod n
820 U256 r = R.x;
821 while (r >= SECP256K1_N) { uint32_t b; r = U256::sub(r, SECP256K1_N, b); }
822 if (r.isZero()) return {};
823
824 // s = k^(-1) * (z + r*d) mod n
825 U256 rd = modN_mul(r, d);
826 U256 zrd = FieldN::add(z, rd);
827 U256 kInv = modN_inv(k);
828 U256 s = modN_mul(kInv, zrd);
829 if (s.isZero()) return {};
830
831 // Low-S normalization (BIP-62)
832 U256 halfN;
833 {
834 uint32_t borrow;
835 U256 one; one.d[0] = 1;
836 halfN = U256::sub(SECP256K1_N, one, borrow);
837 // halfN = (n-1)/2 ... shift right by 1
838 for (int i = 0; i < 7; ++i)
839 halfN.d[i] = (halfN.d[i] >> 1) | (halfN.d[i+1] << 31);
840 halfN.d[7] >>= 1;
841 }
842 if (s.cmp(halfN) > 0) {
843 uint32_t borrow;
844 s = U256::sub(SECP256K1_N, s, borrow);
845 }
846
847 // DER encode
848 uint8_t rBuf[33], sBuf[33];
849 r.toBytes(rBuf + 1); rBuf[0] = 0;
850 s.toBytes(sBuf + 1); sBuf[0] = 0;
851
852 int rOff = 1, sOff = 1;
853 while (rOff < 33 && rBuf[rOff] == 0) rOff++;
854 if (rBuf[rOff] & 0x80) rOff--; // need leading 0
855 while (sOff < 33 && sBuf[sOff] == 0) sOff++;
856 if (sBuf[sOff] & 0x80) sOff--;
857
858 int rLen = 33 - rOff, sLen = 33 - sOff;
859
860 QByteArray sig;
861 sig.append(0x30); // SEQUENCE
862 sig.append(static_cast<char>(rLen + sLen + 4));
863 sig.append(0x02); // INTEGER
864 sig.append(static_cast<char>(rLen));
865 sig.append(reinterpret_cast<const char*>(rBuf + rOff), rLen);
866 sig.append(0x02);
867 sig.append(static_cast<char>(sLen));
868 sig.append(reinterpret_cast<const char*>(sBuf + sOff), sLen);
869 return sig;
870}

907std::vector<uint8_t> convertBits(const uint8_t* data, int len, int fromBits, int toBits, bool pad) {
908 std::vector<uint8_t> ret;
909 int acc = 0, bits = 0;
910 int maxv = (1 << toBits) - 1;
911 for (int i = 0; i < len; ++i) {
912 acc = (acc << fromBits) | data[i];
913 bits += fromBits;
914 while (bits >= toBits) {
915 bits -= toBits;
916 ret.push_back((acc >> bits) & maxv);
917 }
918 }
919 if (pad && bits > 0)
920 ret.push_back((acc << (toBits - bits)) & maxv);
921 return ret;
922}

934 for (int i = 0; i < 6; ++i)
935 ret += BECH32_CHARSET[(polymod >> (5 * (5 - i))) & 31];
936 return ret;
