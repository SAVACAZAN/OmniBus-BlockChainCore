129 def __init__(self, project_root: Path):
130 self.project_root = project_root
131 self.issues: List[SecurityIssue] = []

133 def scan_file(self, filepath: Path) -> List[SecurityIssue]:
134 """Scanează un singur fișier."""
135 issues = []
136
137 try:
138 with open(filepath, 'r', encoding='utf-8') as f:
139 lines = f.readlines()
140 except Exception:
141 return issues
142
143 rel_path = str(filepath.relative_to(self.project_root))
144
145 for line_num, line in enumerate(lines, 1):
146 for category, rule in self.PATTERNS.items():
147 if rule['pattern'].search(line):
148 # Verifică dacă e comentat
149 stripped = line.strip()
150 if stripped.startswith('//'):
151 continue
152
153 issues.append(SecurityIssue(
154 file=rel_path,
155 line=line_num,
156 severity=rule['severity'],
157 category=category,
158 message=rule['message'],
159 code_snippet=line.strip()[:80],
160 recommendation=rule['recommendation']
161 ))
162
163 return issues
164
165 def scan_project(self) -> List[SecurityIssue]:
166 """Scanează întregul proiect."""
167 zig_files = list(self.project_root.rglob('*.zig'))
168 zig_files = [
169 f for f in zig_files
170 if not any(part.startswith('.') or part in ['zig-out', '.zig-cache']
171 for part in f.parts)
172 ]
173
174 for filepath in zig_files:
175 issues = self.scan_file(filepath)
176 self.issues.extend(issues)
177
178 # Sortează după severitate
179 severity_order = {'critical': 0, 'high': 1, 'medium': 2, 'low': 3}
180 self.issues.sort(key=lambda i: severity_order.get(i.severity, 4))
181
182 return self.issues

184 def get_stats(self) -> Dict:
185 """Returnează statistici despre issue-uri."""
186 stats = {
187 'total': len(self.issues),

191
192 for issue in self.issues:
193 stats['by_severity'][issue.severity] += 1
194 stats['by_category'][issue.category] += 1
195
196 return dict(stats)
