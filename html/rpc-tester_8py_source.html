38class RPCTester:
39 def __init__(self, url: str = "http://127.0.0.1:8332", user: str = "", password: str = ""):
40 self.url = url
41 self.headers = {"Content-Type": "application/json"}
42 self.session = requests.Session()
43 self.session.headers.update(self.headers)
44 if user:
45 self.session.auth = (user, password)
46 self.results: List[Dict[str, Any]] = []
47
48 def _call(self, method: str, params: List[Any]) -> Dict[str, Any]:
49 payload = {"jsonrpc": "2.0", "id": 1, "method": method, "params": params}
50 try:
51 resp = self.session.post(self.url, json=payload, timeout=10)
52 return {"method": method, "status_code": resp.status_code, "body": resp.text}
53 except requests.RequestException as e:
54 return {"method": method, "error": str(e)}

56 def test_methods(self) -> None:
57 cprint(YELLOW, "\n--- RPC Method Tests ---")
58 for method, params in RPC_METHODS:
59 res = self._call(method, params)
60 ok = "error" not in res
61 color = GREEN if ok else RED

63 cprint(color, f"{'[PASS]' if ok else '[FAIL]'} {method}")

79 def stress_test(self, concurrency: int = 50, requests_count: int = 500) -> None:
80 cprint(YELLOW, f"\n--- Stress Test ({requests_count} req @ {concurrency} threads) ---")
81 ok = 0
82 fail = 0
83
84 def worker(_: int) -> bool:
85 r = self._call("getblockcount", [])
86 return "error" not in r
87
88 with ThreadPoolExecutor(max_workers=concurrency) as ex:
89 futures = [ex.submit(worker, i) for i in range(requests_count)]
90 for fut in as_completed(futures):
91 if fut.result():
92 ok += 1
93 else:
94 fail += 1
95
96 self.results.append({"check": "stress_test", "status": "PASS" if fail == 0 else "FAIL", "detail": f"{ok} OK, {fail} FAIL"})
97 cprint(GREEN if fail == 0 else RED, f"Stress: {ok} OK, {fail} FAIL")
