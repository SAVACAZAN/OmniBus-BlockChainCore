76class ZigParser:
77 """Parser pentru fișiere Zig."""
78
79 # Regex patterns pentru extragere
80 CONST_PATTERN = re.compile(
81 r'^\s*(pub\s+)?const\s+(\w+)\s*[:=]\s*([^;]+)(?:;|$)',
82 re.MULTILINE
83 )
84
85 ENUM_PATTERN = re.compile(
86 r'(?:pub\s+)?const\s+(\w+)\s*=\s*enum\s*\{([^}]+)\}',
87 re.DOTALL
88 )
89
90 INLINE_ENUM_PATTERN = re.compile(
91 r'pub\s+const\s+(\w+)\s*=\s*enum\s*\‍(\s*\w+\s*\‍)\s*\{([^}]+)\}',
92 re.DOTALL
93 )
94
95 COMMENT_PATTERN = re.compile(r'//(.+)$', re.MULTILINE)
96
97 def __init__(self, content: str, filename: str):
98 self.content = content
99 self.filename = filename
100 self.lines = content.split('\n')
101
102 def get_line_context(self, line_num: int) -> str:
103 """Ia comentariul deasupra unei linii sau de pe aceeași linie."""
104 context = []
105
106 # Comentariu pe aceeași linie
107 same_line_match = self.COMMENT_PATTERN.search(self.lines[line_num - 1])
108 if same_line_match:
109 context.append(same_line_match.group(1).strip())
110
111 # Comentariu deasupra (până la 3 linii)
112 for i in range(line_num - 2, max(0, line_num - 5), -1):
113 line = self.lines[i].strip()
114 if line.startswith('//'):
115 context.insert(0, line[2:].strip())
116 elif line.startswith('///'):
117 context.insert(0, line[3:].strip())
118 elif line and not line.startswith('//'):
119 break
120
121 return ' '.join(context)
122
123 def parse_constants(self) -> List[Constant]:
124 """Extrage toate constantele simple din fișier (fără structuri complexe)."""
125 constants = []
126
127 for match in self.CONST_PATTERN.finditer(self.content):
128 is_pub = match.group(1) is not None
129 name = match.group(2)
130 value = match.group(3).strip()
131
132 # Skip structuri, union-uri și tipuri complexe
133 if self._is_simple_constant(value):
134 # Găsește numărul liniei
135 line_num = self.content[:match.start()].count('\n') + 1
136 context = self.get_line_context(line_num)
137
138 # Trunchiază valori foarte lungi
139 if len(value) > 100:
140 value = value[:97] + "..."
141
142 constants.append(Constant(
143 name=name,
144 value=value,
145 file=self.filename,
146 line=line_num,
147 context=context,
148 is_public=is_pub
149 ))
150
151 return constants
152
153 def _is_simple_constant(self, value: str) -> bool:
154 """Verifică dacă valoarea este o constantă simplă (nu structură)."""
155 value = value.strip()
156
157 # Liste de cuvinte cheie care indică tipuri complexe
158 complex_keywords = [
159 'struct {', 'union {', 'enum {', 'packed struct',
160 'extern struct', 'opaque', 'fn (', '= .{', '=.{',
161 'std.', 'ArrayList', 'HashMap', 'Managed('
162 ]
163
164 for keyword in complex_keywords:
165 if keyword in value:
166 return False
167
168 # Acceptă numere, string-uri, arrays simple
169 return True
170
171 def parse_enums(self) -> List[EnumDef]:
172 """Extrage toate enum-urile din fișier."""
173 enums = []
174
175 # Pattern 1: const Name = enum { ... }
176 for match in self.ENUM_PATTERN.finditer(self.content):
177 name = match.group(1)
178 body = match.group(2)
179 line_num = self.content[:match.start()].count('\n') + 1
180
181 variants = self._parse_enum_body(body)
182 enums.append(EnumDef(
183 name=name,
184 variants=variants,
185 file=self.filename,
186 line=line_num
187 ))
188
189 # Pattern 2: pub const Name = enum(u8) { ... }
190 for match in self.INLINE_ENUM_PATTERN.finditer(self.content):
191 name = match.group(1)
192 body = match.group(2)
193 line_num = self.content[:match.start()].count('\n') + 1
194
195 variants = self._parse_enum_body(body)
196 enums.append(EnumDef(
197 name=name,
198 variants=variants,
199 file=self.filename,
200 line=line_num,
201 is_public=True
202 ))
203
204 return enums
205
206 def _parse_enum_body(self, body: str) -> List[EnumVariant]:
207 """Parsează corpul unui enum."""
208 variants = []
209 lines = body.split('\n')
210 auto_value = 0
211
212 for line in lines:
213 line = line.strip()
214 if not line or line.startswith('//'):
215 continue
216
217 # Pattern: Name = value,
218 if '=' in line:
219 parts = line.split('=', 1)
220 name = parts[0].strip()
221 value_part = parts[1].split(',')[0].strip()
222 comment = parts[1].split('//')[1].strip() if '//' in parts[1] else ""
223
224 try:
225 auto_value = int(value_part)
226 except ValueError:
227 pass
228
229 variants.append(EnumVariant(
230 name=name,
231 value=value_part,
232 description=comment
233 ))
234 else:
235 # Pattern: Name,
236 name = line.rstrip(',').strip()
237 if name:
238 variants.append(EnumVariant(
239 name=name,
240 value=str(auto_value),
241 description=""
242 ))
243 auto_value += 1
244
245 return variants

310 def _extract_message_types(self) -> None:
311 """Extrage MessageType enum."""
312 for enum in self.results.enums:
313 if 'MessageType' in enum.name or 'MsgType' in enum.name:
314 for i, variant in enumerate(enum.variants):
315 desc = variant.description if variant.description else f"Message type {variant.name}"
316 self.results.message_types.append((
317 variant.name,
318 i,
319 desc
320 ))
321
322 def _extract_error_types(self) -> None:
323 """Extrage tipurile de erori."""
324 error_enums = [e for e in self.results.enums if 'Error' in e.name]
325 for enum in error_enums:
326 for variant in enum.variants:
327 self.results.error_types.append((
328 enum.name,
329 variant.name
330 ))

332 def _extract_magic_numbers(self) -> None:
333 """Extrage magic numbers (numere cu semnificație specială)."""
334 magic_patterns = [
335 (r'PORT.*=\s*(\d+)', 'Port'),
336 (r'MAX.*=\s*(\d+)', 'Maximum'),
337 (r'MIN.*=\s*(\d+)', 'Minimum'),
338 (r'TIMEOUT.*=\s*(\d+)', 'Timeout'),
339 (r'INTERVAL.*=\s*(\d+)', 'Interval'),
340 (r'_MS\s*=\s*(\d+)', 'Milliseconds'),
341 (r'_SEC\s*=\s*(\d+)', 'Seconds'),
342 ]
343
344 for const in self.results.constants:
345 for pattern, category in magic_patterns:
346 if re.search(pattern, const.name, re.IGNORECASE):
347 self.results.magic_numbers.append((
348 const.name,
349 const.value,
350 category
351 ))
352 break
