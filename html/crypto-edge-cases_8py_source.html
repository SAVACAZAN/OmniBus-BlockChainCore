74def mod_mul(a: int, b: int, m: int = P) -> int:
75 return (a * b) % m

78def mod_add(a: int, b: int, m: int = P) -> int:

82def mod_inv(a: int, m: int = N) -> int:
83 # Extended Euclidean
84 g, x, _ = extended_gcd(a % m, m)
85 if g != 1:
86 raise ValueError("inverse does not exist")
87 return x % m

90def extended_gcd(a: int, b: int) -> tuple[int, int, int]:
91 if a == 0:
92 return b, 0, 1
93 g, x1, y1 = extended_gcd(b % a, a)
94 x = y1 - (b // a) * x1
95 y = x1
96 return g, x, y

99# ---------------------------------------------------------------------------
100# Point arithmetic (simplified projective / affine)
101# ---------------------------------------------------------------------------
102Point = namedtuple("Point", ["x", "y"])
103
104
105def point_add(p1: Point | None, p2: Point | None) -> Point | None:
106 """Add two points on secp256k1. None represents point at infinity."""
107 if p1 is None:
108 return p2
109 if p2 is None:
110 return p1
111 if p1.x == p2.x and p1.y != p2.y:
112 return None # P + (-P) = O
113 if p1 == p2:
114 # doubling
115 lam = mod_mul(mod_mul(3 * p1.x * p1.x + A, mod_inv(2 * p1.y, P), P), 1, P)
116 else:
117 lam = mod_mul((p2.y - p1.y) % P, mod_inv((p2.x - p1.x) % P, P), P)
118 x3 = (lam * lam - p1.x - p2.x) % P
119 y3 = (lam * (p1.x - x3) - p1.y) % P
120 return Point(x3, y3)

123def scalar_mul(k: int, p: Point | None = None) -> Point | None:
124 if p is None:
125 p = Point(Gx, Gy)
126 result: Point | None = None
127 addend = p
128 while k:
129 if k & 1:
130 result = point_add(result, addend)
131 addend = point_add(addend, addend)
132 k >>= 1
133 return result

136# ---------------------------------------------------------------------------
137# Signing primitive (RFC 6979-like but with controlled k)
138# ---------------------------------------------------------------------------
139def sign_raw(private_key: int, message_hash: bytes, k: int) -> dict[str, Any] | None:
140 """Return raw (r, s) or None if k is invalid."""
141 if not (1 <= private_key < N):
142 return None
143 if not (1 <= k < N):
144 return None
145 z = int.from_bytes(message_hash, "big") % N
146 R = scalar_mul(k)
147 if R is None:
148 return None
149 r = R.x % N
150 if r == 0:
151 return None
152 s = mod_mul(mod_inv(k, N), (z + r * private_key) % N, N)
153 if s == 0:
154 return None
155 return {"r": r, "s": s}

158def verify_low_s(s: int) -> bool:
159 """BIP-62 low-s check: s <= N/2."""
160 return s <= N // 2
