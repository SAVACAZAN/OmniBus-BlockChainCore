2"""
3tx-stress-sim2.py — Long-running TX simulation for OmniBus testnet.
4
5Run for 1h, 24h, or custom duration, generating a steady stream of transactions
6across multiple "levels" so we can stress the chain without hand-typing curl.
7
8Watching:
9 - mempool fill / drain behavior
10 - block production rate when mempool is non-empty
11 - balance propagation to fresh addresses
12 - Rich List growth over time
13 - faucet refill cadence
14 - whether any TX gets silently dropped
15
16Levels
17------
18 L0 — base load : 2 tx / 30s, amounts 1k–10k SAT (forever-safe)
19 L1 — light burst : 3 tx / 8s, amounts 10k–100k SAT
20 L2 — moderate : 15 tx / 10s, amounts 10k–50k SAT (mempool steady fill)
21 L3 — stress : 50 tx / 10s, amounts 1k–20k SAT (race-test, overlap blocks)
22 L4 — extreme : 100 tx / 8s, amounts 1k–15k SAT (push to limits)
23
24Usage
25-----
26 # Local node, default testnet RPC
27 python tx-stress-sim2.py --level 0 --duration 1h
28
29 # VPS testnet (needs OMNIBUS_RPC_TOKEN env var):
30 export OMNIBUS_RPC_TOKEN=31926ece...
31 python tx-stress-sim2.py --level 1 --duration 24h \\
32 --rpc http://omnibusblockchain.cc:18332
33
34 # Burst mode + detailed stats
35 python tx-stress-sim2.py --level 3 --duration 30m --burst-mode --detailed-stats
36
37 # Custom recipient pool size (rotates through real testnet addresses)
38 python tx-stress-sim2.py --level 2 --num-addresses 8 --duration 2h

138def parse_duration(s: str) -> int:
139 """Accepts strings like '1h', '24h', '30m', '90s'. Returns seconds."""
140 s = s.strip().lower()
141 if s.endswith("h"):
142 return int(float(s[:-1]) * 3600)
143 if s.endswith("m"):
144 return int(float(s[:-1]) * 60)
145 if s.endswith("s"):
146 return int(float(s[:-1]))
147 return int(float(s))

150def rpc(url: str, method: str, params: list[Any] | None = None, *, timeout: float = 5.0,
151 token: str | None = None) -> Any:
152 body = json.dumps({"jsonrpc": "2.0", "id": 1, "method": method,
153 "params": params or []}).encode()
154 headers = {"Content-Type": "application/json"}
155 if token:
156 headers["Authorization"] = f"Bearer {token}"
157 req = urllib.request.Request(url, data=body, headers=headers)
158 with urllib.request.urlopen(req, timeout=timeout) as resp:
159 data = json.loads(resp.read().decode())
160 if "error" in data and data["error"]:
161 raise RuntimeError(f"{method}: {data['error'].get('message', data['error'])}")
162 return data.get("result")

165def chain_snapshot(rpc_url: str, token: str | None) -> tuple[int, int]:
166 """(height, mempool_size). Returns (-1, -1) on RPC failure.
167
168 Uses `getblockcount` + `getmempoolinfo` for portability across node versions
169 (`getblockchaininfo` doesn't always include mempool_size).
170 """
171 try:
172 height = int(rpc(rpc_url, "getblockcount", token=token))
173 except Exception:
174 return -1, -1
175 try:
176 mp = rpc(rpc_url, "getmempoolinfo", token=token)
177 mempool = int(mp.get("size", mp.get("mempool_size", 0))) if isinstance(mp, dict) else 0
178 except Exception:
179 mempool = 0
180 return height, mempool

183def send_one(rpc_url: str, to_addr: str, amount_sat: int, token: str | None) -> tuple[bool, str]:
184 """Returns (accepted, txid_or_error).
185
186 Uses `sendtransaction` (signs with node's primary wallet — sender = node).
187 For per-address signing, would need `sendrawtransaction` + local key derivation.
188 """
189 try:
190 result = rpc(rpc_url, "sendtransaction", [to_addr, amount_sat], token=token)
191 if isinstance(result, dict) and result.get("status") == "accepted":
192 return True, result.get("txid", "")
193 return False, str(result)
194 except Exception as e:
195 return False, str(e)
