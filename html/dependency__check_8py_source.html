77 def check_command(self, cmd: str) -> Tuple[bool, Optional[str]]:
78 """Verifică dacă o comandă există și ia versiunea."""
79 # Check if command exists
80 if shutil.which(cmd) is None:
81 return False, None
82
83 # Try to get version
84 version_args = ['--version', '-v', '-V', 'version']
85
86 for arg in version_args:
87 try:
88 result = subprocess.run(
89 [cmd, arg],
90 capture_output=True,
91 text=True,
92 timeout=5
93 )
94 if result.returncode == 0:
95 output = result.stdout.strip() or result.stderr.strip()
96 # Extract first version-like string
97 import re
98 version_match = re.search(r'(\d+\.\d+\.?\d*)', output)
99 if version_match:
100 return True, version_match.group(1)
101 except:
102 pass
103
104 return True, "unknown"

106 def check_version_meets_min(self, version: str, min_version: str) -> bool:
107 """Verifică dacă versiunea >= min_version."""
108 try:
109 v_parts = [int(x) for x in version.split('.')[:3]]
110 min_parts = [int(x) for x in min_version.split('.')[:3]]
111
112 # Pad with zeros
113 while len(v_parts) < 3:
114 v_parts.append(0)
115 while len(min_parts) < 3:
116 min_parts.append(0)
117
118 return v_parts >= min_parts
119 except (ValueError, IndexError):
120 return True # If we can't parse, assume OK

173 def check_all(self) -> List[CheckResult]:
174 """Verifică toate dependințele."""
175 for dep in self.DEPENDENCIES:
176 installed, version = self.check_command(dep.command)
177
178 version_ok = True
179 if installed and version and dep.min_version:
180 version_ok = self.check_version_meets_min(version, dep.min_version)
181
182 self.results.append(CheckResult(
183 dependency=dep,
184 installed=installed,
185 version=version,
186 version_ok=version_ok,
187 path=shutil.which(dep.command) if installed else None
188 ))

190 # Check liboqs separately
191 self.results.append(self.check_liboqs())
