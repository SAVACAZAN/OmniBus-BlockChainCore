1#!/usr/bin/env python3
2"""
3treasury_flow_test.py — end-to-end test for the NS pay-to-claim + treasury
4agent market-maker flow.
5
6What this exercises:
7 1. Derive a wallet (or read one provided via env).
8 2. Read the current ens.omnibus treasury address + initial state
9 (balance, live orders).
10 3. Build a TX that pays 5+ OMNI to ens.omnibus with op_return memo
11 `ns_claim:<name>.omnibus` and submit it via `sendrawtransaction`.
12 4. Wait for the next block, then `resolvename(<name>)` and assert the
13 name resolves to the sender's address.
14 5. Poll `treasury_getStatus` and confirm the agent has new live orders
15 after the balance jump (eventually consistent — gives it ~3 blocks).
16
17Run:
18 python treasury_flow_test.py --rpc http://127.0.0.1:18332 --name test1
19 python treasury_flow_test.py --rpc https://omnibusblockchain.cc:8443/api-testnet \
20 --name test1 --token 31926ece...
21
22Set OMNIBUS_PRIVKEY (32 hex bytes, no 0x) for the sender wallet, OR pass
23--privkey on the command line. If neither is provided we cannot sign —
24the script will print the unsigned TX shape and exit so you can wire it
25through aweb3 or the explorer manually.
26"""

41def rpc_call(rpc_url: str, method: str, params: list, *, token: str | None = None) -> Any:
42 """JSON-RPC 2.0 helper. Returns the unwrapped `result`. Raises on error."""
43 body = json.dumps({
44 "jsonrpc": "2.0",
45 "id": int(time.time() * 1000) & 0x7FFFFFFF,
46 "method": method,
47 "params": params,
48 }).encode()
49 req = urllib.request.Request(
50 rpc_url,
51 data=body,
52 headers={"Content-Type": "application/json"},
53 )
54 if token:
55 req.add_header("Authorization", f"Bearer {token}")
56 try:
57 with urllib.request.urlopen(req, timeout=10) as resp:
58 data = json.loads(resp.read().decode())
59 except urllib.error.HTTPError as e:
60 raise RuntimeError(f"HTTP {e.code} on {method}: {e.read().decode()[:200]}") from e
61 except urllib.error.URLError as e:
62 raise RuntimeError(f"Connection failed to {rpc_url}: {e.reason}") from e
63
64 if "error" in data and data["error"]:
65 raise RuntimeError(f"RPC error on {method}: {data['error']}")
66 return data.get("result")

90def wait_for_blocks(rpc_url: str, n: int, token: str | None, *, max_wait_s: int = 60) -> None:
91 """Block until `n` more blocks are mined or `max_wait_s` elapses."""
92 start_h = get_block_height(rpc_url, token)
93 target = start_h + n
94 deadline = time.time() + max_wait_s
95 while time.time() < deadline:
96 h = get_block_height(rpc_url, token)
97 if h >= target:
98 return
99 time.sleep(2)
100 raise TimeoutError(f"Only {get_block_height(rpc_url, token) - start_h} blocks mined in {max_wait_s}s (needed {n})")
