25def analyze_dump(dump_path: str) -> Dict[str, Any]:
26 if not os.path.isfile(dump_path):
27 raise FileNotFoundError(dump_path)
28
29 with open(dump_path, "r", encoding="utf-8", errors="ignore") as f:
30 content = f.read()
31
32 # Extract stack frames
33 frames: List[Dict[str, str]] = []
34 frame_re = re.compile(r"(\d+):\s+0x[0-9a-f]+\s+in\s+(.+?)\s+\‍((.+?)\‍)")
35 for m in frame_re.finditer(content):
36 frames.append({"index": m.group(1), "function": m.group(2), "location": m.group(3)})
37
38 # Extract signal
39 signal = "UNKNOWN"
40 sig_re = re.compile(r"signal\s+(SIG\w+|\d+)")
41 sm = sig_re.search(content)
42 if sm:
43 signal = sm.group(1)
44
45 # Extract fault address
46 fault_addr = "N/A"
47 addr_re = re.compile(r"fault addr\s+(0x[0-9a-f]+)", re.IGNORECASE)
48 am = addr_re.search(content)
49 if am:
50 fault_addr = am.group(1)
51
52 return {
53 "dump_file": dump_path,
54 "signal": signal,
55 "fault_address": fault_addr,
56 "stack_frames": frames,
57 "frame_count": len(frames),
58 }
