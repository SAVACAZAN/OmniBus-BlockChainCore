39def extract_rpc_methods(filepath: str) -> list[dict]:
40 """
41 Extract RPC method names from rpc_server.zig by matching:
42 std.mem.eql(u8, method, "methodname")
43
44 Also tries to extract parameter hints from nearby Usage comments.
45 """
46 with open(filepath, "r", encoding="utf-8", errors="replace") as f:
47 content = f.read()
48
49 # Find all method string literals
50 pattern = re.compile(r'std\.mem\.eql\‍(u8,\s*method,\s*"([a-zA-Z_][a-zA-Z0-9_]*)"\‍)')
51 raw_methods = pattern.findall(content)
52
53 # Deduplicate while preserving order
54 seen = set()
55 methods = []
56 for name in raw_methods:
57 if name not in seen:
58 seen.add(name)
59 methods.append(name)
60
61 # Try to extract parameter info from Usage comments
62 # Pattern: /// Usage: {"method":"NAME","params":["param1","param2",...],"id":1}
63 usage_pattern = re.compile(
64 r'///\s*Usage:\s*\{[^}]*"method"\s*:\s*"([^"]+)"[^}]*"params"\s*:\s*\‍[([^\‍]]*)\‍]',
65 re.MULTILINE,
66 )
67 usage_params: dict[str, list[str]] = {}
68 for m in usage_pattern.finditer(content):
69 method_name = m.group(1)
70 params_raw = m.group(2).strip()
71 if params_raw:
72 # Extract param names/types from the usage example
73 params = []
74 for p in re.findall(r'"([^"]*)"', params_raw):
75 params.append(p)
76 # Also catch numeric placeholders like count, amount_sat
77 for p in re.findall(r'\b([a-z_]+(?:_[a-z]+)*)\b', params_raw):
78 if p not in params and p not in ("true", "false"):
79 params.append(p)
80 usage_params[method_name] = params
81
82 result = []
83 for name in methods:
84 entry = {"name": name, "params": []}
85 if name in usage_params:
86 for i, p in enumerate(usage_params[name]):
87 # Infer type from name
88 if any(kw in p for kw in ["hash", "hex", "addr", "address", "data", "string", "id"]):
89 ptype = "string"
90 elif any(kw in p for kw in ["height", "count", "amount", "fee", "sat", "port", "nonce"]):
91 ptype = "number"
92 else:
93 ptype = "string"
94 param_name = re.sub(r'[^a-zA-Z0-9_]', '_', p).strip('_') or f"param{i}"
95 entry["params"].append({"name": param_name, "type": ptype})
96 result.append(entry)
97
98 return result

101def generate_typescript(methods: list[dict], class_name: str = "OmniBusRPC") -> str:
102 """Generate a TypeScript RPC client class."""
103 lines = [
104 "// =============================================================================",
105 "// OmniBus Blockchain Core — Auto-generated TypeScript RPC Client",
106 f"// Generated from core/rpc_server.zig ({len(methods)} methods)",
107 "// Do not edit manually — regenerate with: python scripts/frontend/generate-rpc-client.py",
108 "// =============================================================================",
109 "",
110 "export interface RPCResponse<T = unknown> {",
111 " jsonrpc: string;",
112 " id: number;",
113 " result?: T;",
114 " error?: { code: number; message: string };",
115 "}",
116 "",
117 f"export class {class_name} {{",
118 " private url: string;",
119 " private nextId = 1;",
120 "",
121 " constructor(url: string = 'http://127.0.0.1:8332') {",
122 " this.url = url;",
123 " }",
124 "",
125 " private async call<T = unknown>(method: string, params: unknown[] = []): Promise<T> {",
126 " const id = this.nextId++;",
127 " const res = await fetch(this.url, {",
128 " method: 'POST',",
129 " headers: { 'Content-Type': 'application/json' },",
130 " body: JSON.stringify({ jsonrpc: '2.0', id, method, params }),",
131 " });",
132 " if (!res.ok) {",
133 " throw new Error(`HTTP ${res.status}: ${res.statusText}`);",
134 " }",
135 " const data: RPCResponse<T> = await res.json();",
136 " if (data.error) {",
137 " throw new Error(`RPC ${data.error.code}: ${data.error.message}`);",
138 " }",
139 " return data.result as T;",
140 " }",
141 "",
142 ]
143
144 for m in methods:
145 name = m["name"]
146 params = m.get("params", [])
147
148 # Build typed parameter list
149 if params:
150 args_typed = ", ".join(f"{p['name']}: {p['type']}" for p in params)
151 args_pass = ", ".join(p["name"] for p in params)
152 else:
153 args_typed = ""
154 args_pass = ""
155
156 # Camel-case method name for the TS function
157 ts_name = name
158
159 lines.append(f" /** JSON-RPC method: {name} */")
160 lines.append(f" async {ts_name}({args_typed}): Promise<unknown> {{")
161 lines.append(f" return this.call('{name}', [{args_pass}]);")
162 lines.append(" }")
163 lines.append("")
164
165 lines.append("}")
166 lines.append("")
167 return "\n".join(lines)
