47def analyze_file(filepath: Path) -> dict:
48 """Analyze a single .zig file and return metrics."""
49 try:
50 content = filepath.read_text(encoding="utf-8", errors="replace")
51 except Exception as exc:
52 return {"error": str(exc)}
53
54 lines = content.split("\n")
55 total_lines = len(lines)
56 blank_lines = sum(1 for l in lines if not l.strip())
57 comment_lines = sum(1 for l in lines if l.strip().startswith("//"))
58 code_lines = total_lines - blank_lines - comment_lines
59
60 # Count functions: pub fn, fn, export fn
61 fn_pattern = re.compile(r'\b(?:pub\s+)?(?:export\s+)?fn\s+\w+')
62 functions = len(fn_pattern.findall(content))
63
64 # Count test blocks: test "name" {
65 test_pattern = re.compile(r'\btest\s+"[^"]*"')
66 tests = len(test_pattern.findall(content))
67
68 # Count imports: @import("...")
69 import_pattern = re.compile(r'@import\s*\‍(\s*"([^"]+)"\s*\‍)')
70 imports_raw = import_pattern.findall(content)
71 imports = len(imports_raw)
72 import_names = list(set(imports_raw))
73
74 # Count pub fn specifically
75 pub_fn_pattern = re.compile(r'\bpub\s+fn\s+\w+')
76 pub_functions = len(pub_fn_pattern.findall(content))
77
78 # Complexity score
79 score = (
80 code_lines * W_LINES +
81 functions * W_FUNCTIONS +
82 tests * W_TESTS +
83 imports * W_IMPORTS
84 ) / 100.0
85
86 return {
87 "file": filepath.name,
88 "path": str(filepath),
89 "total_lines": total_lines,
90 "code_lines": code_lines,
91 "blank_lines": blank_lines,
92 "comment_lines": comment_lines,
93 "functions": functions,
94 "pub_functions": pub_functions,
95 "tests": tests,
96 "imports": imports,
97 "import_names": import_names,
98 "complexity_score": round(score, 2),
99 }

102def print_table(results: list, top_n: int | None):
103 """Print a formatted table of results."""
104 # Sort by complexity descending
105 results.sort(key=lambda r: r.get("complexity_score", 0), reverse=True)
106 if top_n:
107 results = results[:top_n]
108
109 # Header
110 print(f"\n{BG_BLUE}{WHITE}{BOLD} OmniBus Module Complexity Analysis {RESET}\n")
111 header = (
112 f" {'#':>3s} {'Module':<35s} {'Lines':>6s} {'Code':>6s} "
113 f"{'Funcs':>5s} {'PubFn':>5s} {'Tests':>5s} {'Imps':>4s} {'Score':>7s}"
114 )
115 print(f"{BOLD}{header}{RESET}")
116 print(f" {DIM}{'─' * 95}{RESET}")
117
118 for i, r in enumerate(results, 1):
119 if "error" in r:
120 print(f" {i:3d} {r['file']:<35s} {RED}ERROR: {r['error']}{RESET}")
121 continue
122
123 # Colour score
124 score = r["complexity_score"]
125 if score > 20:
126 score_col = RED
127 elif score > 10:
128 score_col = YELLOW
129 else:
130 score_col = GREEN
131
132 name = r["file"]
133 if len(name) > 35:
134 name = name[:32] + "..."
135
136 print(
137 f" {i:3d} {CYAN}{name:<35s}{RESET} "
138 f"{r['total_lines']:6d} {r['code_lines']:6d} "
139 f"{r['functions']:5d} {r['pub_functions']:5d} "
140 f"{r['tests']:5d} {r['imports']:4d} "
141 f"{score_col}{score:7.2f}{RESET}"
142 )
143
144 print(f" {DIM}{'─' * 95}{RESET}")

147def print_summary(results: list):
148 """Print aggregate summary."""
149 valid = [r for r in results if "error" not in r]
150 if not valid:
151 print(f" {YELLOW}No valid files analyzed.{RESET}")
152 return
153
154 total_lines = sum(r["total_lines"] for r in valid)
155 total_code = sum(r["code_lines"] for r in valid)
156 total_funcs = sum(r["functions"] for r in valid)
157 total_pub = sum(r["pub_functions"] for r in valid)
158 total_tests = sum(r["tests"] for r in valid)
159 total_imports = sum(r["imports"] for r in valid)
160 avg_score = sum(r["complexity_score"] for r in valid) / len(valid)
161
162 print(f"\n {MAGENTA}{BOLD}Summary{RESET}")
163 print(f" {DIM}{'─' * 40}{RESET}")
164 print(f" Files analyzed: {BOLD}{len(valid)}{RESET}")
165 print(f" Total lines: {total_lines:,}")
166 print(f" Code lines: {total_code:,}")
167 print(f" Functions: {total_funcs:,} ({total_pub:,} public)")
168 print(f" Test blocks: {total_tests:,}")
169 print(f" Imports: {total_imports:,}")
170 print(f" Avg complexity: {avg_score:.2f}")
171 print(f" Most complex: {YELLOW}{valid[0]['file']}{RESET} (score: {valid[0]['complexity_score']})")
172 print()
