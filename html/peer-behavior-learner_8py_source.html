56def load_peer_score_jsonl(data_dir: str) -> list[dict[str, Any]]:
57 records: list[dict[str, Any]] = []
58 pattern = os.path.join(data_dir, "*.jsonl")
59 for fpath in glob.glob(pattern):
60 with open(fpath, "r", encoding="utf-8", errors="replace") as fh:
61 for line in fh:
62 line = line.strip()
63 if not line:
64 continue
65 try:
66 records.append(json.loads(line))
67 except json.JSONDecodeError:
68 continue
69 return records

84def parse_scoring_weights(repo_path: str) -> dict[str, Any]:
85 """Extract heuristic weights from peer_scoring.zig if available."""
86 fpath = os.path.join(repo_path, "core", "peer_scoring.zig")
87 if not os.path.isfile(fpath):
88 return {}
89 text = Path(fpath).read_text(encoding="utf-8", errors="replace")
90 weights: dict[str, Any] = {}
91 for m in re.finditer(r"(\w+)\s*[=:]\s*(\d+)", text):
92 key, val = m.group(1), int(m.group(2))
93 if any(k in key.lower() for k in ["score", "weight", "penalt", "ban", "threshold"]):
94 weights[key] = val
95 return weights

101def analyze_records(records: list[dict[str, Any]]) -> dict[str, Any]:
102 peer_stats: dict[str, dict[str, Any]] = defaultdict(
103 lambda: {"events": 0, "penalties": 0, "last_score": None, "reasons": defaultdict(int)}
104 )
105
106 for rec in records:
107 peer_id = rec.get("peer_id") or rec.get("id") or rec.get("addr", "unknown")
108 stats = peer_stats[peer_id]
109 stats["events"] += 1
110 score = rec.get("score")
111 if score is not None:
112 if stats["last_score"] is not None and score < stats["last_score"]:
113 stats["penalties"] += 1
114 reason = rec.get("reason", "unknown")
115 stats["reasons"][reason] += 1
116 stats["last_score"] = score
117
118 # Identify malicious patterns
119 malicious: list[dict[str, Any]] = []
120 for peer_id, stats in peer_stats.items():
121 if stats["penalties"] >= 3:
122 top_reason = max(stats["reasons"].items(), key=lambda kv: kv[1])[0] if stats["reasons"] else "unknown"
123 malicious.append(
124 {
125 "peer_id": peer_id,
126 "total_events": stats["events"],
127 "penalties": stats["penalties"],
128 "top_reason": top_reason,
129 }
130 )
131
132 # Optimize weights (simple heuristic: increase weight for top penalty reasons)
133 global_reasons: dict[str, int] = defaultdict(int)
134 for stats in peer_stats.values():
135 for reason, count in stats["reasons"].items():
136 global_reasons[reason] += count
137
138 optimized_weights: dict[str, float] = {}
139 total = sum(global_reasons.values()) or 1
140 for reason, count in global_reasons.items():
141 optimized_weights[reason] = round(count / total, 4)
142
143 return {
144 "peers_observed": len(peer_stats),
145 "malicious_peers_detected": len(malicious),
146 "malicious_list": malicious,
147 "penalty_reason_distribution": dict(global_reasons),
148 "optimized_weight_map": optimized_weights,
149 }
