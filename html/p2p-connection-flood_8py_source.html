36class ConnectionTracker:
37 def __init__(self):
38 self.lock = threading.Lock()
39 self.connected = 0
40 self.rejected = 0
41 self.errors = 0
42 self.timeouts = 0

47 def record_connect(self, sock, connect_time_ms: float):
48 with self.lock:
49 self.connected += 1
50 self.sockets.append(sock)
51 self.connect_times.append(connect_time_ms)

53 def record_reject(self):
54 with self.lock:
55 self.rejected += 1

59 def record_error(self):

63 def record_timeout(self):
64 with self.lock:
65 self.timeouts += 1

67 def close_all(self):
68 with self.lock:

77def build_handshake_partial() -> bytes:
78 """Build a partial OmniBus P2P handshake message.
79
80 Structure (Bitcoin-inspired):
81 4 bytes: magic (OMNI)
82 12 bytes: command (version\x00...)
83 4 bytes: payload length
84 4 bytes: checksum (first 4 bytes of double-SHA256)
85 ... partial payload (intentionally incomplete)
86 """
87 magic = OMNIBUS_MAGIC
88 command = b"version\x00\x00\x00\x00\x00" # 12 bytes
89 # Claim large payload but only send partial
90 payload_len = struct.pack("<I", 1024)
91 checksum = b"\x00\x00\x00\x00" # fake checksum
92
93 # Send just the header + a few bytes of "payload"
94 partial_payload = b"\x00" * 20 # Only 20 of claimed 1024 bytes
95
96 return magic + command + payload_len + checksum + partial_payload

99def connection_worker(host: str, port: int, conn_id: int,
100 tracker: ConnectionTracker, hold_time: float):
101 """Open one connection, send partial handshake, hold open."""
102 sock = None
103 try:
104 sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
105 sock.settimeout(5)
106
107 t0 = time.time()
108 sock.connect((host, port))
109 connect_ms = (time.time() - t0) * 1000
110
111 # Send partial handshake
112 handshake = build_handshake_partial()
113 sock.sendall(handshake)
114
115 tracker.record_connect(sock, connect_ms)
116
117 # Hold connection open
118 sock.settimeout(hold_time + 1)
119 try:
120 # Try to read response (node might send version back)
121 data = sock.recv(1024)
122 except socket.timeout:
123 pass
124
125 # Keep alive
126 time.sleep(hold_time)
127
128 except ConnectionRefusedError:
129 tracker.record_reject()
130 except socket.timeout:
131 tracker.record_timeout()
132 except OSError as e:
133 if "refused" in str(e).lower() or "reset" in str(e).lower():
134 tracker.record_reject()
135 else:
136 tracker.record_error()
137 except Exception:
138 tracker.record_error()
139 finally:
140 if sock:
141 try:
142 sock.close()
143 except Exception:
144 pass

161 import http.client
