94 def audit_secp256k1(self) -> None:
95 """Verify secp256k1 test vectors by checking source file presence."""
96 secp_file = os.path.join(self.core_dir, "secp256k1.zig")
97 if not os.path.isfile(secp_file):
98 self._add_result("secp256k1 presence", "FAIL", "secp256k1.zig not found")
99 return
100 with open(secp_file, "r", encoding="utf-8", errors="ignore") as f:
101 content = f.read()
102 # Check for constant-time patterns (simple heuristic)
103 has_ct = "subtle.ConstantTimeEq" in content or "constant_time" in content.lower() or "timing_safe" in content.lower()
104 if has_ct:
105 self._add_result("secp256k1 constant-time", "PASS", "Constant-time primitives detected")
106 else:
107 self._add_result("secp256k1 constant-time", "WARN", "No explicit constant-time primitives found")
108 # Test vector presence heuristic
109 has_nist = "nist" in content.lower() or "test vector" in content.lower()
110 if has_nist:
111 self._add_result("secp256k1 NIST vectors", "PASS", "NIST test vectors referenced")
112 else:
113 self._add_result("secp256k1 NIST vectors", "WARN", "NIST test vectors not explicitly referenced")
114 # Wycheproof

157 def audit_timing(self) -> None:
158 """Run simple timing analysis to detect non-constant-time branches."""
159 files = ["secp256k1.zig", "crypto.zig", "schnorr.zig"]
160 issues = []
161 for fname in files:
162 fpath = os.path.join(self.core_dir, fname)
163 if not os.path.isfile(fpath):
164 continue
165 with open(fpath, "r", encoding="utf-8", errors="ignore") as f:
166 lines = f.readlines()
167 for i, line in enumerate(lines, 1):
168 stripped = line.strip()
169 if stripped.startswith("if (") and ("secret" in stripped.lower() or "key" in stripped.lower()):
170 issues.append(f"{fname}:{i} potential branch on secret")
171 if issues:
172 self._add_result("Timing analysis", "WARN", f"{len(issues)} potential timing leaks: " + issues[0])
173 else:
174 self._add_result("Timing analysis", "PASS", "No obvious secret-dependent branches detected")
