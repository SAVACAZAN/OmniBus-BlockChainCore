85 def verify_merkle_proof(self, tx_hash: str, merkle_root: str, proof: List[str]) -> bool:
86 """Verify a Merkle proof for a transaction."""
87 current = bytes.fromhex(tx_hash)
88
89 for sibling in proof:
90 sibling_bytes = bytes.fromhex(sibling)
91 # Concatenate and hash
92 if current < sibling_bytes:
93 current = hashlib.sha256(current + sibling_bytes).digest()
94 else:
95 current = hashlib.sha256(sibling_bytes + current).digest()
96
97 return current.hex() == merkle_root

99 def verify_bitcoin_spv(self, tx: BridgeTransaction, block_header: Dict, merkle_proof: List[str]) -> Tuple[bool, str]:
100 """Verify Bitcoin SPV (Simplified Payment Verification) proof."""
101 print(f" Verifying Bitcoin SPV for tx: {tx.tx_hash}")
102
103 # 1. Verify transaction hash format
104 if not self._is_valid_hash(tx.tx_hash, 64):
105 return False, "Invalid transaction hash format"
106
107 # 2. Verify block header
108 if not block_header or 'merkle_root' not in block_header:
109 return False, "Invalid block header"
110
111 # 3. Verify Merkle proof
112 if not merkle_proof:
113 return False, "Merkle proof required"
114
115 is_valid = self.verify_merkle_proof(
116 tx.tx_hash,
117 block_header['merkle_root'],
118 merkle_proof
119 )
120
121 if not is_valid:
122 return False, "Merkle proof verification failed"
123
124 # 4. Check confirmations
125 min_conf = self.MIN_CONFIRMATIONS[ChainType.BITCOIN]
126 if tx.confirmations < min_conf:
127 return False, f"Insufficient confirmations: {tx.confirmations}/{min_conf}"
128
129 return True, f"SPV verified ({tx.confirmations} confirmations)"

131 def verify_ethereum_proof(self, tx: BridgeTransaction, receipt: Dict) -> Tuple[bool, str]:
132 """Verify Ethereum transaction receipt and logs."""
133 print(f" Verifying Ethereum receipt for tx: {tx.tx_hash}")
134
135 # 1. Check receipt exists
136 if not receipt:
137 return False, "Receipt not found"
138
139 # 2. Verify status (1 = success)
140 if receipt.get('status') != '0x1':
141 return False, "Transaction failed on Ethereum"
142
143 # 3. Verify to address is bridge contract
144 bridge_addr = self.BRIDGE_CONTRACTS[ChainType.ETHEREUM]
145 if receipt.get('to', '').lower() != bridge_addr.lower():
146 return False, "Not sent to bridge contract"
147
148 # 4. Check logs for Lock event
149 logs = receipt.get('logs', [])
150 has_lock_event = any(
151 'Lock' in log.get('topics', [])[0]
152 for log in logs if log.get('topics')
153 )
154
155 if not has_lock_event:
156 return False, "No Lock event found in logs"
157
158 # 5. Check confirmations
159 min_conf = self.MIN_CONFIRMATIONS[ChainType.ETHEREUM]
160 if tx.confirmations < min_conf:
161 return False, f"Insufficient confirmations: {tx.confirmations}/{min_conf}"
162
163 return True, f"Ethereum receipt verified ({tx.confirmations} confirmations)"

185 def validate_bridge_tx(self, tx: BridgeTransaction, proof_data: Dict) -> Tuple[bool, str]:
186 """Validate a complete bridge transaction."""
187 print(f"\nValidating bridge transaction:")
188 print(f" Hash: {tx.tx_hash}")
189 print(f" Source: {tx.source_chain.value}")
190 print(f" Target: {tx.target_chain.value}")
191 print(f" Amount: {tx.amount} {tx.token}")

193 # Route to appropriate verifier
194 if tx.source_chain == ChainType.BITCOIN:
195 return self.verify_bitcoin_spv(
196 tx,
197 proof_data.get('block_header', {}),
198 proof_data.get('merkle_proof', [])
199 )
200
201 elif tx.source_chain == ChainType.ETHEREUM:
202 return self.verify_ethereum_proof(tx, proof_data.get('receipt', {}))

204 elif tx.source_chain == ChainType.SOLANA:
205 return self.verify_solana_proof(tx, proof_data.get('signature_status', {}))
206
207 elif tx.source_chain == ChainType.OMNIBUS:
208 # Internal OmniBus transaction
209 return self._verify_omnibus_tx(tx, proof_data)
210
211 return False, f"Unsupported source chain: {tx.source_chain}"

222 def check_double_spend(self, tx: BridgeTransaction) -> bool:
223 """Check if this transaction is a double-spend."""
224 # Check against validated transactions
225 for validated in self.validated_txs:
226 if (validated.tx_hash == tx.tx_hash and
227 validated.source_chain == tx.source_chain):
228 return True
229
230 # Check against pending transactions
231 for pending in self.pending_txs:
232 if (pending.tx_hash == tx.tx_hash and
233 pending.source_chain == tx.source_chain and
234 pending.status == TxStatus.CONFIRMED):
235 return True
236
237 return False

239 def estimate_completion_time(self, tx: BridgeTransaction) -> Dict:
240 """Estimate time until transaction is finalized."""
241 min_conf = self.MIN_CONFIRMATIONS[tx.source_chain]
242 remaining = max(0, min_conf - tx.confirmations)
243
244 # Average block times (seconds)
245 block_times = {
246 ChainType.BITCOIN: 600,
247 ChainType.ETHEREUM: 12,
248 ChainType.SOLANA: 0.4,
249 ChainType.OMNIBUS: 10,
250 }
251
252 estimated_seconds = remaining * block_times.get(tx.source_chain, 60)
253
254 return {
255 "current_confirmations": tx.confirmations,
256 "required_confirmations": min_conf,
257 "remaining_confirmations": remaining,
258 "estimated_seconds": estimated_seconds,
259 "estimated_minutes": estimated_seconds / 60,
260 }
