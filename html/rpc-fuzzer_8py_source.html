224def send_rpc(url: str, payload: dict[str, Any] | list[Any] | str, timeout: float = 10.0) -> dict[str, Any]:
225 result: dict[str, Any] = {
226 "http_status": None,
227 "response": None,
228 "error": None,
229 "elapsed_ms": 0.0,
230 }
231 headers = {"Content-Type": "application/json"}
232 start = time.perf_counter()
233 try:
234 if isinstance(payload, str):
235 resp = requests.post(url, data=payload, headers=headers, timeout=timeout)
236 else:
237 resp = requests.post(url, json=payload, headers=headers, timeout=timeout)
238 result["http_status"] = resp.status_code
239 try:
240 result["response"] = resp.json()
241 except Exception:
242 result["response"] = resp.text[:512]
243 except requests.exceptions.ConnectionError as exc:
244 result["error"] = f"ConnectionError: {exc}"
245 except requests.exceptions.Timeout:
246 result["error"] = "Timeout"
247 except Exception as exc:
248 result["error"] = f"Exception: {exc}"
249 result["elapsed_ms"] = round((time.perf_counter() - start) * 1000, 2)
250 return result
