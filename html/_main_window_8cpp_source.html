170void MainWindow::refreshWalletToolbar() {
171 auto& wm = WalletManager::instance();
172
173 // Block signals to prevent re-triggering currentIndexChanged
174 m_walletCombo->blockSignals(true);
175 m_walletCombo->clear();
176
177 auto wallets = wm.listWallets();
178 for (const auto& w : wallets) {
179 m_walletCombo->addItem(w.name, w.id);
180 }
181
182 // Select current wallet
183 for (int i = 0; i < m_walletCombo->count(); ++i) {
184 if (m_walletCombo->itemData(i).toString() == wm.currentWalletId()) {
185 m_walletCombo->setCurrentIndex(i);
186 break;
187 }
188 }
189 m_walletCombo->blockSignals(false);
190
191 // Update address and balance
192 if (wm.isUnlocked()) {
193 QString addr = wm.primaryAddress();
194 if (addr.length() > 20) {
195 m_walletAddrLabel->setText(addr.left(10) + "..." + addr.right(8));
196 } else {
197 m_walletAddrLabel->setText(addr);
198 }
199 m_walletAddrLabel->setToolTip(addr + "\nClick to copy");
200
201 // Balance comes from node if connected, otherwise show "offline"
202 if (m_mode == NodeMode) {
203 // Balance will be updated by NodeService signals
204 } else {
205 m_walletBalanceLabel->setText("Offline Wallet");
206 }
207 } else {
208 m_walletAddrLabel->setText("(locked)");
209 m_walletBalanceLabel->setText("");
210 }
