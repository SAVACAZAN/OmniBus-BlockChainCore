34def run_zig_benchmark(test_name: str, iterations: int = 1000) -> Dict:
35 """Run a Zig test group and measure time."""
36 import time as _time
37 start = _time.time()
38 try:
39 result = subprocess.run(
40 ["zig", "build", test_name],
41 cwd=ROOT, capture_output=True, text=True, timeout=120
42 )
43 elapsed_ms = (_time.time() - start) * 1000
44 passed = result.returncode == 0
45 return {
46 "test": test_name,
47 "iterations": iterations,
48 "total_time_ms": round(elapsed_ms),
49 "ops_per_sec": round(iterations / (elapsed_ms / 1000)) if elapsed_ms > 0 else 0,
50 "passed": passed,
51 }
52 except Exception as e:
53 return {
54 "test": test_name,
55 "iterations": iterations,
56 "total_time_ms": 0,
57 "ops_per_sec": 0,
58 "passed": False,
59 "error": str(e),
60 }
