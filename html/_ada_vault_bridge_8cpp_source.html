22bool AdaVaultBridge::loadLibrary(const QString& path) {
23 if (m_loaded) return true;
24
25 // Try: same dir as exe, then system path
26 QStringList searchPaths = {
27 QCoreApplication::applicationDirPath() + "/" + path,
28 path
29 };
30
31 for (const auto& p : searchPaths) {
32 m_lib.setFileName(p);
33 if (m_lib.load()) break;
34 }
35
36 if (!m_lib.isLoaded()) return false;
37
38 // Resolve all symbols
39 m_libInit = (FnVoid) m_lib.resolve("vault_lib_init");
40 m_init = (FnInt) m_lib.resolve("vault_init");
41 m_lock = (FnInt) m_lib.resolve("vault_lock");
42 m_save = (FnInt) m_lib.resolve("vault_save");
43 m_isLoaded = (FnInt) m_lib.resolve("vault_is_loaded");
44 m_addKey = (FnAddKey) m_lib.resolve("vault_add_key");
45 m_delKey = (FnDelKey) m_lib.resolve("vault_delete_key");
46 m_updKey = (FnUpdKey) m_lib.resolve("vault_update_key");
47 m_setStat = (FnSetStat) m_lib.resolve("vault_set_status");
48 m_keyCount = (FnIntI) m_lib.resolve("vault_key_count");
49 m_hasKeys = (FnIntI) m_lib.resolve("vault_has_keys");
50 m_getKey = (FnGetKey) m_lib.resolve("vault_get_key");
51 m_getSec = (FnGetSec) m_lib.resolve("vault_get_secret");
52 m_wipe = (FnWipe) m_lib.resolve("vault_wipe");
53 m_getPath = (FnGetPath) m_lib.resolve("vault_get_path");
54
55 if (!m_init || !m_lock || !m_save || !m_isLoaded ||
56 !m_addKey || !m_getKey) {
57 m_lib.unload();
58 return false;
59 }
60
61 // Initialize Ada runtime
62 if (m_libInit) m_libInit();
63
64 m_loaded = true;
65 return true;
66}

91bool AdaVaultBridge::addKey(VaultExchange ex, const QString& name,
92 const QString& apiKey, const QString& secret,
93 VaultKeyStatus status) {
94 if (!m_loaded || !m_addKey) return false;
95 auto n = name.toUtf8();
96 auto k = apiKey.toUtf8();
97 auto s = secret.toUtf8();
98 int rc = m_addKey(ex, n.constData(), k.constData(), s.constData(), status);
99
100 // Secure wipe local copies
101 if (m_wipe) {
102 m_wipe(const_cast<char*>(s.constData()), s.size());
103 }
104 return rc == 0;
105}

145VaultKeyEntry AdaVaultBridge::getKey(VaultExchange ex, int slot) const {
146 VaultKeyEntry e;
147 if (!m_loaded || !m_getKey) return e;
148
149 char nameBuf[256] = {};
150 char keyBuf[256] = {};
151 int statusOut = 0, inUseOut = 0;
152
153 int rc = m_getKey(ex, slot, nameBuf, 256, keyBuf, 256,
154 &statusOut, &inUseOut);
155 if (rc == 0) {
156 e.inUse = (inUseOut != 0);
157 e.status = static_cast<VaultKeyStatus>(statusOut);
158 e.name = QString::fromUtf8(nameBuf);
159 e.apiKey = QString::fromUtf8(keyBuf);
160 }
161 return e;
162}
