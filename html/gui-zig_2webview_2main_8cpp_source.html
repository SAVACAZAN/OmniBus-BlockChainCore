36// DPAPI encrypt
37static std::vector<uint8_t> dpapiEncrypt(const std::vector<uint8_t>& plain) {
38 DATA_BLOB in_blob, out_blob;
39 in_blob.pbData = const_cast<BYTE*>(plain.data());
40 in_blob.cbData = (DWORD)plain.size();
41 if (!CryptProtectData(&in_blob, L"OmniBusVault", NULL, NULL, NULL, 0, &out_blob))
42 return {};
43 std::vector<uint8_t> result(out_blob.pbData, out_blob.pbData + out_blob.cbData);
44 LocalFree(out_blob.pbData);
45 return result;
46}

48// DPAPI decrypt
49static std::vector<uint8_t> dpapiDecrypt(const std::vector<uint8_t>& cipher) {
50 DATA_BLOB in_blob, out_blob;
51 in_blob.pbData = const_cast<BYTE*>(cipher.data());
52 in_blob.cbData = (DWORD)cipher.size();
53 if (!CryptUnprotectData(&in_blob, NULL, NULL, NULL, NULL, 0, &out_blob))
54 return {};
55 std::vector<uint8_t> result(out_blob.pbData, out_blob.pbData + out_blob.cbData);
56 SecureZeroMemory(out_blob.pbData, out_blob.cbData);
57 LocalFree(out_blob.pbData);
58 return result;
59}

61// Serialize string: [4-byte len][data]
62static void writeStr(std::vector<uint8_t>& buf, const std::string& s) {
63 uint32_t len = (uint32_t)s.size();
64 buf.insert(buf.end(), (uint8_t*)&len, (uint8_t*)&len + 4);
65 buf.insert(buf.end(), s.begin(), s.end());
66}

68static std::string readStr(const uint8_t*& ptr, const uint8_t* end) {
69 if (ptr + 4 > end) return "";
70 uint32_t len = *(uint32_t*)ptr; ptr += 4;
71 if (ptr + len > end || len > 65536) return "";
72 std::string s((char*)ptr, len); ptr += len;
73 return s;
74}

76// Save vault JSON as DPAPI-encrypted file
77static bool saveVault(const std::string& json) {
78 std::vector<uint8_t> plain;
79 // Header
80 plain.insert(plain.end(), VAULT_MAGIC, VAULT_MAGIC + 4);
81 uint32_t ver = VAULT_VERSION;
82 plain.insert(plain.end(), (uint8_t*)&ver, (uint8_t*)&ver + 4);
83 // JSON payload
84 writeStr(plain, json);
85
86 auto encrypted = dpapiEncrypt(plain);
87 SecureZeroMemory(plain.data(), plain.size());
88 if (encrypted.empty()) return false;
89
90 std::string path = getVaultPath();
91 FILE* f = fopen(path.c_str(), "wb");
92 if (!f) return false;
93 fwrite(encrypted.data(), 1, encrypted.size(), f);
94 fclose(f);
95 return true;
96}

98// Load vault: read file, DPAPI decrypt, return JSON
99static std::string loadVault() {
100 std::string path = getVaultPath();
101 FILE* f = fopen(path.c_str(), "rb");
102 if (!f) return "[]";
103 fseek(f, 0, SEEK_END);
104 long sz = ftell(f);
105 fseek(f, 0, SEEK_SET);
106 if (sz <= 0 || sz > 10*1024*1024) { fclose(f); return "[]"; }
107 std::vector<uint8_t> cipher(sz);
108 fread(cipher.data(), 1, sz, f);
109 fclose(f);
110
111 auto plain = dpapiDecrypt(cipher);
112 if (plain.size() < 12) return "[]"; // magic(4) + version(4) + len(4) min
113
114 // Verify magic
115 if (memcmp(plain.data(), VAULT_MAGIC, 4) != 0) {
116 SecureZeroMemory(plain.data(), plain.size());
117 return "[]";
118 }
119
120 const uint8_t* ptr = plain.data() + 8; // skip magic + version
121 const uint8_t* end = plain.data() + plain.size();
122 std::string json = readStr(ptr, end);
123 SecureZeroMemory(plain.data(), plain.size());
124 return json;
125}

161 // Bind: vault_load() → returns JSON array of API keys
162 webview_bind(w, "vault_load",
163 [](const char* id, const char* req, void* arg) {
164 webview_t wv = (webview_t)arg;
165 std::string json = loadVault();
166 // Escape for JSON string return
167 std::string escaped = "\"";
168 for (char c : json) {
169 if (c == '"') escaped += "\\\"";
170 else if (c == '\\') escaped += "\\\\";
171 else if (c == '\n') escaped += "\\n";
172 else escaped += c;
173 }
174 escaped += "\"";
175 webview_return(wv, id, 0, escaped.c_str());
176 }, w);
177
178 // Bind: vault_save(json_string) → saves encrypted
179 webview_bind(w, "vault_save",
180 [](const char* id, const char* req, void* arg) {
181 webview_t wv = (webview_t)arg;
182 // req is JSON array of arguments: ["json_string"]
183 std::string r(req);
184 // Extract first argument (the JSON string)
185 // Format: ["escaped_json_content"]
186 size_t first_quote = r.find('"');
187 size_t last_quote = r.rfind('"');
188 std::string json_escaped;
189 if (first_quote != std::string::npos && last_quote > first_quote) {
190 json_escaped = r.substr(first_quote + 1, last_quote - first_quote - 1);
191 }
192 // Unescape
193 std::string json;
194 for (size_t i = 0; i < json_escaped.size(); i++) {
195 if (json_escaped[i] == '\\' && i + 1 < json_escaped.size()) {
196 char next = json_escaped[i+1];
197 if (next == '"') { json += '"'; i++; }
198 else if (next == '\\') { json += '\\'; i++; }
199 else if (next == 'n') { json += '\n'; i++; }
200 else json += json_escaped[i];
201 } else {
202 json += json_escaped[i];
203 }
204 }
205 bool ok = saveVault(json);
206 webview_return(wv, id, 0, ok ? "true" : "false");
207 }, w);
208
