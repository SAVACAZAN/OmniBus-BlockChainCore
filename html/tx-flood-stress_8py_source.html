47class StressResults:
48 def __init__(self):
49 self.lock = threading.Lock()
50 self.latencies = []
51 self.errors = 0
52 self.successes = 0
53 self.by_method = defaultdict(lambda: {"ok": 0, "err": 0, "latencies": []})
54 self.start_time = time.time()
55
56 def record(self, method: str, latency: float, success: bool):
57 with self.lock:
58 self.latencies.append(latency)
59 if success:
60 self.successes += 1
61 self.by_method[method]["ok"] += 1
62 else:
63 self.errors += 1
64 self.by_method[method]["err"] += 1
65 self.by_method[method]["latencies"].append(latency)
66
67 def total(self) -> int:
68 return self.successes + self.errors
69
70 def rps(self) -> float:
71 elapsed = time.time() - self.start_time
72 return self.total() / max(elapsed, 0.001)
73
74 def percentile(self, p: float) -> float:
75 with self.lock:
76 if not self.latencies:
77 return 0.0
78 sorted_lat = sorted(self.latencies)
79 idx = int(len(sorted_lat) * p / 100)
80 return sorted_lat[min(idx, len(sorted_lat) - 1)]

83def rpc_request(host: str, port: int, method: str, params: list = None) -> tuple:
84 """Send JSON-RPC request. Returns (success, latency_ms)."""
85 payload = json.dumps({
86 "jsonrpc": "2.0",
87 "id": random.randint(1, 999999),
88 "method": method,
89 "params": params or [],
90 })
91
92 t0 = time.time()
93 try:
94 conn = http.client.HTTPConnection(host, port, timeout=10)
95 conn.request("POST", "/", payload, {"Content-Type": "application/json"})
96 resp = conn.getresponse()
97 data = resp.read()
98 conn.close()
99 latency_ms = (time.time() - t0) * 1000
100
101 result = json.loads(data)
102 success = "error" not in result or result.get("result") is not None
103 return success, latency_ms
104 except Exception:
105 latency_ms = (time.time() - t0) * 1000
106 return False, latency_ms

109def worker_valid_requests(host: str, port: int, count: int, results: StressResults):
110 """Send valid RPC requests."""
111 methods = ["getblockcount", "getbestblockhash", "getblockchaininfo",
112 "getmempoolinfo", "getpeerinfo", "getnetworkinfo"]
113 for _ in range(count):
114 method = random.choice(methods)
115 success, latency = rpc_request(host, port, method)
116 results.record(method, latency, success)

119def worker_raw_tx(host: str, port: int, count: int, results: StressResults):
120 """Send sendrawtransaction with random hex data."""

128def worker_connection_hold(host: str, port: int, duration: float, results: StressResults):
129 """Open connection and hold it."""
130 import socket
131 try:
132 sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
133 sock.settimeout(5)
134 t0 = time.time()
135 sock.connect((host, port))
136 # Send partial HTTP request to hold connection
137 sock.sendall(b"POST / HTTP/1.1\r\nHost: localhost\r\nContent-Length: 999999\r\n\r\n")
138 time.sleep(duration)
139 sock.close()
140 latency = (time.time() - t0) * 1000
141 results.record("connection_hold", latency, True)
142 except Exception:
143 results.record("connection_hold", 0, False)

146def print_dashboard(results: StressResults, phase: str):
147 """Print live stats."""
148 total = results.total()
149 rps = results.rps()
150 err_rate = (results.errors / max(total, 1)) * 100
151 p50 = results.percentile(50)
152 p95 = results.percentile(95)
153 p99 = results.percentile(99)
154
155 sys.stdout.write(f"{CLEAR_LINE}{CYAN}[{phase}]{RESET} "
156 f"total={total} rps={rps:.0f} "
157 f"err={err_rate:.1f}% "
158 f"p50={p50:.1f}ms p95={p95:.1f}ms p99={p99:.1f}ms")
159 sys.stdout.flush()

162def run_phase(name: str, target, args_list: list, results: StressResults,
163 thread_count: int):
164 """Run a phase with N threads."""
165 threads = []
166 for i in range(thread_count):
167 t = threading.Thread(target=target, args=args_list[i % len(args_list)],
168 daemon=True)
169 threads.append(t)
170
171 for t in threads:
172 t.start()
173
174 # Monitor
175 while any(t.is_alive() for t in threads):
176 print_dashboard(results, name)
177 time.sleep(0.5)
178
179 for t in threads:
180 t.join(timeout=15)
181
182 print_dashboard(results, name)
183 print() # newline after dashboard
