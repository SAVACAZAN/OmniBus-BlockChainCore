29def simulate_difficulty(block_times: List[float]) -> List[Dict[str, Any]]:
30 """Given a list of block times, return difficulty per interval."""
31 difficulties: List[Dict[str, Any]] = []
32 current_diff = 1.0
33
34 for i in range(0, len(block_times), DIFF_ADJUSTMENT_INTERVAL):
35 interval = block_times[i : i + DIFF_ADJUSTMENT_INTERVAL]
36 if len(interval) < 2:
37 break
38 avg_time = sum(interval) / len(interval)
39 # Simple proportional adjustment
40 adjustment = TARGET_BLOCK_TIME / avg_time if avg_time > 0 else 1.0
41 adjustment = max(0.25, min(4.0, adjustment)) # limit adjustment factor
42 current_diff *= adjustment
43 difficulties.append({
44 "interval_start": i,
45 "interval_end": i + len(interval),
46 "avg_block_time_sec": round(avg_time, 2),
47 "adjustment_factor": round(adjustment, 4),
48 "new_difficulty": round(current_diff, 4),
49 })
50 return difficulties

53def generate_synthetic_history(num_blocks: int, noise_factor: float = 0.2) -> List[float]:
