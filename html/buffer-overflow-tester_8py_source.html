48class RPCClient:
49 """Minimal JSON-RPC 2.0 client using stdlib http.client."""

51 def __init__(self, host: str, port: int, timeout: int = 10):
52 self.host = host
53 self.port = port
54 self.timeout = timeout
55 self._id = 0
56
57 def send_raw(self, body: bytes | str, timeout: int | None = None) -> tuple[int, str]:
58 """Send raw bytes to RPC endpoint, return (status_code, body)."""
59 t = timeout or self.timeout
60 try:
61 conn = http.client.HTTPConnection(self.host, self.port, timeout=t)
62 if isinstance(body, str):
63 body = body.encode("utf-8", errors="replace")
64 conn.request("POST", "/", body, {"Content-Type": "application/json"})
65 resp = conn.getresponse()
66 data = resp.read().decode("utf-8", errors="replace")
67 conn.close()
68 return resp.status, data
69 except (ConnectionRefusedError, OSError) as e:
70 return -1, str(e)
71 except Exception as e:
72 return -2, str(e)
73
74 def call(self, method: str, params: list | dict | None = None, timeout: int | None = None) -> tuple[int, str]:
75 """Send a JSON-RPC 2.0 request."""
76 self._id += 1
77 payload = {
78 "jsonrpc": "2.0",
79 "method": method,
80 "params": params or [],
81 "id": self._id,
82 }
83 body = json.dumps(payload)
84 return self.send_raw(body, timeout)

86 def is_alive(self) -> bool:
87 """Check if node is responding."""
88 try:
89 status, body = self.call("getblockcount", [], timeout=5)
90 return status == 200
91 except Exception:
92 return False

95def classify_result(status: int, body: str) -> str:
96 """Classify response as: error (good), crash (bad), or accepted (investigate)."""
97 if status == -1:
98 return "CRASH" # Connection refused — node may have died
99 if status == -2:
100 return "CRASH" # Timeout or other error
101 if status >= 400:
102 return "ERROR" # Server rejected — good
103 if status == 200:
104 try:
105 resp = json.loads(body)
106 if "error" in resp and resp["error"]:
107 return "ERROR" # JSON-RPC error — good
108 return "ACCEPTED" # Silently accepted — investigate
109 except json.JSONDecodeError:
110 return "ERROR"
111 return "ERROR"

114def build_nested_json(depth: int) -> dict:
115 """Build a JSON object nested depth levels deep."""
116 obj: dict = {"leaf": True}
117 for i in range(depth):
118 obj = {"level": i, "child": obj}
119 return obj
