50def socks5_connect(proxy_host: str, proxy_port: int,
51 target_host: str, target_port: int) -> socket.socket:
52 """SOCKS5 connection (no auth)."""
53 sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
54 sock.settimeout(10)
55 sock.connect((proxy_host, proxy_port))
56 sock.sendall(b"\x05\x01\x00")
57 resp = sock.recv(2)
58 if resp != b"\x05\x00":
59 sock.close()
60 raise ConnectionError("SOCKS5 handshake failed")
61
62 addr_bytes = socket.inet_aton(target_host)
63 port_bytes = struct.pack(">H", target_port)
64 sock.sendall(b"\x05\x01\x00\x01" + addr_bytes + port_bytes)
65 resp = sock.recv(10)
66 if len(resp) < 2 or resp[1] != 0:
67 sock.close()
68 raise ConnectionError(f"SOCKS5 connect failed: {resp.hex()}")
69 return sock

72def rpc_call_direct(host: str, port: int, method: str, params=None) -> tuple:
73 """Direct RPC call. Returns (json_response, http_headers_dict)."""
74 payload = json.dumps({"jsonrpc": "2.0", "id": 1, "method": method, "params": params or []})
75 try:
76 conn = http.client.HTTPConnection(host, port, timeout=10)
77 conn.request("POST", "/", payload, {"Content-Type": "application/json"})
78 resp = conn.getresponse()
79 headers = dict(resp.getheaders())
80 body = json.loads(resp.read().decode())
81 conn.close()
82 return body, headers
83 except Exception as exc:
84 return {"error": str(exc)}, {}

87def rpc_call_tor(proxy_host: str, proxy_port: int,
88 target_host: str, target_port: int,
89 method: str, params=None) -> tuple:
90 """RPC call through Tor. Returns (json_response, raw_headers_str)."""
91 payload = json.dumps({"jsonrpc": "2.0", "id": 1, "method": method, "params": params or []})

93 sock = socks5_connect(proxy_host, proxy_port, target_host, target_port)
94 http_req = (

96 f"Host: {target_host}:{target_port}\r\n"
97 f"Content-Type: application/json\r\n"
98 f"Content-Length: {len(payload)}\r\n"
99 f"Connection: close\r\n"
100 f"\r\n{payload}"
101 )
102 sock.sendall(http_req.encode())
103
104 response = b""
105 while True:
106 try:
107 chunk = sock.recv(4096)
108 if not chunk:
109 break
110 response += chunk
111 except socket.timeout:
112 break
113 sock.close()
114
115 resp_text = response.decode("utf-8", errors="replace")
116 header_end = resp_text.find("\r\n\r\n")
117 if header_end >= 0:
118 headers_raw = resp_text[:header_end]
119 body = resp_text[header_end + 4:]
120 return json.loads(body), headers_raw
121 return {"error": "no body"}, resp_text[:500]
122 except Exception as exc:
123 return {"error": str(exc)}, ""

126def find_ips_in_data(data, exclude_private: bool = True) -> list:
127 """Recursively search for IP addresses in JSON data."""
128 found = []
129 text = json.dumps(data) if not isinstance(data, str) else data
130
131 for ip in IPV4_PATTERN.findall(text):
132 if exclude_private and PRIVATE_IPS.match(ip):
133 continue
134 found.append({"type": "IPv4", "ip": ip})
135
136 for ip in IPV6_PATTERN.findall(text):
137 if exclude_private and PRIVATE_IPS.match(ip):
138 continue
139 found.append({"type": "IPv6", "ip": ip})
140
141 return found

153def check_version_leak(data: dict) -> dict:
154 """Check if response leaks version/OS information."""
155 leaks = []
156 text = json.dumps(data).lower()
157
158 # Check for OS/platform info
159 os_patterns = ["windows", "linux", "darwin", "macos", "freebsd",
160 "x86_64", "amd64", "arm64", "aarch64"]
161 for pat in os_patterns:
162 if pat in text:
163 leaks.append(f"OS info leaked: {pat}")
164
165 # Check for version details
166 version_pattern = re.compile(r'"version"\s*:\s*"([^"]+)"')
167 for match in version_pattern.finditer(json.dumps(data)):
168 leaks.append(f"Version string: {match.group(1)}")
169
170 # Check for user-agent
171 ua_pattern = re.compile(r'"(?:user.?agent|subversion)"\s*:\s*"([^"]+)"', re.I)
172 for match in ua_pattern.finditer(json.dumps(data)):
173 leaks.append(f"User-Agent: {match.group(1)}")
174
175 return {
176 "leaks_found": len(leaks),
177 "details": leaks,
178 "passed": len(leaks) == 0,
179 }
