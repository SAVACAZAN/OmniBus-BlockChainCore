14inline QString satToOmni(qint64 sat) {
15 bool negative = sat < 0;
16 qint64 abs = negative ? -sat : sat;
17 qint64 whole = abs / SAT_PER_OMNI;
18 qint64 frac = abs % SAT_PER_OMNI;
19 QString s = QString("%1%2.%3")
20 .arg(negative ? "-" : "")
21 .arg(whole)
22 .arg(frac, 9, 10, QChar('0'));
23 // Trim trailing zeros but keep at least 4 decimal places
24 while (s.endsWith('0') && s.indexOf('.') < s.length() - 5)
25 s.chop(1);
26 return s;
27}
