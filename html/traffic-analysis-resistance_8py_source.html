40def build_p2p_message(msg_type: str, payload_size: int) -> bytes:
41 """Build an OmniBus-format P2P message."""
42 magic = OMNIBUS_MAGIC
43 command = msg_type.encode().ljust(12, b"\x00")[:12]
44 payload = secrets.token_bytes(payload_size)
45 length = struct.pack("<I", len(payload))
46 # Fake checksum (first 4 bytes of double-SHA256)
47 import hashlib
48 checksum = hashlib.sha256(hashlib.sha256(payload).digest()).digest()[:4]
49 return magic + command + length + checksum + payload

52def pearson_correlation(x: list, y: list) -> float:
53 """Compute Pearson correlation coefficient."""
54 n = len(x)
55 if n < 2:
56 return 0.0
57
58 mean_x = sum(x) / n
59 mean_y = sum(y) / n
60
61 cov = sum((x[i] - mean_x) * (y[i] - mean_y) for i in range(n))
62 std_x = math.sqrt(sum((xi - mean_x) ** 2 for xi in x))
63 std_y = math.sqrt(sum((yi - mean_y) ** 2 for yi in y))
64
65 if std_x == 0 or std_y == 0:
66 return 0.0
67
68 return cov / (std_x * std_y)

71def coefficient_of_variation(values: list) -> float:
72 """CV = std_dev / mean. High CV = high variance (good for privacy)."""
73 if not values:
74 return 0.0
75 mean = sum(values) / len(values)
76 if mean == 0:
77 return 0.0
78 variance = sum((v - mean) ** 2 for v in values) / len(values)
79 return math.sqrt(variance) / mean

82def analyze_padding(sizes: list) -> dict:
83 """Analyze if message sizes show padding patterns."""
84 if not sizes:
85 return {"padded": False, "note": "no data"}
86
87 unique_sizes = set(sizes)
88 # Good padding: few distinct sizes (messages padded to fixed blocks)
89 # Bad: many distinct sizes correlating with content
90
91 # Check if sizes are multiples of common block sizes
92 block_sizes = [16, 32, 64, 128, 256, 512, 1024]
93 alignment_scores = {}
94 for bs in block_sizes:
95 aligned = sum(1 for s in sizes if s % bs == 0)
96 alignment_scores[bs] = aligned / len(sizes)
97
98 best_alignment = max(alignment_scores.items(), key=lambda x: x[1])
99
100 is_padded = best_alignment[1] > 0.8 or len(unique_sizes) <= 5
101
102 return {
103 "unique_sizes": len(unique_sizes),
104 "total_messages": len(sizes),
105 "size_min": min(sizes),
106 "size_max": max(sizes),
107 "size_mean": round(sum(sizes) / len(sizes), 1),
108 "cv": round(coefficient_of_variation(sizes), 4),
109 "best_alignment_block": best_alignment[0],
110 "alignment_ratio": round(best_alignment[1], 3),
111 "padded": is_padded,
112 "privacy_assessment": "GOOD — sizes appear padded" if is_padded
113 else "WEAK — predictable size distribution",
114 }

117def run_traffic_analysis(host: str, port: int, message_count: int) -> dict:
118 """Send messages and record timing/size data."""
119 send_times = []
120 recv_times = []
121 send_sizes = []
122 recv_sizes = []
123 errors = 0
124
125 # Message types to send
126 msg_types = ["ping", "getaddr", "inv", "getdata", "getblocks", "version"]
127
128 for i in range(message_count):
129 try:
130 sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
131 sock.settimeout(5)
132 sock.connect((host, port))
133
134 # Random message
135 msg_type = random.choice(msg_types)
136 payload_size = random.randint(20, 500)
137 message = build_p2p_message(msg_type, payload_size)
138
139 # Random delay (0-200ms)
140 delay = random.uniform(0.001, 0.200)
141 time.sleep(delay)
142
143 # Send
144 t_send = time.time()
145 sock.sendall(message)
146 send_times.append(t_send)
147 send_sizes.append(len(message))
148
149 # Receive (with timeout)
150 try:
151 sock.settimeout(2)
152 response = sock.recv(4096)
153 t_recv = time.time()
154 recv_times.append(t_recv)
155 recv_sizes.append(len(response))
156 except socket.timeout:
157 recv_times.append(time.time())
158 recv_sizes.append(0)
159
160 sock.close()
161
162 except Exception:
163 errors += 1
164 send_times.append(time.time())
165 recv_times.append(time.time())
166 send_sizes.append(0)
167 recv_sizes.append(0)
168
169 if (i + 1) % 25 == 0:
170 print(f" {CYAN}Sent {i+1}/{message_count} messages ...{RESET}")
171
172 return {
173 "send_times": send_times,
174 "recv_times": recv_times,
175 "send_sizes": send_sizes,
176 "recv_sizes": recv_sizes,
177 "errors": errors,
178 }
