64 def parse_commits(self, since: Optional[str] = None) -> List[Change]:
65 """Parse git commits for conventional commit format."""
66 cmd = ["git", "log", "--pretty=format:%H|%s|%an"]
67 if since:
68 cmd.extend([f"{since}..HEAD"])
69 else:
70 cmd.extend(["-30"]) # Last 30 commits
71
72 try:
73 result = subprocess.run(
74 cmd,
75 cwd=self.root,
76 capture_output=True,
77 text=True,
78 timeout=30
79 )
80
81 changes = []
82 for line in result.stdout.strip().split('\n'):
83 if '|' not in line:
84 continue
85
86 parts = line.split('|')
87 if len(parts) < 3:
88 continue
89
90 commit_hash, message, author = parts[0], parts[1], parts[2]
91
92 # Parse conventional commit format: type(scope): message
93 match = re.match(r'^(\w+)(?:\‍(([^)]+)\‍))?\s*:\s*(.+)$', message)
94 if match:
95 commit_type, scope, msg = match.groups()
96 category = self._map_commit_type(commit_type)
97
98 changes.append(Change(
99 category=category,
100 message=msg,
101 commit_hash=commit_hash[:7],
102 author=author,
103 scope=scope or ""
104 ))
105
106 return changes
107
108 except subprocess.CalledProcessError as e:
109 print(f"Error running git: {e}")
110 return []
111 except FileNotFoundError:
112 print("Git not found in PATH")
113 return []

136 def suggest_version_bump(self, changes: List[Change]) -> Tuple[str, str]:
137 """Suggest version bump based on changes."""
138 has_breaking = any("breaking" in c.message.lower() or "BREAKING" in c.message for c in changes)
139 has_features = any(c.category == "added" for c in changes)
140 has_fixes = any(c.category == "fixed" for c in changes)
141
142 if has_breaking:
143 return "major", "Breaking changes detected - suggest MAJOR version bump"
144 elif has_features:
145 return "minor", "New features detected - suggest MINOR version bump"
146 elif has_fixes:
147 return "patch", "Bug fixes detected - suggest PATCH version bump"
148 else:
149 return "none", "No significant changes - version can remain"

151 def generate_changelog(self, changes: List[Change], version: str = "Unreleased") -> str:
152 """Generate changelog content."""
153 # Group by category
154 by_category = defaultdict(list)
155 for change in changes:
156 by_category[change.category].append(change)
157
158 lines = [

166 f"## [{version}] - {datetime.now().strftime('%Y-%m-%d')}",
167 "",
168 ]
169
170 # Add categories in order
171 for cat_key, cat_name in self.CATEGORIES.items():
172 if cat_key in by_category:
173 lines.append(f"### {cat_name}")
174 lines.append("")
175 for change in by_category[cat_key]:
176 scope = f"**{change.scope}**: " if change.scope else ""
177 lines.append(f"- {scope}{change.message} ({change.commit_hash})")
178 lines.append("")

182 def generate_release_notes(self, changes: List[Change], version: str) -> str:

231 def bump_version(self, current: str, bump_type: str) -> str:
232 """Bump version according to semver."""
233 # Remove 'v' prefix if present
234 version = current.lstrip('v')
235 parts = version.split('.')
236
237 if len(parts) != 3:
238 parts = ['0', '0', '0']
239
240 major, minor, patch = int(parts[0]), int(parts[1]), int(parts[2])
241
242 if bump_type == "major":
243 return f"v{major + 1}.0.0"
244 elif bump_type == "minor":
245 return f"v{major}.{minor + 1}.0"
246 elif bump_type == "patch":
247 return f"v{major}.{minor}.{patch + 1}"
248 else:
249 return current
