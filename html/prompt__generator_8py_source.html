108 def get_relevant_todos(self, keyword: Optional[str] = None) -> List[str]:
109 """Ia TODO-uri relevante."""

117 with open(filepath, 'r', encoding='utf-8') as f:

120 rel_path = str(filepath.relative_to(self.project_root))
121
122 for i, line in enumerate(lines, 1):
123 if 'TODO' in line or 'FIXME' in line or 'BUG' in line:
124 if keyword and keyword.lower() not in line.lower():

126
127 todo_text = line.strip()
128 if len(todo_text) > 100:
129 todo_text = todo_text[:97] + "..."

131 todos.append(f"{rel_path}:{i}: {todo_text}")
