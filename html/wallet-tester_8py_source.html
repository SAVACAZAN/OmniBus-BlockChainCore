49def derive_child(parent_key: bytes, parent_chain: bytes, index: int, hardened: bool) -> Tuple[bytes, bytes]:
50 # Simplified BIP-32 child derivation using HMAC-SHA512
51 if hardened:
52 # BIP-32: hardened child uses 0x80000000 + index
53 data = b"\x00" + parent_key + (0x80000000 + index).to_bytes(4, "big")
54 else:
55 # pubkey placeholder (simplified: use raw key bytes)
56 data = parent_key + index.to_bytes(4, "big")
57 h = hmac.new(parent_chain, data, hashlib.sha512).digest()
