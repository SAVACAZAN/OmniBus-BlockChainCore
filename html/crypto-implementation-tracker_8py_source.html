61def get_crypto_commits(cwd: str, max_commits: int = 300) -> list[dict[str, str]]:
62 fmt = "%H|%aI|%s"
63 # commits touching crypto-related files
64 paths = ["core/secp256k1.zig", "core/crypto.zig", "core/bip32_wallet.zig", "core/schnorr.zig"]
65 out = run_git(["log", f"--format={fmt}", f"-n{max_commits}", "--"] + paths, cwd=cwd)
66 commits: list[dict[str, str]] = []
67 seen = set()
68 for line in out.strip().splitlines():
69 if "|" not in line:
70 continue
71 parts = line.split("|", 2)
72 h = parts[0]
73 if h in seen:
74 continue
75 seen.add(h)
76 commits.append({"hash": h, "date": parts[1], "message": parts[2]})
77 return commits

85def get_file_diff(cwd: str, commit_hash: str, file_path: str) -> str:
86 try:
87 parent = run_git(["rev-list", "--parents", "-n1", commit_hash], cwd=cwd).strip().split()[1]
88 except Exception:
89 parent = f"{commit_hash}^"

103def classify_commit(message: str) -> list[str]:
104 tags: list[str] = []
105 m = message.lower()
106 if any(w in m for w in ["fix", "bug", "patch", "repair"]):
107 tags.append("bugfix")
108 if any(w in m for w in ["test", "spec", "vector"]):
109 tags.append("test-related")
110 if any(w in m for w in ["perf", "optim", "speed", "fast"]):
111 tags.append("performance")
112 if any(w in m for w in ["refactor", "clean", "restructure"]):
113 tags.append("refactor")
114 if any(w in m for w in ["feat", "add", "introduce", "implement", "new"]):
115 tags.append("feature")
116 if not tags:
117 tags.append("misc")
118 return tags

121def extract_bip32_paths(source: str) -> list[str]:
122 """Find BIP-32 derivation path strings in source code."""
123 return list(dict.fromkeys(re.findall(r"m/[\d'h/]+", source)))

126def extract_secp256k1_functions(source: str) -> list[str]:
127 """Find public function names in secp256k1 source."""
128 return re.findall(r"\bpub\s+fn\s+([\w_]+)", source)

131def extract_test_vectors(source: str) -> list[str]:
132 """Find hard-coded hex test vectors (>= 32 hex chars)."""
133 return list(dict.fromkeys(re.findall(r"[0-9a-fA-F]{64,}", source)))

136def analyze_crypto_evolution(cwd: str, max_commits: int) -> dict[str, Any]:
137 log_info("Fetching crypto-related commits …")
138 commits = get_crypto_commits(cwd, max_commits)
139 log_info(f"Unique crypto commits: {len(commits)}")
140
141 secp_timeline: list[dict[str, Any]] = []
142 bip32_timeline: list[dict[str, Any]] = []
143 crypto_api_changes: list[dict[str, Any]] = []
144
145 secp_funcs_over_time: dict[str, list[str]] = {}
146 bip32_paths_over_time: dict[str, list[str]] = {}
147
148 for idx, commit in enumerate(commits):
149 changed = get_changed_files(cwd, commit["hash"])
150 tags = classify_commit(commit["message"])
151
152 # secp256k1 tracking
153 if "core/secp256k1.zig" in changed:
154 src = get_file_at_commit(cwd, commit["hash"], "core/secp256k1.zig")
155 diff = get_file_diff(cwd, commit["hash"], "core/secp256k1.zig")
156 funcs = extract_secp256k1_functions(src)
157 vectors = extract_test_vectors(src)
158 secp_timeline.append(
159 {
160 "commit": commit["hash"],
161 "date": commit["date"],
162 "message": commit["message"],
163 "tags": tags,
164 "functions": funcs,
165 "function_count": len(funcs),
166 "test_vectors_count": len(vectors),
167 "diff_lines": len(diff.splitlines()),
168 }
169 )
170 secp_funcs_over_time[commit["hash"]] = funcs
171
172 # BIP-32 tracking
173 if "core/bip32_wallet.zig" in changed:
174 src = get_file_at_commit(cwd, commit["hash"], "core/bip32_wallet.zig")
175 diff = get_file_diff(cwd, commit["hash"], "core/bip32_wallet.zig")
176 paths = extract_bip32_paths(src)
177 bip32_timeline.append(
178 {
179 "commit": commit["hash"],
180 "date": commit["date"],
181 "message": commit["message"],
182 "tags": tags,
183 "derivation_paths": paths,
184 "path_count": len(paths),
185 "diff_lines": len(diff.splitlines()),
186 }
187 )
188 bip32_paths_over_time[commit["hash"]] = paths
189
190 # General crypto API surface
191 crypto_files = [c for c in changed if "crypto" in c.lower() and c.endswith(".zig")]
192 for cf in crypto_files:
193 src = get_file_at_commit(cwd, commit["hash"], cf)
194 funcs = extract_secp256k1_functions(src)
195 crypto_api_changes.append(
196 {
197 "commit": commit["hash"],
198 "date": commit["date"],
199 "file": cf,
200 "message": commit["message"],
201 "tags": tags,
202 "public_functions": funcs,
203 }
204 )
205
206 if (idx + 1) % 20 == 0 or idx == len(commits) - 1:
207 log_info(f" processed {idx + 1}/{len(commits)} commits")
208
209 # Summaries
210 bugfix_count = sum(1 for e in secp_timeline if "bugfix" in e["tags"])
211 test_change_count = sum(1 for e in secp_timeline if "test-related" in e["tags"])
212
213 report = {
214 "generated_at": datetime.now(timezone.utc).isoformat(),
215 "commits_scanned": len(commits),
216 "summary": {
217 "secp256k1_commits": len(secp_timeline),
218 "secp256k1_bugfixes": bugfix_count,
219 "secp256k1_test_changes": test_change_count,
220 "bip32_commits": len(bip32_timeline),
221 "crypto_api_changes": len(crypto_api_changes),
222 },
223 "secp256k1_timeline": secp_timeline,
224 "bip32_timeline": bip32_timeline,
225 "crypto_api_changes": crypto_api_changes,
226 }
227 return report
