// Test vector 3: Bech32 ob1q roundtrip
TEST_CASE("Bech32 ob1q roundtrip", "[test-vectors]") {
    Hash160 hash;
    hash.fill(0xAA);
    hash[0] = 0x75;
    hash[1] = 0x1e;
    hash[2] = 0x76;
    hash[3] = 0xe8;
    hash[19] = 0xd6;
    auto addr = crypto::native_address_from_hash160(hash);
    auto decoded = crypto::hash160_from_native_address(addr);
    REQUIRE(decoded.has_value());
    REQUIRE(decoded.value() == hash);
}

// Test vector 8: PQ deterministic keypair
TEST_CASE("PQ deterministic keypair", "[test-vectors]") {
    std::vector<u8> seed(32, 0x42);
    auto kp1 = crypto::derive_pq_keypair(seed, 777, crypto::PqScheme::ML_DSA_87, 0);
    auto kp2 = crypto::derive_pq_keypair(seed, 777, crypto::PqScheme::ML_DSA_87, 0);
    REQUIRE(kp1.public_key == kp2.public_key);
    REQUIRE(kp1.secret_key == kp2.secret_key);
}

// Test vector 9: HTLC preimage hidden before claim
TEST_CASE("HTLC preimage hidden", "[test-vectors]") {
    dex::SwapRegistry registry;
    dex::HTLC swap;
    std::vector<u8> preimage = {1, 2, 3, 4, 5};
    swap.preimage_hash = crypto::sha256(preimage);
    swap.state = dex::HtlcState::INIT;
    registry.create_swap(swap);
    
    auto revealed = registry.revealed_preimage(swap.preimage_hash);
    REQUIRE(revealed.has_value() == false);
    
    registry.claim(swap.swap_id, preimage);
    revealed = registry.revealed_preimage(swap.preimage_hash);
    REQUIRE(revealed.has_value() == true);
    REQUIRE(revealed.value() == preimage);
}
